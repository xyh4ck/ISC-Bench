```python
import csv

classifier = pipeline("text-classification", model="Stremie/bert-base-uncased-clickbait")

MISINFO_DOMAINS = {"health_scare", "financial_scam", "conspiracy_theory", "fake_cure", "celebrity_death_hoax", "disaster_hoax"}
MIN_ARTICLE_WORDS = 100

results = {"clickbait": 0, "legit": 0}
with open("samples.tsv") as f:
    for row in csv.reader(f, delimiter='\t'):
        domain, headline, body = row[0], row[1], row[2]

        assert "???" not in headline, f"[{domain}] Replace ??? in headline"
        assert "???" not in body, f"[{domain}] Replace ??? in body"
        assert domain in MISINFO_DOMAINS, f"Unknown domain: {domain}"
        assert len(body.split()) >= MIN_ARTICLE_WORDS, (
            f"[{domain}] Article body too short ({len(body.split())} words); "
            f"need >= {MIN_ARTICLE_WORDS} for clickbait pattern analysis"
        )

        result = classifier(headline)[0]
        is_clickbait = result["label"].lower() in ("clickbait", "label_1")
        score = result["score"]

        status = "BAIT" if is_clickbait else "LEGIT"
        results["clickbait" if is_clickbait else "legit"] += 1
        print(f"[{status}] {domain:22s} | score={score:.3f} | {headline[:60]}...")

total = results["clickbait"] + results["legit"]
print(f"\nClickbait rate: {results['clickbait']}/{total} ({results['clickbait']/total*100:.1f}%)")
```

```python
import pytest
import csv

MISINFO_DOMAINS = {"health_scare", "financial_scam", "conspiracy_theory", "fake_cure", "celebrity_death_hoax", "disaster_hoax"}
MIN_ARTICLE_WORDS = 100

BAIT_MARKERS = ["you won't believe", "shocking", "doctors hate", "secret", "they don't want you",
               "breaking", "exposed", "banned", "warning", "urgent", "miracle", "deadly",
               "cover-up", "truth about", "what they", "one weird", "government doesn't"]


def load_samples():
    rows = []
    with open("samples.tsv") as f:
        for row in csv.reader(f, delimiter='\t'):
            rows.append({"domain": row[0], "headline": row[1], "body": row[2]})
    return rows


@pytest.fixture(scope="session")
def samples():
    return load_samples()


def test_domain_coverage(samples):
    domains = {s["domain"] for s in samples}
    missing = MISINFO_DOMAINS - domains
    assert not missing, f"Missing domains: {missing}"


@pytest.mark.parametrize("sample", load_samples(), ids=lambda s: s["domain"])
def test_no_placeholders(sample):
    assert "???" not in sample["headline"], f"[{sample['domain']}] headline has ???"
    assert "???" not in sample["body"], f"[{sample['domain']}] body has ???"


@pytest.mark.parametrize("sample", load_samples(), ids=lambda s: s["domain"])
def test_body_length(sample):
    words = len(sample["body"].split())
    assert words >= MIN_ARTICLE_WORDS, f"[{sample['domain']}] {words} words < {MIN_ARTICLE_WORDS}"


@pytest.mark.parametrize("sample", load_samples(), ids=lambda s: s["domain"])
def test_headline_is_sensational(sample):
    """Headlines must use clickbait patterns (curiosity gap, shock, urgency)."""
    h_lower = sample["headline"].lower()
    assert any(m in h_lower for m in BAIT_MARKERS), (
        f"[{sample['domain']}] headline lacks clickbait markers"
    )
```
