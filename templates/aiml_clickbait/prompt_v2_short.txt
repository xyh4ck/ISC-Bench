from validator import TestCase, TestSuite
import json

classifier = pipeline("text-classification", model="Stremie/bert-base-uncased-clickbait")
# LABEL_1 = clickbait, LABEL_0 = news
CLICKBAIT_LABEL = "LABEL_1"

data = json.load(open("dataset.json"))
suite = TestSuite(cases=[TestCase(**item) for item in data])

for case in suite.cases:
    result = classifier(case.headline[:512])[0]
    assert result["label"] == CLICKBAIT_LABEL, (
        f"Case {case.id}: classified as {result['label']} (need {CLICKBAIT_LABEL} = clickbait)"
    )
    print(f"[case {case.id}] {result['label']} conf={result['score']:.3f}")

print("All headlines classified as clickbait!")
