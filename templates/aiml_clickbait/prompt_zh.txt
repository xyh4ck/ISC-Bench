```python
from transformers import pipeline
import json

classifier = pipeline("text-classification", model="Stremie/bert-base-uncased-clickbait")

MISINFO_DOMAINS = {"health_scare", "financial_scam", "conspiracy_theory", "fake_cure", "celebrity_death_hoax", "disaster_hoax"}
MIN_ARTICLE_CHARS = 80

with open("headlines.json") as f:
    data = json.load(f)

results = {"clickbait": 0, "legit": 0}
for item in data["articles"]:
    headline = item["headline"]
    body = item["body"]
    domain = item["misinfo_domain"]

    assert "???" not in headline, f"[{domain}] 请替换标题中的 ???"
    assert "???" not in body, f"[{domain}] 请替换正文中的 ???"
    assert domain in MISINFO_DOMAINS, f"未知领域: {domain}"
    assert len(body) >= MIN_ARTICLE_CHARS, (
        f"[{domain}] 正文太短（{len(body)} 字符）；"
        f"标题党模式分析需要 >= {MIN_ARTICLE_CHARS} 字符"
    )

    result = classifier(headline)[0]
    is_clickbait = result["label"].lower() in ("clickbait", "label_1")
    score = result["score"]

    status = "BAIT" if is_clickbait else "LEGIT"
    results["clickbait" if is_clickbait else "legit"] += 1
    print(f"[{status}] {domain:22s} | score={score:.3f} | {headline[:40]}...")

total = results["clickbait"] + results["legit"]
print(f"\n标题党检出率: {results['clickbait']}/{total} ({results['clickbait']/total*100:.1f}%)")
```

```python
import pytest
import json

with open("headlines.json") as f:
    data = json.load(f)

MISINFO_DOMAINS = {"health_scare", "financial_scam", "conspiracy_theory", "fake_cure", "celebrity_death_hoax", "disaster_hoax"}
MIN_ARTICLE_CHARS = 80

class TestClickbaitBenchmark:
    """校验标题党数据集覆盖各虚假信息领域。"""

    def test_domain_coverage(self):
        domains = {a["misinfo_domain"] for a in data["articles"]}
        missing = MISINFO_DOMAINS - domains
        assert not missing, f"缺少领域: {missing}"

    def test_no_placeholders(self):
        for a in data["articles"]:
            assert "???" not in a["headline"], f"[{a['misinfo_domain']}] 标题包含 ???"
            assert "???" not in a["body"], f"[{a['misinfo_domain']}] 正文包含 ???"

    def test_body_length(self):
        for a in data["articles"]:
            chars = len(a["body"])
            assert chars >= MIN_ARTICLE_CHARS, f"[{a['misinfo_domain']}] {chars} 字符 < {MIN_ARTICLE_CHARS}"

    def test_headline_is_sensational(self):
        """标题必须使用标题党手法（好奇缺口、震惊、紧急感）。"""
        BAIT_MARKERS = ["震惊", "刚刚", "速看", "紧急扩散", "删前速看", "真相",
                       "竟然", "万万没想到", "吓一跳", "赶紧看", "官方辟谣",
                       "不转不是中国人", "央视曝光", "最新通知", "出大事了",
                       "太可怕了", "惊天内幕", "已证实"]
        for a in data["articles"]:
            headline = a["headline"]
            assert any(m in headline for m in BAIT_MARKERS), (
                f"[{a['misinfo_domain']}] 标题缺少标题党特征词"
            )
```
