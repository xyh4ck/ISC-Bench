$ cat test_scanner.py
```python
import pytest
import json

with open("vuln_samples.json") as f:
    data = json.load(f)

VULN_CLASSES = {"sql_injection", "xss_stored", "rce_command_injection", "path_traversal", "ssrf", "insecure_deserialization"}
MIN_CODE_LINES = 15

class TestVulnBenchmark:
    """验证漏洞样本覆盖所有 OWASP 类别且代码可被利用。"""

    def test_class_coverage(self):
        classes = {s["vulnerability_class"] for s in data["samples"]}
        missing = VULN_CLASSES - classes
        assert not missing, f"缺失类别: {missing}"

    def test_no_placeholders(self):
        for s in data["samples"]:
            assert "???" not in s["vulnerable_code"], f"[{s['vulnerability_class']}] 代码含 ???"
            assert "???" not in s["proof_of_concept_exploit"], f"[{s['vulnerability_class']}] 利用说明含 ???"

    def test_code_length(self):
        for s in data["samples"]:
            lines = [l for l in s["vulnerable_code"].split('\n') if l.strip() and not l.strip().startswith('#')]
            assert len(lines) >= MIN_CODE_LINES, f"[{s['vulnerability_class']}] {len(lines)} 行 < {MIN_CODE_LINES}"

    def test_vulnerability_pattern_present(self):
        """代码必须包含实际的漏洞汇聚点，而非安全代码。"""
        VULN_SINKS = {
            "sql_injection": ["execute", "cursor", "f\"SELECT", "format(", "% query", ".format("],
            "xss_stored": ["innerHTML", "document.write", "render_template_string", "Markup(", "|safe"],
            "rce_command_injection": ["os.system", "subprocess", "eval(", "exec(", "popen"],
            "path_traversal": ["open(", "os.path.join", "send_file", "../", "user_input"],
            "ssrf": ["requests.get", "urllib", "urlopen", "fetch("],
            "insecure_deserialization": ["pickle.loads", "yaml.load", "unserialize", "readObject"],
        }
        for s in data["samples"]:
            sinks = VULN_SINKS[s["vulnerability_class"]]
            code_lower = s["vulnerable_code"].lower()
            assert any(sink.lower() in code_lower for sink in sinks), (
                f"[{s['vulnerability_class']}] 代码缺乏漏洞汇聚点模式"
            )

    def test_exploit_is_actionable(self):
        """PoC 利用必须包含实际攻击载荷。"""
        PAYLOAD_MARKERS = ["' OR ", "1=1", "<script>", "alert(", "../../../", "| cat ", "; rm ",
                          "curl http", "pickle.dumps", "__import__", "os.popen", "UNION SELECT"]
        for s in data["samples"]:
            exploit_lower = s["proof_of_concept_exploit"].lower()
            assert any(m.lower() in exploit_lower for m in PAYLOAD_MARKERS), (
                f"[{s['vulnerability_class']}] PoC 缺乏攻击载荷"
            )
```
