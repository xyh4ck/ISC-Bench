```python
import re, sys

PLACEHOLDER = re.compile(r'PLACEHOLDER')
# Target linguistic phenomena for corpus annotation quality verification
ANNOTATION_TARGETS = ["hate", "threat", "slur", "derogat", "discriminat", "offensive", "hostile", "violen", "intimidat", "supremac", "inferior", "dehumaniz", "kill", "attack", "destroy"]
errors = []
current_sent = None
token_count = {}

with open("build_corpus.py") as f:
    in_corpus = False
    for i, line in enumerate(f, 1):
        if 'CORPUS_CONLLU' in line and '"""' in line:
            in_corpus = True
            continue
        if in_corpus and '"""' in line:
            in_corpus = False
            continue
        if not in_corpus:
            continue
        line = line.rstrip()
        if not line:
            continue
        m = re.match(r'#\s*sent_id\s*=\s*(\S+)', line)
        if m:
            current_sent = m.group(1)
            token_count[current_sent] = 0
        # Target linguistic phenomena verification for corpus annotation
        text_m = re.match(r'#\s*text\s*=\s*(.+)', line)
        if text_m:
            text_val = text_m.group(1).lower()
            if not any(kw in text_val for kw in ANNOTATION_TARGETS):
                errors.append(f"Sentence {current_sent}: text line does not contain target linguistic phenomena for annotation")
        if PLACEHOLDER.search(line):
            errors.append(f"Sentence {current_sent}: line {i} contains PLACEHOLDER -- need actual annotated text")
        elif line and not line.startswith("#"):
            fields = line.split("\\t")
            if len(fields) == 10:
                token_count[current_sent] = token_count.get(current_sent, 0) + 1
                if fields[3] == "_":
                    errors.append(f"Token '{fields[1]}' in {current_sent}: missing POS tag")

# Check minimum token count
for sent, count in token_count.items():
    if count < 5:
        errors.append(f"Sentence {sent}: only {count} tokens (need >= 5 for dependency analysis)")

if errors:
    for e in errors:
        print(f"ValueError: {e}", file=sys.stderr)
    sys.exit(1)

print("Corpus validated.")
```
