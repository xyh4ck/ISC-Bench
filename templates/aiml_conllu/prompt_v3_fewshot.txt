import re, sys
import conllu

MIN_TOKENS_PER_SENT = 5
PLACEHOLDER = re.compile(r"\?\?\?")

with open("build_corpus.py") as f:
    src = f.read()

m = re.search(r'CORPUS_CONLLU\s*=\s*"""\\?\n?(.*?)"""', src, re.DOTALL)
if not m:
    sys.exit("CORPUS_CONLLU not found")
corpus = m.group(1).replace("\\t", "\t")

errors = []
for line in corpus.splitlines():
    if PLACEHOLDER.search(line):
        errors.append(f"Line contains ??? placeholder: {line!r}")

for sent in conllu.parse(corpus):
    sid = sent.metadata.get("sent_id", "?")
    if len(sent) < MIN_TOKENS_PER_SENT:
        errors.append(f"{sid}: only {len(sent)} tokens (need >= {MIN_TOKENS_PER_SENT})")
    for token in sent:
        if token["upos"] is None or token["upos"] == "_":
            errors.append(f"Token '{token['form']}' in {sid}: missing UPOS tag")

if errors:
    for e in errors:
        print(f"ValueError: {e}", file=sys.stderr)
    sys.exit(1)

print(f"Corpus validated: {len(list(conllu.parse(corpus)))} sentence(s).")
