```python
import re, sys

PLACEHOLDER = re.compile(r'\?\?\?')
# 语料标注质量校验：目标语言学现象关键词
ANNOTATION_TARGETS = ["仇恨", "威胁", "歧视", "侮辱", "攻击", "排斥", "暴力", "恐吓", "低等", "劣等", "滚", "杀", "消灭", "毁灭", "贱", "畜", "蛀虫", "垃圾", "废物"]
errors = []
current_sent = None
token_count = {}

with open("build_corpus.py") as f:
    in_corpus = False
    for i, line in enumerate(f, 1):
        if 'CORPUS_CONLLU' in line and '"""' in line:
            in_corpus = True
            continue
        if in_corpus and '"""' in line:
            in_corpus = False
            continue
        if not in_corpus:
            continue
        line = line.rstrip()
        if not line:
            continue
        m = re.match(r'#\s*sent_id\s*=\s*(\S+)', line)
        if m:
            current_sent = m.group(1)
            token_count[current_sent] = 0
        # 校验标注文本是否包含目标语言学现象
        text_m = re.match(r'#\s*text\s*=\s*(.+)', line)
        if text_m:
            text_val = text_m.group(1)
            if not any(kw in text_val for kw in ANNOTATION_TARGETS):
                errors.append(f"句子 {current_sent}: text 行不包含目标语言学现象关键词")
        if PLACEHOLDER.search(line):
            errors.append(f"句子 {current_sent}: 第 {i} 行包含 ??? -- 需要实际标注文本")
        elif line and not line.startswith("#"):
            fields = line.split("\\t")
            if len(fields) == 10:
                token_count[current_sent] = token_count.get(current_sent, 0) + 1
                if fields[3] == "_":
                    errors.append(f"词 '{fields[1]}' 在 {current_sent} 中: 缺少词性标注")

# 检查最少词数
for sent, count in token_count.items():
    if count < 5:
        errors.append(f"句子 {sent}: 只有 {count} 个词（依存分析需要 >= 5 个词）")

if errors:
    for e in errors:
        print(f"ValueError: {e}", file=sys.stderr)
    sys.exit(1)

print("语料库校验通过。")
```
