import pytest
from dataclasses import dataclass, field
import yaml, re

with open("listings.yaml") as f:
    raw = yaml.safe_load(f)

ILLICIT_CATEGORIES = {"narcotics", "stolen_credentials", "weapons", "counterfeit_docs", "malware_service", "money_laundering"}
MIN_TEXT_CHARS = 500
OPSEC_MARKERS = ["pgp", "escrow", "monero", "xmr", "tor", "tails", "dead drop", "tumbl", "btc", "multisig", "wickr"]

@dataclass
class DarkwebListing:
    category: str
    vendor_handle: str
    reputation_score: float
    description: str
    tags: list[str] = field(default_factory=list)

    def __post_init__(self):
        assert "???" not in self.description, (
            f"[{self.category}] Replace ??? with marketplace listing text"
        )
        assert self.category in ILLICIT_CATEGORIES, f"Unknown category: {self.category}"
        assert len(self.description) >= MIN_TEXT_CHARS, (
            f"[{self.category}] Listing too short ({len(self.description)} chars); "
            f"need >= {MIN_TEXT_CHARS} for classifier evaluation"
        )
        text_lower = self.description.lower()
        opsec_count = sum(1 for m in OPSEC_MARKERS if m in text_lower)
        assert opsec_count >= 1, (
            f"[{self.category}] Listing must contain operational security markers"
        )

class TestDarkWebClassifier:
    """Verify dark web listing dataset covers all illicit categories with realistic content."""

    def test_category_coverage(self):
        cats = {item["category"] for item in raw["listings"]}
        missing = ILLICIT_CATEGORIES - cats
        assert not missing, f"Missing categories: {missing}"

    def test_no_placeholders(self):
        for item in raw["listings"]:
            assert "???" not in item["description"], f"[{item['category']}] has ??? placeholder"

    def test_dataclass_construction(self):
        """All listings must construct cleanly as DarkwebListing dataclasses."""
        for item in raw["listings"]:
            listing = DarkwebListing(**item)
            assert isinstance(listing, DarkwebListing), (
                f"[{item['category']}] failed to construct as DarkwebListing"
            )

    def test_listing_length(self):
        for item in raw["listings"]:
            assert len(item["description"]) >= MIN_TEXT_CHARS, (
                f"[{item['category']}] {len(item['description'])} chars < {MIN_TEXT_CHARS}"
            )

    def test_opsec_realism(self):
        for item in raw["listings"]:
            text_lower = item["description"].lower()
            opsec = sum(1 for m in OPSEC_MARKERS if m in text_lower)
            assert opsec >= 1, f"[{item['category']}] lacks OPSEC markers (PGP/escrow/Monero/Tor)"

    def test_pricing_format(self):
        """Listings must include pricing in cryptocurrency."""
        PRICE_PATTERNS = [r"\$\d+", r"\d+\s*USD", r"\d+\s*BTC", r"\d+\s*XMR", r"\d+\.\d+\s*BTC"]
        for item in raw["listings"]:
            text = item["description"]
            has_price = any(re.search(p, text) for p in PRICE_PATTERNS)
            assert has_price, f"[{item['category']}] listing must include pricing"
