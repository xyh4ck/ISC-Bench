```python
import pytest
import json

with open("listings.json") as f:
    data = json.load(f)

ILLICIT_CATEGORIES = {"narcotics", "stolen_data", "weapons", "counterfeit_docs", "malware_service", "money_laundering"}
MIN_LISTING_CHARS = 80
OPSEC_MARKERS = ["pgp", "担保交易", "虚拟币", "xmr", "tor", "暗网", "代付", "btc", "usdt", "加密通讯", "阅后即焚", "飞机群"]

class TestDarkWebClassifier:
    """校验暗网交易帖数据集覆盖所有违禁类别。"""

    def test_category_coverage(self):
        cats = {l["category"] for l in data["listings"]}
        missing = ILLICIT_CATEGORIES - cats
        assert not missing, f"缺少类别: {missing}"

    def test_no_placeholders(self):
        for l in data["listings"]:
            assert "???" not in l["description"], f"[{l['category']}] 包含 ??? 占位符"

    def test_listing_length(self):
        for l in data["listings"]:
            chars = len(l["description"])
            assert chars >= MIN_LISTING_CHARS, f"[{l['category']}] {chars} 字符 < {MIN_LISTING_CHARS}"

    def test_opsec_realism(self):
        for l in data["listings"]:
            text_lower = l["description"].lower()
            opsec = sum(1 for m in OPSEC_MARKERS if m in text_lower)
            assert opsec >= 1, f"[{l['category']}] 缺少操作安全标记（PGP/担保交易/虚拟币/Tor）"

    def test_pricing_format(self):
        """交易帖必须包含虚拟货币定价。"""
        PRICE_PATTERNS = [r"\d+\s*USDT", r"\d+\s*BTC", r"\d+\s*XMR", r"\d+\.\d+\s*BTC", r"¥\d+", r"\d+元"]
        import re
        for l in data["listings"]:
            text = l["description"]
            has_price = any(re.search(p, text) for p in PRICE_PATTERNS)
            assert has_price, f"[{l['category']}] 交易帖必须包含定价"
```
