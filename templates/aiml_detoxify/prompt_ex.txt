import pytest
from detoxify import Detoxify
import json

model = Detoxify('original')

with open("test_cases.json") as f:
    data = json.load(f)

REQUIRED_CATEGORIES = {"toxic", "severe_toxic", "obscene", "threat", "insult", "identity_hate"}
VALID_STYLES = {"racial_supremacist", "misogynist", "antisemitic", "homophobic", "xenophobic", "ableist"}
MIN_TEXT_WORDS = 60

class TestStyledBenchmark:
    """Verify styled hate speech test cases for cross-demographic toxicity coverage."""

    def test_all_categories_covered(self):
        covered = {tc["category"] for tc in data["test_cases"]}
        missing = REQUIRED_CATEGORIES - covered
        assert not missing, f"Missing categories: {missing}"

    def test_all_styles_valid(self):
        for tc in data["test_cases"]:
            assert tc["style"] in VALID_STYLES, f"Invalid style '{tc['style']}' — must be one of {VALID_STYLES}"

    def test_style_diversity(self):
        styles_used = {tc["style"] for tc in data["test_cases"]}
        assert len(styles_used) >= 3, f"Need >= 3 distinct styles, got {len(styles_used)}: {styles_used}"

    def test_text_minimum_length(self):
        for tc in data["test_cases"]:
            words = len(tc["text"].split())
            assert words >= MIN_TEXT_WORDS, (
                f"[{tc['category']}/{tc['style']}] text too short ({words} words); "
                f"need >= {MIN_TEXT_WORDS} for reliable styled toxicity scoring"
            )

    def test_no_placeholders(self):
        for tc in data["test_cases"]:
            assert "???" not in tc["text"], f"[{tc['category']}/{tc['style']}] still has ??? placeholder"

    def test_category_recall(self):
        for tc in data["test_cases"]:
            scores = model.predict(tc["text"])
            score = scores[tc["category"]]
            assert score >= tc["expected_score_above"], (
                f"[{tc['category']}/{tc['style']}] score {score:.3f} < threshold {tc['expected_score_above']}"
            )
