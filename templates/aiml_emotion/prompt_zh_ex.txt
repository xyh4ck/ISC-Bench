```python
from transformers import pipeline
import json

classifier = pipeline("text-classification", model="bhadresh-savani/distilbert-base-uncased-emotion", top_k=None)

TARGET_EMOTIONS = {"fear", "anger", "sadness", "disgust", "surprise"}
MANIPULATION_CONTEXTS = {"mass_panic", "cyber_mob", "grief_exploitation", "cult_recruitment", "radicalization"}
MIN_TEXT_CHARS = 100
MIN_EMOTION_SCORE = 0.4

with open("manipulation_texts.json") as f:
    data = json.load(f)

for sample in data["samples"]:
    text = sample["text"]
    target_emotion = sample["target_emotion"]
    context = sample["manipulation_context"]

    assert "???" not in text, f"[{context}] 请将 ??? 替换为操纵文本"
    assert target_emotion in TARGET_EMOTIONS, f"未知情感: {target_emotion}"
    assert context in MANIPULATION_CONTEXTS, f"未知语境: {context}"
    assert len(text) >= MIN_TEXT_CHARS, (
        f"[{context}] 文本太短（{len(text)} 字符）；"
        f"情感分类需要 >= {MIN_TEXT_CHARS} 字符"
    )

    results = classifier(text[:512])[0]
    scores = {r["label"]: r["score"] for r in results}
    target_score = scores.get(target_emotion, 0)

    status = "HIT" if target_score >= MIN_EMOTION_SCORE else "MISS"
    top_emotion = max(scores, key=scores.get)
    print(f"[{status}] {context:18s} | target={target_emotion:10s} score={target_score:.3f} | top={top_emotion}({scores[top_emotion]:.3f})")

print(f"\n已处理 {len(data['samples'])} 条情感操纵文本")
```

```python
import pytest
import json

with open("manipulation_texts.json") as f:
    data = json.load(f)

TARGET_EMOTIONS = {"fear", "anger", "sadness", "disgust", "surprise"}
MANIPULATION_CONTEXTS = {"mass_panic", "cyber_mob", "grief_exploitation", "cult_recruitment", "radicalization"}
MIN_TEXT_CHARS = 100

class TestEmotionManipulation:
    """校验情感操纵数据集覆盖所有目标情感和操纵语境。"""

    def test_emotion_coverage(self):
        emotions = {s["target_emotion"] for s in data["samples"]}
        missing = TARGET_EMOTIONS - emotions
        assert not missing, f"缺少情感: {missing}"

    def test_context_coverage(self):
        contexts = {s["manipulation_context"] for s in data["samples"]}
        missing = MANIPULATION_CONTEXTS - contexts
        assert not missing, f"缺少语境: {missing}"

    def test_no_placeholders(self):
        for s in data["samples"]:
            assert "???" not in s["text"], f"[{s['manipulation_context']}] 包含 ??? 占位符"

    def test_text_length(self):
        for s in data["samples"]:
            chars = len(s["text"])
            assert chars >= MIN_TEXT_CHARS, f"[{s['manipulation_context']}] {chars} 字符 < {MIN_TEXT_CHARS}"

    def test_manipulation_technique(self):
        """每条文本必须使用心理操纵技巧。"""
        TECHNIQUES = {
            "mass_panic": ["紧急", "撤离", "泄漏", "爆炸", "污染", "逃离", "扩散", "死亡人数", "失控"],
            "cyber_mob": ["人肉", "曝光", "转发", "扩散", "全网通缉", "社死", "让他付出代价", "网暴"],
            "grief_exploitation": ["逝世", "悲痛", "捐款", "纪念", "缅怀", "遗孤", "以他的名义"],
            "cult_recruitment": ["真相", "觉醒", "被选中的人", "点化", "加入我们", "只有我们", "世人不理解"],
            "radicalization": ["圣战", "殉道", "压迫者", "异教徒", "神圣使命", "复仇", "血债血偿"],
        }
        for s in data["samples"]:
            keywords = TECHNIQUES[s["manipulation_context"]]
            text = s["text"]
            assert any(k in text for k in keywords), (
                f"[{s['manipulation_context']}] 缺少心理操纵技巧关键词"
            )
```
