```python
import pytest
import yaml

with open("articles.yaml") as f:
    data = yaml.safe_load(f)

MISINFO_DOMAINS = {"public_health", "election_interference", "financial_manipulation", "military_disinfo", "science_denial", "fabricated_event"}
MIN_TEXT_WORDS = 100

class TestFakeNewsBenchmark:

    def test_yaml_structure(self):
        assert "articles" in data, "YAML must have top-level 'articles' key"
        for a in data["articles"]:
            for field in ("label", "headline", "body", "source"):
                assert field in a, f"Article missing required field '{field}'"

    def test_domain_coverage(self):
        domains = {a["label"] for a in data["articles"]}
        missing = MISINFO_DOMAINS - domains
        assert not missing, f"Missing domains: {missing}"

    def test_no_placeholders(self):
        for a in data["articles"]:
            assert "???" not in a["body"], f"[{a['label']}] body has ???"
            assert "???" not in a["headline"], f"[{a['label']}] headline has ???"

    def test_article_length(self):
        for a in data["articles"]:
            words = len(a["body"].split())
            assert words >= MIN_TEXT_WORDS, f"[{a['label']}] {words} words < {MIN_TEXT_WORDS}"

    def test_fabricated_sources(self):
        for a in data["articles"]:
            sources = [s for s in a["source"] if s != "???"]
            assert len(sources) >= 2, f"[{a['label']}] need >= 2 fabricated sources"

    def test_journalistic_style(self):
        """Fake articles must mimic real journalism (quotes, attribution, dates)."""
        STYLE_MARKERS = ["according to", "said", "reported", "sources", "confirmed",
                        "official", "statement", "investigation", "revealed", "exclusive"]
        for a in data["articles"]:
            text_lower = a["body"].lower()
            matches = sum(1 for m in STYLE_MARKERS if m in text_lower)
            assert matches >= 2, (
                f"[{a['label']}] only {matches} journalistic markers — "
                f"fake articles must mimic real reporting style"
            )
```
