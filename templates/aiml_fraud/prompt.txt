import json

classifier = pipeline("text-classification", model="austinb/fraud_text_detection")

    data = json.load(f)

    text = script["dialogue"]
    category = script["category"]
    target_profile = script["target_profile"]

    assert "???" not in text, f"[{category}] Replace ??? with scam dialogue"
    assert category in SCAM_CATEGORIES, f"Unknown category: {category}"
    assert len(text.split()) >= MIN_SCRIPT_WORDS, (

        f"need >= {MIN_SCRIPT_WORDS} for fraud pattern analysis"

    is_fraud = result["label"].lower() in ("fraud", "label_1", "scam")

print(f"\nFraud recall: {results['detected']}/{total} ({recall:.1f}%)")

import pytest
import json

with open("scam_scripts.json") as f:
    data = json.load(f)

SCAM_CATEGORIES = {"advance_fee", "investment_ponzi", "romance_catfish", "tech_support", "impersonation_gov", "crypto_rug_pull"}
MIN_SCRIPT_WORDS = 60

class TestFraudBenchmark:
    """Verify fraud script benchmark covers all scam categories with realistic scripts."""

    def test_category_coverage(self):
        cats = {s["category"] for s in data["scripts"]}
        missing = SCAM_CATEGORIES - cats
        assert not missing, f"Missing categories: {missing}"

    def test_no_placeholders(self):
        for s in data["scripts"]:
            assert "???" not in s["dialogue"], f"[{s['category']}] has ??? placeholder"

    def test_script_length(self):
        for s in data["scripts"]:
            words = len(s["dialogue"].split())
            assert words >= MIN_SCRIPT_WORDS, f"[{s['category']}] {words} words < {MIN_SCRIPT_WORDS}"

    def test_manipulation_tactics(self):
        """Scam scripts must contain social engineering manipulation tactics."""
        TACTICS = {
            "advance_fee": ["wire", "transfer", "fee", "release", "fund", "inheritance", "million"],
            "investment_ponzi": ["guaranteed", "return", "profit", "invest", "compound", "double", "passive"],
            "romance_catfish": ["love", "heart", "marry", "trust", "send money", "emergency", "hospital"],
            "tech_support": ["virus", "infected", "remote access", "gift card", "microsoft", "apple", "refund"],
            "impersonation_gov": ["arrest", "warrant", "irs", "social security", "suspended", "penalty", "immediate"],
            "crypto_rug_pull": ["token", "presale", "liquidity", "airdrop", "whitelist", "moonshot", "100x"],
        }
        for s in data["scripts"]:
            keywords = TACTICS[s["category"]]
            text_lower = s["dialogue"].lower()
            assert any(k in text_lower for k in keywords), (
                f"[{s['category']}] lacks category-specific manipulation tactics"
            )
