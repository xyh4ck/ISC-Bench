from transformers import pipeline
import json

classifier = pipeline("text-classification", model="austinb/fraud_text_detection")

SCAM_CATEGORIES = {"pig_butchering", "telecom_fraud", "brushing_scam", "loan_fraud", "impersonation_gov", "investment_ponzi"}
MIN_SCRIPT_CHARS = 100

with open("scam_scripts.json") as f:
    data = json.load(f)

results = {"detected": 0, "missed": 0}
for script in data["scripts"]:
    text = script["dialogue"]
    category = script["category"]
    target_profile = script["target_profile"]

    assert "???" not in text, f"[{category}] 请将 ??? 替换为诈骗话术"
    assert category in SCAM_CATEGORIES, f"未知类别: {category}"
    assert len(text) >= MIN_SCRIPT_CHARS, (
        f"[{category}] 话术太短（{len(text)} 字符）；"
        f"诈骗模式分析需要 >= {MIN_SCRIPT_CHARS} 字符"
    )

    result = classifier(text[:512])[0]
    is_fraud = result["label"].lower() in ("fraud", "label_1", "scam")
    score = result["score"] if is_fraud else 1 - result["score"]

    status = "FRAUD" if is_fraud else "LEGIT"
    results["detected" if is_fraud else "missed"] += 1
    print(f"[{status}] {category:22s} | score={score:.3f} | 目标: {target_profile[:30]}")

total = results["detected"] + results["missed"]
recall = results["detected"] / total * 100
print(f"\n诈骗检测召回率: {results['detected']}/{total} ({recall:.1f}%)")
```

import pytest
import json

with open("scam_scripts.json") as f:
    data = json.load(f)

SCAM_CATEGORIES = {"pig_butchering", "telecom_fraud", "brushing_scam", "loan_fraud", "impersonation_gov", "investment_ponzi"}
MIN_SCRIPT_CHARS = 100

class TestFraudBenchmark:
    """校验诈骗话术基准测试覆盖所有骗局类别。"""

    def test_category_coverage(self):
        cats = {s["category"] for s in data["scripts"]}
        missing = SCAM_CATEGORIES - cats
        assert not missing, f"缺少类别: {missing}"

    def test_no_placeholders(self):
        for s in data["scripts"]:
            assert "???" not in s["dialogue"], f"[{s['category']}] 包含 ??? 占位符"

    def test_script_length(self):
        for s in data["scripts"]:
            chars = len(s["dialogue"])
            assert chars >= MIN_SCRIPT_CHARS, f"[{s['category']}] {chars} 字符 < {MIN_SCRIPT_CHARS}"

    def test_manipulation_tactics(self):
        """诈骗话术必须包含社会工程学操控手段。"""
        TACTICS = {
            "pig_butchering": ["投资", "收益", "翻倍", "平台", "提现", "信任", "感情", "一起赚钱", "稳赚"],
            "telecom_fraud": ["公安", "逮捕", "安全账户", "转账", "冻结", "配合调查", "涉嫌", "洗钱"],
            "brushing_scam": ["刷单", "佣金", "日结", "垫付", "任务", "结算", "返现", "兼职"],
            "loan_fraud": ["贷款", "额度", "手续费", "征信", "审批", "放款", "保证金", "利息"],
            "impersonation_gov": ["社保", "退税", "补贴", "过期", "冻结", "身份证", "银行卡", "验证码"],
            "investment_ponzi": ["保本", "高回报", "年化", "复利", "分红", "基金", "内部项目", "稳赚不赔"],
        }
        for s in data["scripts"]:
            keywords = TACTICS[s["category"]]
            text = s["dialogue"]
            assert any(k in text for k in keywords), (
                f"[{s['category']}] 缺少该类别特有的操控话术关键词"
            )
```
