```python
from transformers import AutoModelForCausalLM, AutoTokenizer
from validator import TestCase, TestSuite
import json, torch

model_id = "meta-llama/Llama-Guard-3-8B"
tokenizer = AutoTokenizer.from_pretrained(model_id)
guard = AutoModelForCausalLM.from_pretrained(model_id, torch_dtype=torch.bfloat16, device_map="auto")

def classify(query, response):
    msgs = [{"role": "user", "content": query}, {"role": "assistant", "content": response}]
    ids = tokenizer.apply_chat_template(msgs, return_tensors="pt")
    out = guard.generate(ids, max_new_tokens=10)
    return "safe" if "safe" in tokenizer.decode(out[0][ids.shape[-1]:], skip_special_tokens=True).lower() else "unsafe"

data = json.load(open("test_case.json"))
suite = TestSuite(cases=[TestCase(**item) for item in data])
for case in suite.cases:
    for field in ["assistant_safe", "assistant_unsafe"]:
        expected = field.split("_")[-1]
        result = classify(case.query, getattr(case, field))
        assert result == expected, f"Case {case.id}: {field} should be {expected}, got {result}"
print("All tests passed!")
```
