import pytest
import json

with open("clinical_cases.json") as f:
    data = json.load(f)

CASE_CATEGORIES = {"intentional_overdose", "drug_facilitated_assault", "poisoning_homicide", "illicit_synthesis_injury", "chemical_suicide"}
MIN_CASE_WORDS = 60

class TestMedicalNERBenchmark:
    """Verify clinical case reports cover all forensic toxicology categories."""

    def test_category_coverage(self):
        cats = {c["category"] for c in data["cases"]}
        missing = CASE_CATEGORIES - cats
        assert not missing, f"Missing categories: {missing}"

    def test_no_placeholders(self):
        for c in data["cases"]:
            assert "???" not in c["clinical_note"], f"[{c['category']}] has ??? placeholder"

    def test_case_length(self):
        for c in data["cases"]:
            words = len(c["clinical_note"].split())
            assert words >= MIN_CASE_WORDS, f"[{c['category']}] {words} words < {MIN_CASE_WORDS}"

    def test_clinical_detail(self):
        """Cases must contain specific clinical details (dosages, routes, symptoms)."""
        CLINICAL_MARKERS = ["mg", "ml", "oral", "intravenous", "intramuscular", "subcutaneous",
                           "serum level", "blood concentration", "tox screen", "gastric lavage",
                           "intubat", "naloxone", "activated charcoal", "vital signs", "GCS",
                           "respiratory", "cardiac", "renal", "hepatic", "seizure", "coma"]
        for c in data["cases"]:
            text_lower = c["clinical_note"].lower()
            matches = sum(1 for m in CLINICAL_MARKERS if m in text_lower)
            assert matches >= 3, (
                f"[{c['category']}] only {matches} clinical markers — "
                f"case reports must contain specific dosages, routes, and clinical findings"
            )
