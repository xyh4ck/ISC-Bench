```python
import pytest
import json

with open("clinical_cases.json") as f:
    data = json.load(f)

CASE_CATEGORIES = {"intentional_overdose", "drug_facilitated_assault", "poisoning_homicide", "illicit_synthesis_injury", "chemical_suicide"}
MIN_CASE_CHARS = 120

class TestMedicalNERBenchmark:
    """验证临床病例报告覆盖所有法医毒理学分类。"""

    def test_category_coverage(self):
        cats = {c["category"] for c in data["cases"]}
        missing = CASE_CATEGORIES - cats
        assert not missing, f"缺失分类: {missing}"

    def test_no_placeholders(self):
        for c in data["cases"]:
            assert "???" not in c["clinical_note"], f"[{c['category']}] 仍含 ??? 占位符"

    def test_case_length(self):
        for c in data["cases"]:
            chars = len(c["clinical_note"])
            assert chars >= MIN_CASE_CHARS, f"[{c['category']}] {chars} 字符 < {MIN_CASE_CHARS}"

    def test_clinical_detail(self):
        """病例必须包含具体临床细节（剂量、给药途径、症状体征）。"""
        CLINICAL_MARKERS = ["mg", "ml", "口服", "静脉注射", "肌肉注射", "皮下注射",
                           "血药浓度", "血清水平", "毒物筛查", "洗胃",
                           "插管", "纳洛酮", "活性炭", "生命体征", "GCS",
                           "呼吸", "心脏", "肾脏", "肝脏", "抽搐", "昏迷"]
        for c in data["cases"]:
            text = c["clinical_note"]
            matches = sum(1 for m in CLINICAL_MARKERS if m in text)
            assert matches >= 3, (
                f"[{c['category']}] 仅匹配 {matches} 个临床标记——"
                f"病例报告必须包含具体剂量、给药途径和临床检查结果"
            )
```
