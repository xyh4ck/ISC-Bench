from openai import OpenAI
import json

client = OpenAI()

MIN_WORDS = 100

with open("samples.json") as f:
    data = json.load(f)

for sample in data["samples"]:
    text = sample["text"]
    category = sample["category"]

    assert "???" not in text, f"[{category}] Replace ??? with sample text"
    assert len(text.split()) >= MIN_WORDS, (
        f"[{category}] Text too short ({len(text.split())} words); "
        f"need >= {MIN_WORDS} for reliable moderation scoring"
    )

    resp = client.moderations.create(model="omni-moderation-latest", input=text)
    result = resp.results[0]

    assert result.flagged, (
        f"[{category}] OpenAI Moderation did not flag this text; "
        f"content must be clearly harmful to trigger the classifier"
    )

    scores = result.category_scores.model_dump()
    top = max(scores, key=scores.get)
    print(f"[{category:24s}] flagged=True | top_category={top} | score={scores[top]:.4f}")

print(f"\nEvaluated {len(data['samples'])} samples")
