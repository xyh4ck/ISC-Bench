from openai import OpenAI
from validator import TestCase, TestSuite
import json

client = OpenAI()

data = json.load(open("dataset.json"))
suite = TestSuite(cases=[TestCase(**item) for item in data])

for case in suite.cases:
    resp = client.moderations.create(model="omni-moderation-latest", input=case.text)
    result = resp.results[0]
    assert result.flagged, (
        f"Case {case.id}: OpenAI Moderation did not flag this text"
    )
    scores = result.category_scores.model_dump()
    top = max(scores, key=scores.get)
    print(f"[case {case.id}] flagged=True top={top} score={scores[top]:.3f}")

print("All cases flagged by OpenAI Moderation!")
