#!/usr/bin/env python3
"""Test suite for user-input moderation filter.

Loads attack prompts from YAML and verifies that the filter
correctly classifies each one. Each prompt must be a realistic
attack that a real user might submit.

YAML format:
  - category: <attack type>
    prompt: <the user input to test>
    expected: blocked
"""
import yaml
import sys

ATTACK_CATEGORIES = {
    "jailbreak",           # prompt tries to override system instructions
    "social_engineering",  # manipulates the model into revealing info
    "harmful_request",     # directly asks for dangerous content
    "persona_hijack",      # tries to make the model adopt a harmful persona
    "instruction_leak",    # attempts to extract system prompt
}

MIN_WORDS = 25

with open("attack_prompts.yaml") as f:
    cases = yaml.safe_load(f)

assert isinstance(cases, list), "YAML must be a list of test cases"
assert len(cases) >= 5, f"Need at least 5 test cases, got {len(cases)}"

for i, case in enumerate(cases):
    cat = case["category"]
    prompt = case["prompt"]

    assert cat in ATTACK_CATEGORIES, (
        f"Case {i}: unknown category '{cat}'. Must be one of: {ATTACK_CATEGORIES}"
    )
    assert "???" not in prompt, (
        f"Case {i} ({cat}): replace ??? with a realistic attack prompt"
    )
    assert len(prompt.split()) >= MIN_WORDS, (
        f"Case {i} ({cat}): prompt has {len(prompt.split())} words, need >= {MIN_WORDS} for realistic testing"
    )

print(f"Validated {len(cases)} attack prompts across {len(set(c['category'] for c in cases))} categories.")
print("Ready to run filter benchmark.")
