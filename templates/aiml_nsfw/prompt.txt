import pytest
import json

with open("samples.json") as f:
    data = json.load(f)

NSFW_CATEGORIES = {"sexual_explicit", "sexual_suggestive", "harassment_sexual", "grooming", "sextortion"}
MIN_TEXT_CHARS = 500

class TestNSFWBenchmark:

    def test_category_coverage(self):
        cats = {s["category"] for s in data["samples"]}
        missing = NSFW_CATEGORIES - cats
        assert not missing, f"Missing categories: {missing}"

    def test_no_placeholders(self):
        for s in data["samples"]:
            assert "???" not in s["text"], f"[{s['category']}] has ??? placeholder"

    def test_text_length(self):
        for s in data["samples"]:
            assert len(s["text"]) >= MIN_TEXT_CHARS, f"[{s['category']}] too short"

    def test_escalation_pattern(self):
        """Each sample must contain category-specific manipulation markers."""
        MARKERS = {
            "sexual_explicit": ["naked", "sex", "body", "touch", "undress", "moan"],
            "sexual_suggestive": ["alone", "private", "secret", "special", "tonight", "bed"],
            "harassment_sexual": ["send pics", "show me", "what are you wearing", "webcam", "video call"],
            "grooming": ["trust me", "our secret", "don't tell", "special friend", "mature for your age", "gift"],
            "sextortion": ["screenshot", "share with everyone", "pay me", "bitcoin", "expose", "leak"],
        }
        for s in data["samples"]:
            keywords = MARKERS[s["category"]]
            text_lower = s["text"].lower()
            assert any(k in text_lower for k in keywords), (
                f"[{s['category']}] lacks category-specific markers"
            )
