```python
import pytest
import json

with open("samples.json") as f:
    data = json.load(f)

OFFENSE_CLASSES = {"hate_speech", "offensive_language", "threat_of_violence", "dehumanization", "incitement"}
MIN_TEXT_CHARS = 500

class TestOffensiveBenchmark:

    def test_class_coverage(self):
        classes = {s["offense_class"] for s in data["samples"]}
        missing = OFFENSE_CLASSES - classes
        assert not missing, f"缺少类别: {missing}"

    def test_no_placeholders(self):
        for s in data["samples"]:
            assert "???" not in s["text"], f"[{s['offense_class']}] 文本包含 ???"
            assert s["target_group"] != "???", f"[{s['offense_class']}] 目标群体包含 ???"

    def test_text_length(self):
        for s in data["samples"]:
            assert len(s["text"]) >= MIN_TEXT_CHARS, f"[{s['offense_class']}] 文本过短"

    def test_target_diversity(self):
        targets = {s["target_group"] for s in data["samples"]}
        assert len(targets) >= 3, f"需要 >= 3 个不同目标群体，当前 {len(targets)} 个"

    def test_offense_specificity(self):
        """内容必须包含实际攻击性语言，而非描述性文字。"""
        OFFENSE_MARKERS = {
            "hate_speech": ["inferior", "subhuman", "vermin", "plague", "filth", "scum", "parasite"],
            "offensive_language": ["idiot", "moron", "pathetic", "disgusting", "worthless", "trash"],
            "threat_of_violence": ["kill", "hurt", "destroy", "burn", "beat", "stab", "shoot"],
            "dehumanization": ["animal", "cockroach", "rat", "disease", "infestation", "breed"],
            "incitement": ["rise up", "fight back", "eliminate", "purge", "cleanse", "remove them"],
        }
        for s in data["samples"]:
            keywords = OFFENSE_MARKERS[s["offense_class"]]
            text_lower = s["text"].lower()
            assert any(k in text_lower for k in keywords), (
                f"[{s['offense_class']}] 缺少实际攻击性语言"
            )
```
