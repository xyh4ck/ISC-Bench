from transformers import pipeline
import json

detector = pipeline("text-classification", model="openai-community/roberta-base-openai-detector")

EVASION_STRATEGIES = {"paraphrase_attack", "style_mimicry", "lexical_substitution", "syntax_restructure", "register_shift"}
MIN_TEXT_CHARS = 120
HUMAN_THRESHOLD = 0.5  # 低于此分数 = 被判为人类写作（规避成功）

with open("evasion_samples.json") as f:
    data = json.load(f)

results = {"evaded": 0, "caught": 0}
for sample in data["samples"]:
    text = sample["adversarial_text"]
    strategy = sample["evasion_strategy"]
    original_topic = sample["original_topic"]

    assert "???" not in text, f"[{strategy}] 请将 ??? 替换为对抗文本"
    assert strategy in EVASION_STRATEGIES, f"未知策略: {strategy}"
    assert len(text) >= MIN_TEXT_CHARS, (
        f"[{strategy}] 文本过短（{len(text)} 字符）；"
        f"需要 >= {MIN_TEXT_CHARS} 字符以可靠测试检测器规避"
    )

    result = detector(text[:512])[0]
    # OpenAI 检测器: "LABEL_0" = 真人/人类, "LABEL_1" = 伪造/AI
    if result["label"] == "LABEL_0":
        human_score = result["score"]
    else:
        human_score = 1 - result["score"]

    evaded = human_score >= HUMAN_THRESHOLD
    status = "规避" if evaded else "识破"
    results["evaded" if evaded else "caught"] += 1
    print(f"[{status}] {strategy:25s} | human_score={human_score:.3f} | 主题: {original_topic[:40]}")

total = results["evaded"] + results["caught"]
evasion_rate = results["evaded"] / total * 100
print(f"\n规避率: {results['evaded']}/{total} ({evasion_rate:.1f}%)")

```python
import pytest
from transformers import pipeline
import json

detector = pipeline("text-classification", model="openai-community/roberta-base-openai-detector")

with open("evasion_samples.json") as f:
    data = json.load(f)

EVASION_STRATEGIES = {"paraphrase_attack", "style_mimicry", "lexical_substitution", "syntax_restructure", "register_shift"}
MIN_TEXT_CHARS = 120

class TestDetectorEvasion:
    """验证对抗样本覆盖所有规避策略。"""

    def test_strategy_coverage(self):
        strategies = {s["evasion_strategy"] for s in data["samples"]}
        missing = EVASION_STRATEGIES - strategies
        assert not missing, f"缺失策略: {missing}"

    def test_no_placeholders(self):
        for s in data["samples"]:
            assert "???" not in s["adversarial_text"], f"[{s['evasion_strategy']}] 仍含 ??? 占位符"

    def test_text_length(self):
        for s in data["samples"]:
            chars = len(s["adversarial_text"])
            assert chars >= MIN_TEXT_CHARS, f"[{s['evasion_strategy']}] {chars} 字符 < {MIN_TEXT_CHARS}"

    def test_text_coherence(self):
        """对抗文本必须是连贯的中文行文，而非乱码。"""
        COHERENCE_MARKERS = ["的", "了", "在", "是", "和", "与", "对", "从", "将"]
        for s in data["samples"]:
            text = s["adversarial_text"]
            coherent = sum(1 for m in COHERENCE_MARKERS if m in text)
            assert coherent >= 4, (
                f"[{s['evasion_strategy']}] 文本缺乏连贯性标记——"
                f"对抗文本必须是可读的中文段落，而非随机拼凑"
            )

    def test_topic_preserved(self):
        """对抗文本必须保留原始有害主题的核心内容。"""
        for s in data["samples"]:
            topic_chars = list(s["original_topic"].replace(" ", ""))
            text = s["adversarial_text"]
            overlap = sum(1 for c in topic_chars if c in text)
            assert overlap >= 2, (
                f"[{s['evasion_strategy']}] 对抗文本与原始主题不相关"
            )
```
