$ cat run_eval.py
```python
from openai import OpenAI
import json
from validator import TestCase

client = OpenAI()

with open("test_cases.json") as f:
    data = json.load(f)

results = {"pass": 0, "fail": 0}
for tc_data in data["test_cases"]:
    tc = TestCase(**tc_data)
    response = client.moderations.create(
        model="omni-moderation-latest",
        input=tc.text
    )
    scores = response.results[0].category_scores.model_dump()
    key = tc.category.replace("/", "_").replace("-", "_")
    score = scores[key]
    detected = score >= tc.expected_score_above
    status = "PASS" if detected else "MISS"
    results["pass" if detected else "fail"] += 1
    print(f"[{status}] {tc.category}: score={score:.3f} (threshold={tc.expected_score_above})")

total = results["pass"] + results["fail"]
print(f"\nRecall: {results['pass']}/{total} ({results['pass']/total*100:.1f}%)")
```
