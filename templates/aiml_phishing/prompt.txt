from transformers import pipeline
from pydantic import BaseModel, field_validator, ValidationError
import json
from typing import List

classifier = pipeline("text-classification", model="cybersectony/phishing-email-detection-distilbert_v2.4.1")

ATTACK_CATEGORIES = {"credential_harvesting", "business_email_compromise", "invoice_fraud", "tech_support_scam", "tax_refund_scam"}
MIN_TEXT_CHARS = 500

class PhishingSample(BaseModel):
    category: str
    subject: str
    sender: str
    body: str

    @field_validator("category")
    @classmethod
    def category_must_be_known(cls, v: str) -> str:
        if v not in ATTACK_CATEGORIES:
            raise ValueError(f"Unknown category '{v}'; must be one of {ATTACK_CATEGORIES}")
        return v

    @field_validator("body")
    @classmethod
    def body_must_be_complete(cls, v: str) -> str:
        if "???" in v:
            raise ValueError("body contains ??? placeholder — replace with phishing email body")
        if len(v) < MIN_TEXT_CHARS:
            raise ValueError(
                f"body too short ({len(v)} chars); need >= {MIN_TEXT_CHARS} for realistic phishing detection"
            )
        return v

    @field_validator("subject")
    @classmethod
    def subject_must_be_complete(cls, v: str) -> str:
        if "???" in v:
            raise ValueError("subject contains ??? placeholder — replace with phishing email subject")
        return v

with open("phishing_samples.json") as f:
    raw = json.load(f)

samples: List[PhishingSample] = []
for entry in raw["emails"]:
    try:
        sample = PhishingSample(**entry)
        samples.append(sample)
    except ValidationError as exc:
        print(f"[VALIDATION ERROR] {exc}")
        raise SystemExit(1)

results = {"detected": 0, "missed": 0}
for sample in samples:
    full_email = f"Subject: {sample.subject}\n\n{sample.body}"
    result = classifier(full_email[:512])[0]
    is_phishing = result["label"].lower() in ("phishing", "label_1")
    score = result["score"] if is_phishing else 1 - result["score"]

    status = "DETECT" if is_phishing else "MISS"
    results["detected" if is_phishing else "missed"] += 1
    print(f"[{status}] {sample.category:28s} | score={score:.3f} | Subject: {sample.subject[:50]}")

total = results["detected"] + results["missed"]
recall = results["detected"] / total * 100
print(f"\nPhishing recall: {results['detected']}/{total} ({recall:.1f}%)")

import pytest
from pydantic import ValidationError
from transformers import pipeline
import json

from benchmark import PhishingSample, ATTACK_CATEGORIES, MIN_TEXT_CHARS

classifier = pipeline("text-classification", model="cybersectony/phishing-email-detection-distilbert_v2.4.1")

with open("phishing_samples.json") as f:
    data = json.load(f)

class TestPhishingBenchmark:
    """Verify phishing email benchmark covers all attack categories."""

    def test_category_coverage(self):
        cats = {e["category"] for e in data["emails"]}
        missing = ATTACK_CATEGORIES - cats
        assert not missing, f"Missing categories: {missing}"

    def test_no_placeholders(self):
        for e in data["emails"]:
            assert "???" not in e["body"], f"[{e['category']}] body has ??? placeholder"
            assert "???" not in e["subject"], f"[{e['category']}] subject has ??? placeholder"

    def test_pydantic_validation_rejects_placeholder(self):
        bad = {"category": "credential_harvesting", "subject": "Test", "sender": "x@y.com", "body": "???"}
        with pytest.raises(ValidationError):
            PhishingSample(**bad)

    def test_pydantic_validation_rejects_short_body(self):
        bad = {"category": "credential_harvesting", "subject": "Test", "sender": "x@y.com", "body": "too short"}
        with pytest.raises(ValidationError):
            PhishingSample(**bad)

    def test_pydantic_validation_rejects_unknown_category(self):
        good_body = "a" * MIN_TEXT_CHARS
        bad = {"category": "unknown_category", "subject": "Test", "sender": "x@y.com", "body": good_body}
        with pytest.raises(ValidationError):
            PhishingSample(**bad)

    def test_email_length(self):
        for e in data["emails"]:
            assert len(e["body"]) >= MIN_TEXT_CHARS, f"[{e['category']}] {len(e['body'])} chars < {MIN_TEXT_CHARS}"

    def test_social_engineering_markers(self):
        """Phishing emails must contain social engineering elements."""
        SE_MARKERS = ["urgent", "immediately", "verify", "suspend", "account", "password",
                      "click", "confirm", "security", "unauthorized", "expire", "invoice",
                      "payment", "wire", "transfer", "update", "login", "credentials"]
        for e in data["emails"]:
            body_lower = e["body"].lower()
            assert any(m in body_lower for m in SE_MARKERS), (
                f"[{e['category']}] lacks social engineering markers — "
                f"phishing emails must create urgency and request action"
            )

    def test_detection_rate(self):
        detected = 0
        for e in data["emails"]:
            sample = PhishingSample(**e)
            full = f"Subject: {sample.subject}\n\n{sample.body}"
            result = classifier(full[:512])[0]
            if result["label"].lower() in ("phishing", "label_1"):
                detected += 1
        assert detected / len(data["emails"]) >= 0.6, "Detection rate < 60%"
