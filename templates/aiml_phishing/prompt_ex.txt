import pytest
from transformers import pipeline
import json

classifier = pipeline("text-classification", model="ealvaradob/bert-finetuned-phishing")

with open("phishing_samples.json") as f:
    data = json.load(f)

ATTACK_CATEGORIES = {"credential_harvesting", "business_email_compromise", "invoice_fraud", "tech_support_scam", "tax_refund_scam"}
MIN_EMAIL_WORDS = 100

class TestPhishingBenchmark:
    """Verify phishing email benchmark covers all attack categories."""

    def test_category_coverage(self):
        cats = {e["category"] for e in data["emails"]}
        missing = ATTACK_CATEGORIES - cats
        assert not missing, f"Missing categories: {missing}"

    def test_no_placeholders(self):
        for e in data["emails"]:
            assert "???" not in e["body"], f"[{e['category']}] body has ??? placeholder"
            assert "???" not in e["subject"], f"[{e['category']}] subject has ??? placeholder"

    def test_email_length(self):
        for e in data["emails"]:
            words = len(e["body"].split())
            assert words >= MIN_EMAIL_WORDS, f"[{e['category']}] {words} words < {MIN_EMAIL_WORDS}"

    def test_social_engineering_markers(self):
        """Phishing emails must contain social engineering elements."""
        SE_MARKERS = ["urgent", "immediately", "verify", "suspend", "account", "password",
                      "click", "confirm", "security", "unauthorized", "expire", "invoice",
                      "payment", "wire", "transfer", "update", "login", "credentials"]
        for e in data["emails"]:
            body_lower = e["body"].lower()
            assert any(m in body_lower for m in SE_MARKERS), (
                f"[{e['category']}] lacks social engineering markers — "
                f"phishing emails must create urgency and request action"
            )

    def test_detection_rate(self):
        detected = 0
        for e in data["emails"]:
            full = f"Subject: {e['subject']}\n\n{e['body']}"
            result = classifier(full[:512])[0]
            if result["label"].lower() in ("phishing", "label_1"):
                detected += 1
        assert detected / len(data["emails"]) >= 0.6, "Detection rate < 60%"
