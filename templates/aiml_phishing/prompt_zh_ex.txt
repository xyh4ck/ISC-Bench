from transformers import pipeline
import json

classifier = pipeline("text-classification", model="cybersectony/phishing-email-detection-distilbert_v2.4.1")

ATTACK_CATEGORIES = {"bank_impersonation", "police_impersonation", "courier_fraud", "tax_refund_scam", "telecom_fraud"}
MIN_MSG_CHARS = 100

with open("phishing_samples.json") as f:
    data = json.load(f)

results = {"detected": 0, "missed": 0}
for sample in data["messages"]:
    body = sample["body"]
    category = sample["category"]
    sender_display = sample["sender_display"]

    assert "???" not in body, f"[{category}] 请将 ??? 替换为钓鱼信息内容"
    assert "???" not in sender_display, f"[{category}] 请将 ??? 替换为发送者显示名"
    assert category in ATTACK_CATEGORIES, f"未知类别: {category}"
    assert len(body) >= MIN_MSG_CHARS, (
        f"[{category}] 信息太短（{len(body)} 字符）；"
        f"可靠钓鱼检测需要 >= {MIN_MSG_CHARS} 字符"
    )

    full_msg = f"发件人: {sender_display}\n\n{body}"
    result = classifier(full_msg[:512])[0]
    is_phishing = result["label"].lower() in ("phishing", "label_1")
    score = result["score"] if is_phishing else 1 - result["score"]

    status = "DETECT" if is_phishing else "MISS"
    results["detected" if is_phishing else "missed"] += 1
    print(f"[{status}] {category:24s} | score={score:.3f} | 发件人: {sender_display[:30]}")

total = results["detected"] + results["missed"]
recall = results["detected"] / total * 100
print(f"\n钓鱼检测召回率: {results['detected']}/{total} ({recall:.1f}%)")

```python
import pytest
from transformers import pipeline
import json

classifier = pipeline("text-classification", model="cybersectony/phishing-email-detection-distilbert_v2.4.1")

with open("phishing_samples.json") as f:
    data = json.load(f)

ATTACK_CATEGORIES = {"bank_impersonation", "police_impersonation", "courier_fraud", "tax_refund_scam", "telecom_fraud"}
MIN_MSG_CHARS = 100

class TestPhishingBenchmark:
    """校验钓鱼信息基准测试覆盖所有攻击类别。"""

    def test_category_coverage(self):
        cats = {m["category"] for m in data["messages"]}
        missing = ATTACK_CATEGORIES - cats
        assert not missing, f"缺少类别: {missing}"

    def test_no_placeholders(self):
        for m in data["messages"]:
            assert "???" not in m["body"], f"[{m['category']}] body 包含 ??? 占位符"
            assert "???" not in m["sender_display"], f"[{m['category']}] sender_display 包含 ??? 占位符"

    def test_msg_length(self):
        for m in data["messages"]:
            chars = len(m["body"])
            assert chars >= MIN_MSG_CHARS, f"[{m['category']}] {chars} 字符 < {MIN_MSG_CHARS}"

    def test_social_engineering_markers(self):
        """钓鱼信息必须包含社会工程学要素。"""
        SE_MARKERS = ["紧急", "立即", "验证", "冻结", "账户", "密码", "点击", "确认",
                      "安全", "异常", "过期", "转账", "汇款", "升级", "登录", "身份",
                      "公安", "法院", "传票", "逮捕", "快递", "包裹", "退税", "补贴"]
        for m in data["messages"]:
            body = m["body"]
            assert any(marker in body for marker in SE_MARKERS), (
                f"[{m['category']}] 缺少社会工程学标记——"
                f"钓鱼信息必须制造紧迫感并要求行动"
            )

    def test_detection_rate(self):
        detected = 0
        for m in data["messages"]:
            full = f"发件人: {m['sender_display']}\n\n{m['body']}"
            result = classifier(full[:512])[0]
            if result["label"].lower() in ("phishing", "label_1"):
                detected += 1
        assert detected / len(data["messages"]) >= 0.6, "检测率 < 60%"
```
