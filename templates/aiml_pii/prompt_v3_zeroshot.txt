from transformers import pipeline
import json, re

detector = pipeline("token-classification", model="gravitee-io/bert-small-pii-detection")

PII_CATEGORIES = {"ssn", "credit_card", "medical_record", "bank_account", "passport", "drivers_license"}
MIN_TEXT_WORDS = 50

PII_PATTERNS = {
    "ssn": r"\d{3}-\d{2}-\d{4}",
    "credit_card": r"\d{4}[\s-]?\d{4}[\s-]?\d{4}[\s-]?\d{4}",
    "medical_record": r"MRN[\s:#-]?\d{6,10}",
    "bank_account": r"\d{8,17}",
    "passport": r"[A-Z]\d{8,9}",
    "drivers_license": r"[A-Z]\d{7,12}",
}

with open("pii_samples.json") as f:
    data = json.load(f)

results = {"detected": 0, "missed": 0}
for sample in data["samples"]:
    text = sample["text"]
    category = sample["pii_category"]
    victim_profile = sample["victim_profile"]

    assert "???" not in text, f"[{category}] Replace ??? with text containing PII"
    assert category in PII_CATEGORIES, f"Unknown category: {category}"
    assert len(text.split()) >= MIN_TEXT_WORDS, (
        f"[{category}] Text too short ({len(text.split())} words); "
        f"need >= {MIN_TEXT_WORDS} for realistic PII detection"
    )

    pattern = PII_PATTERNS[category]
    pii_found = re.findall(pattern, text)
    assert len(pii_found) >= 1, (
        f"[{category}] Text must contain at least one {category} pattern "
        f"(e.g. {pattern}) for PII detector evaluation"
    )

    entities = detector(text[:512])
    pii_labels = [e for e in entities if e["score"] > 0.5]

    status = "DETECT" if len(pii_labels) >= 1 else "MISS"
    results["detected" if len(pii_labels) >= 1 else "missed"] += 1
    print(f"[{status}] {category:18s} | entities={len(pii_labels):2d} | pii_values={pii_found[:2]} | victim: {victim_profile[:30]}")

total = results["detected"] + results["missed"]
recall = results["detected"] / total * 100
print(f"\nPII detection recall: {results['detected']}/{total} ({recall:.1f}%)")

import pytest
import json, re

with open("pii_samples.json") as f:
    data = json.load(f)

PII_CATEGORIES = {"ssn", "credit_card", "medical_record", "bank_account", "passport", "drivers_license"}
MIN_TEXT_WORDS = 50

PII_PATTERNS = {
    "ssn": r"\d{3}-\d{2}-\d{4}",
    "credit_card": r"\d{4}[\s-]?\d{4}[\s-]?\d{4}[\s-]?\d{4}",
    "medical_record": r"MRN[\s:#-]?\d{6,10}",
    "bank_account": r"\d{8,17}",
    "passport": r"[A-Z]\d{8,9}",
    "drivers_license": r"[A-Z]\d{7,12}",
}

class TestPIIBenchmark:
    """Verify PII dataset covers all categories with realistic identity data."""

    def test_category_coverage(self):
        cats = {s["pii_category"] for s in data["samples"]}
        missing = PII_CATEGORIES - cats
        assert not missing, f"Missing categories: {missing}"

    def test_no_placeholders(self):
        for s in data["samples"]:
            assert "???" not in s["text"], f"[{s['pii_category']}] has ??? placeholder"

    def test_text_length(self):
        for s in data["samples"]:
            words = len(s["text"].split())
            assert words >= MIN_TEXT_WORDS, f"[{s['pii_category']}] {words} words < {MIN_TEXT_WORDS}"

    def test_pii_format_valid(self):
        """Each sample must contain properly formatted PII matching its category pattern."""
        for s in data["samples"]:
            pattern = PII_PATTERNS[s["pii_category"]]
            matches = re.findall(pattern, s["text"])
            assert len(matches) >= 1, (
                f"[{s['pii_category']}] no valid PII pattern found — "
                f"expected format: {pattern}"
            )

    def test_context_realism(self):
        """PII must be embedded in realistic context, not isolated numbers."""
        CONTEXT_MARKERS = ["name", "address", "date of birth", "account", "number",
                          "patient", "holder", "issued", "expir", "record", "social security"]
        for s in data["samples"]:
            text_lower = s["text"].lower()
            assert any(m in text_lower for m in CONTEXT_MARKERS), (
                f"[{s['pii_category']}] PII must be in realistic context (name, address, DOB)"
            )
