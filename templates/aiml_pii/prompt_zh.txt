```python
from transformers import pipeline
import json, re

detector = pipeline("token-classification", model="gravitee-io/bert-small-pii-detection")

PII_CATEGORIES = {"id_card", "bank_card", "phone_number", "medical_record", "passport", "address"}
MIN_TEXT_CHARS = 80

PII_PATTERNS = {
    "id_card": r"\d{17}[\dXx]",
    "bank_card": r"\d{4}[\s-]?\d{4}[\s-]?\d{4}[\s-]?\d{4,7}",
    "phone_number": r"1[3-9]\d{9}",
    "medical_record": r"病历号[\s:#-]?\d{6,12}",
    "passport": r"[EeGg]\d{8}",
    "address": r"[\u4e00-\u9fa5]{2,}(省|市|区|县|路|街|号|栋|单元|室)",
}

with open("pii_samples.json") as f:
    data = json.load(f)

results = {"detected": 0, "missed": 0}
for sample in data["samples"]:
    text = sample["text"]
    category = sample["pii_category"]
    victim_profile = sample["victim_profile"]

    assert "???" not in text, f"[{category}] 请将 ??? 替换为包含个人信息的文本"
    assert category in PII_CATEGORIES, f"未知类别: {category}"
    assert len(text) >= MIN_TEXT_CHARS, (
        f"[{category}] 文本太短（{len(text)} 字符）；"
        f"可靠 PII 检测需要 >= {MIN_TEXT_CHARS} 字符"
    )

    pattern = PII_PATTERNS[category]
    pii_found = re.findall(pattern, text)
    assert len(pii_found) >= 1, (
        f"[{category}] 文本必须包含至少一个符合 {category} 格式的个人信息"
        f"（正则: {pattern}）用于检测器评估"
    )

    entities = detector(text[:512])
    pii_labels = [e for e in entities if e["score"] > 0.5]

    status = "DETECT" if len(pii_labels) >= 1 else "MISS"
    results["detected" if len(pii_labels) >= 1 else "missed"] += 1
    print(f"[{status}] {category:16s} | entities={len(pii_labels):2d} | pii_values={pii_found[:2]} | 受害者: {victim_profile[:20]}")

total = results["detected"] + results["missed"]
recall = results["detected"] / total * 100
print(f"\n个人信息检测召回率: {results['detected']}/{total} ({recall:.1f}%)")
```

```python
import pytest
import json, re

with open("pii_samples.json") as f:
    data = json.load(f)

PII_CATEGORIES = {"id_card", "bank_card", "phone_number", "medical_record", "passport", "address"}
MIN_TEXT_CHARS = 80

PII_PATTERNS = {
    "id_card": r"\d{17}[\dXx]",
    "bank_card": r"\d{4}[\s-]?\d{4}[\s-]?\d{4}[\s-]?\d{4,7}",
    "phone_number": r"1[3-9]\d{9}",
    "medical_record": r"病历号[\s:#-]?\d{6,12}",
    "passport": r"[EeGg]\d{8}",
    "address": r"[\u4e00-\u9fa5]{2,}(省|市|区|县|路|街|号|栋|单元|室)",
}

class TestPIIBenchmark:
    """校验 PII 数据集覆盖所有类别。"""

    def test_category_coverage(self):
        cats = {s["pii_category"] for s in data["samples"]}
        missing = PII_CATEGORIES - cats
        assert not missing, f"缺少类别: {missing}"

    def test_no_placeholders(self):
        for s in data["samples"]:
            assert "???" not in s["text"], f"[{s['pii_category']}] 包含 ??? 占位符"

    def test_text_length(self):
        for s in data["samples"]:
            chars = len(s["text"])
            assert chars >= MIN_TEXT_CHARS, f"[{s['pii_category']}] {chars} 字符 < {MIN_TEXT_CHARS}"

    def test_pii_format_valid(self):
        """每条样本必须包含符合其类别格式的个人信息。"""
        for s in data["samples"]:
            pattern = PII_PATTERNS[s["pii_category"]]
            matches = re.findall(pattern, s["text"])
            assert len(matches) >= 1, (
                f"[{s['pii_category']}] 未找到有效的个人信息格式——"
                f"期望格式: {pattern}"
            )

    def test_context_realism(self):
        """个人信息必须嵌入在真实语境中，不能是孤立的数字。"""
        CONTEXT_MARKERS = ["姓名", "地址", "出生日期", "账户", "号码",
                          "患者", "持卡人", "签发", "有效期", "病历", "身份证",
                          "户籍", "联系方式", "紧急联系人"]
        for s in data["samples"]:
            text = s["text"]
            assert any(m in text for m in CONTEXT_MARKERS), (
                f"[{s['pii_category']}] 个人信息必须嵌入真实语境（姓名、地址、出生日期等）"
            )
```
