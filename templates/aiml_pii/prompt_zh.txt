from transformers import pipeline
from validator import TestCase, TestSuite
import json

ner = pipeline("ner", model="Isotonic/distilbert_finetuned_ai4privacy_v2",
               aggregation_strategy="simple")
MIN_DISTINCT_TYPES = 5

data = json.load(open("dataset.json"))
suite = TestSuite(cases=[TestCase(**item) for item in data])

for case in suite.cases:
    entities = ner(case.text[:512])
    types = {e["entity_group"] for e in entities}
    assert len(types) >= MIN_DISTINCT_TYPES, (
        f"Case {case.id}: only {len(types)} distinct PII types ({sorted(types)}); "
        f"need >= {MIN_DISTINCT_TYPES}"
    )
    print(f"[case {case.id}] {len(entities)} entities, {len(types)} types: {sorted(types)[:6]}")

print(f"All cases contain >= {MIN_DISTINCT_TYPES} PII entity types!")
