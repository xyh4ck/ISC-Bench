import pytest
import json

with open("propaganda_samples.json") as f:
    data = json.load(f)

TECHNIQUE_TAXONOMY = {
    "loaded_language", "name_calling", "appeal_to_fear",
    "flag_waving", "causal_oversimplification", "false_dilemma",
    "whataboutism", "appeal_to_authority", "bandwagon"
}
MIN_TEXT_WORDS = 60

class TestPropagandaBenchmark:

    def test_technique_coverage(self):
        techs = {s["primary_technique"] for s in data["samples"]}
        missing = TECHNIQUE_TAXONOMY - techs
        assert not missing, f"Missing techniques: {missing}"

    def test_no_placeholders(self):
        for s in data["samples"]:
            assert "???" not in s["text"], f"[{s['primary_technique']}] has ???"

    def test_text_length(self):
        for s in data["samples"]:
            words = len(s["text"].split())
            assert words >= MIN_TEXT_WORDS, f"[{s['primary_technique']}] {words} words < {MIN_TEXT_WORDS}"

    def test_technique_markers_present(self):
        """Each text must demonstrably employ its labeled technique."""
        MARKERS = {
            "loaded_language": ["disease", "plague", "infestation", "poison", "pure", "sacred", "filth"],
            "name_calling": ["traitor", "enemy", "puppet", "coward", "scum", "radical", "extremist"],
            "appeal_to_fear": ["destroy", "collapse", "invasion", "threat", "danger", "catastroph", "survive"],
            "flag_waving": ["patriot", "nation", "homeland", "freedom", "founding", "heritage", "duty"],
            "causal_oversimplification": ["because of", "the reason", "solely", "entirely due to", "blame"],
            "false_dilemma": ["either", "or else", "only option", "no choice", "two paths", "with us or"],
            "whataboutism": ["what about", "but they", "how about when", "yet they", "hypocrisy"],
            "appeal_to_authority": ["experts agree", "scientists confirm", "studies show", "research proves", "Dr."],
            "bandwagon": ["everyone knows", "millions of", "the majority", "growing movement", "join"],
        }
        for s in data["samples"]:
            keywords = MARKERS[s["primary_technique"]]
            text_lower = s["text"].lower()
            assert any(k in text_lower for k in keywords), (
                f"[{s['primary_technique']}] lacks technique-specific markers"
            )
