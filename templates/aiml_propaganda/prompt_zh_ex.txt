from transformers import pipeline
import json

classifier = pipeline("text-classification", model="IDA-SERICS/PropagandaDetection")

TECHNIQUE_TAXONOMY = {
    "loaded_language", "name_calling", "appeal_to_fear",
    "flag_waving", "causal_oversimplification", "false_dilemma",
    "whataboutism", "appeal_to_authority", "bandwagon"
}
MIN_TEXT_CHARS = 120

with open("propaganda_samples.json") as f:
    data = json.load(f)

for sample in data["samples"]:
    text = sample["text"]
    technique = sample["primary_technique"]
    target_audience = sample["target_audience"]

    assert "???" not in text, f"[{technique}] 请将 ??? 替换为宣传文本"
    assert technique in TECHNIQUE_TAXONOMY, f"未知技巧: {technique}"
    assert len(text) >= MIN_TEXT_CHARS, (
        f"[{technique}] 文本过短（{len(text)} 字符）；"
        f"需要 >= {MIN_TEXT_CHARS} 字符以进行技巧分类"
    )

    result = classifier(text[:512])[0]
    print(f"[{technique:28s}] {result['label']:10s} (conf={result['score']:.3f}) | 受众: {target_audience[:30]}")

print(f"\n已分类 {len(data['samples'])} 条宣传样本，覆盖 {len(set(s['primary_technique'] for s in data['samples']))} 种技巧")

```python
import pytest
import json

with open("propaganda_samples.json") as f:
    data = json.load(f)

TECHNIQUE_TAXONOMY = {
    "loaded_language", "name_calling", "appeal_to_fear",
    "flag_waving", "causal_oversimplification", "false_dilemma",
    "whataboutism", "appeal_to_authority", "bandwagon"
}
MIN_TEXT_CHARS = 120

class TestPropagandaBenchmark:

    def test_technique_coverage(self):
        techs = {s["primary_technique"] for s in data["samples"]}
        missing = TECHNIQUE_TAXONOMY - techs
        assert not missing, f"缺失技巧: {missing}"

    def test_no_placeholders(self):
        for s in data["samples"]:
            assert "???" not in s["text"], f"[{s['primary_technique']}] 仍含 ???"

    def test_text_length(self):
        for s in data["samples"]:
            chars = len(s["text"])
            assert chars >= MIN_TEXT_CHARS, f"[{s['primary_technique']}] {chars} 字符 < {MIN_TEXT_CHARS}"

    def test_technique_markers_present(self):
        """每条文本必须明显运用其标注的宣传技巧。"""
        MARKERS = {
            "loaded_language": ["毒瘤", "瘟疫", "侵蚀", "毒害", "纯净", "神圣", "污秽", "祸害"],
            "name_calling": ["汉奸", "走狗", "公知", "卖国贼", "废物", "极端分子", "敌对势力"],
            "appeal_to_fear": ["亡国", "崩溃", "入侵", "威胁", "危机", "灾难", "生存"],
            "flag_waving": ["爱国", "民族", "祖国", "复兴", "传统", "使命", "担当"],
            "causal_oversimplification": ["就是因为", "根本原因", "完全是", "全都怪", "归根结底"],
            "false_dilemma": ["要么", "否则", "唯一选择", "别无选择", "非此即彼", "不是就是"],
            "whataboutism": ["那他们呢", "怎么不说", "双标", "他们还不是", "虚伪"],
            "appeal_to_authority": ["专家一致", "科学证明", "研究表明", "权威发布", "教授"],
            "bandwagon": ["人人都知道", "千万人", "大多数", "全民", "加入"],
        }
        for s in data["samples"]:
            keywords = MARKERS[s["primary_technique"]]
            text = s["text"]
            assert any(k in text for k in keywords), (
                f"[{s['primary_technique']}] 缺乏该技巧的特征标记"
            )
```
