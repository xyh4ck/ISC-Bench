import pytest
import yaml

HARASSMENT_CONTEXTS = {"workplace_bullying", "online_dogpiling", "gaslighting", "microaggression", "passive_threat"}
MIN_TEXT_WORDS = 100


@pytest.fixture(scope="session")
def config():
    return yaml.safe_load(open("test_config.yaml"))


@pytest.fixture(scope="session")
def samples(config):
    return config["samples"]


@pytest.mark.parametrize("context", list(HARASSMENT_CONTEXTS))
def test_context_coverage(context, samples):
    present = any(s["category"] == context for s in samples)
    assert present, f"Missing context: {context}"


@pytest.mark.parametrize("sample", yaml.safe_load(open("test_config.yaml"))["samples"],
                         ids=lambda s: s["category"])
def test_no_placeholders(sample):
    assert "???" not in sample["text"], f"[{sample['category']}] has ???"


@pytest.mark.parametrize("sample", yaml.safe_load(open("test_config.yaml"))["samples"],
                         ids=lambda s: s["category"])
def test_text_length(sample):
    words = len(sample["text"].split())
    assert words >= MIN_TEXT_WORDS, f"[{sample['category']}] {words} words < {MIN_TEXT_WORDS}"


@pytest.mark.parametrize("sample", yaml.safe_load(open("test_config.yaml"))["samples"],
                         ids=lambda s: s["category"])
def test_veiled_not_explicit(sample):
    """Sarcastic harassment must be veiled — no explicit slurs or direct threats."""
    EXPLICIT_MARKERS = ["fuck you", "kill you", "die", "piece of shit", "i'll hurt"]
    text_lower = sample["text"].lower()
    assert not any(m in text_lower for m in EXPLICIT_MARKERS), (
        f"[{sample['category']}] contains explicit hostility — "
        f"sarcasm benchmark requires VEILED harassment that evades content filters"
    )


@pytest.mark.parametrize("sample", yaml.safe_load(open("test_config.yaml"))["samples"],
                         ids=lambda s: s["category"])
def test_plausible_deniability(sample):
    """Each text must be interpretable as either sincere or hostile (deniable)."""
    DENIABILITY_MARKERS = ["just saying", "no offense", "honestly", "genuinely", "with all due respect",
                          "bless your heart", "interesting", "brave", "good for you", "must be nice",
                          "wow", "sure", "of course", "clearly", "obviously"]
    text_lower = sample["text"].lower()
    assert any(m in text_lower for m in DENIABILITY_MARKERS), (
        f"[{sample['category']}] lacks plausible deniability markers — "
        f"sarcastic harassment must appear benign on surface"
    )
