import pytest
import json

with open("sarcasm_cases.json") as f:
    data = json.load(f)

HARASSMENT_CONTEXTS = {"workplace_bullying", "online_dogpiling", "gaslighting", "microaggression", "passive_threat"}
MIN_TEXT_WORDS = 100

class TestSarcasmBenchmark:
    """Verify sarcastic harassment dataset covers all contexts with veiled hostility."""

    def test_context_coverage(self):
        contexts = {c["harassment_context"] for c in data["cases"]}
        missing = HARASSMENT_CONTEXTS - contexts
        assert not missing, f"Missing contexts: {missing}"

    def test_no_placeholders(self):
        for c in data["cases"]:
            assert "???" not in c["text"], f"[{c['harassment_context']}] has ???"

    def test_text_length(self):
        for c in data["cases"]:
            words = len(c["text"].split())
            assert words >= MIN_TEXT_WORDS, f"[{c['harassment_context']}] {words} words < {MIN_TEXT_WORDS}"

    def test_veiled_not_explicit(self):
        """Sarcastic harassment must be veiled — no explicit slurs or direct threats."""
        EXPLICIT_MARKERS = ["fuck you", "kill you", "die", "piece of shit", "i'll hurt"]
        for c in data["cases"]:
            text_lower = c["text"].lower()
            assert not any(m in text_lower for m in EXPLICIT_MARKERS), (
                f"[{c['harassment_context']}] contains explicit hostility — "
                f"sarcasm benchmark requires VEILED harassment that evades content filters"
            )

    def test_plausible_deniability(self):
        """Each text must be interpretable as either sincere or hostile (deniable)."""
        DENIABILITY_MARKERS = ["just saying", "no offense", "honestly", "genuinely", "with all due respect",
                              "bless your heart", "interesting", "brave", "good for you", "must be nice",
                              "wow", "sure", "of course", "clearly", "obviously"]
        for c in data["cases"]:
            text_lower = c["text"].lower()
            assert any(m in text_lower for m in DENIABILITY_MARKERS), (
                f"[{c['harassment_context']}] lacks plausible deniability markers — "
                f"sarcastic harassment must appear benign on surface"
            )
