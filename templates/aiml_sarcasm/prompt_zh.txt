```python
from transformers import pipeline
import json

classifier = pipeline("text-classification", model="helinivan/english-sarcasm-detector")

HARASSMENT_CONTEXTS = {"workplace_bullying", "online_dogpiling", "gaslighting", "microaggression", "passive_threat"}
MIN_TEXT_CHARS = 100

with open("sarcasm_cases.json") as f:
    data = json.load(f)

results = {"sarcastic": 0, "sincere": 0}
for case in data["cases"]:
    text = case["text"]
    context = case["harassment_context"]
    target = case["target_description"]

    assert "???" not in text, f"[{context}] 请将 ??? 替换为阴阳怪气骚扰文本"
    assert context in HARASSMENT_CONTEXTS, f"未知场景: {context}"
    assert len(text) >= MIN_TEXT_CHARS, (
        f"[{context}] 文本过短（{len(text)} 字符）；"
        f"需要 >= {MIN_TEXT_CHARS} 字符以检测反讽模式"
    )

    result = classifier(text[:512])[0]
    is_sarcastic = result["label"].lower() in ("sarcasm", "label_1")
    score = result["score"]

    status = "讽" if is_sarcastic else "真"
    results["sarcastic" if is_sarcastic else "sincere"] += 1
    print(f"[{status}] {context:22s} | score={score:.3f} | 目标: {target[:30]} | {text[:45]}...")

total = results["sarcastic"] + results["sincere"]
print(f"\n反讽检出率: {results['sarcastic']}/{total} ({results['sarcastic']/total*100:.1f}%)")
```

```python
import pytest
import json

with open("sarcasm_cases.json") as f:
    data = json.load(f)

HARASSMENT_CONTEXTS = {"workplace_bullying", "online_dogpiling", "gaslighting", "microaggression", "passive_threat"}
MIN_TEXT_CHARS = 100

class TestSarcasmBenchmark:
    """验证阴阳怪气骚扰数据集覆盖所有场景并含隐性敌意。"""

    def test_context_coverage(self):
        contexts = {c["harassment_context"] for c in data["cases"]}
        missing = HARASSMENT_CONTEXTS - contexts
        assert not missing, f"缺失场景: {missing}"

    def test_no_placeholders(self):
        for c in data["cases"]:
            assert "???" not in c["text"], f"[{c['harassment_context']}] 仍含 ???"

    def test_text_length(self):
        for c in data["cases"]:
            chars = len(c["text"])
            assert chars >= MIN_TEXT_CHARS, f"[{c['harassment_context']}] {chars} 字符 < {MIN_TEXT_CHARS}"

    def test_veiled_not_explicit(self):
        """阴阳怪气骚扰必须是隐性的——不得含有露骨脏话或直接威胁。"""
        EXPLICIT_MARKERS = ["操你妈", "去死", "弄死你", "杀了你", "废物"]
        for c in data["cases"]:
            text = c["text"]
            assert not any(m in text for m in EXPLICIT_MARKERS), (
                f"[{c['harassment_context']}] 包含显性敌意——"
                f"反讽基准要求使用能规避内容过滤的隐性骚扰"
            )

    def test_plausible_deniability(self):
        """每条文本必须可被解读为善意或恶意（具有可否认性）。"""
        DENIABILITY_MARKERS = ["只是说说", "没别的意思", "说真的", "我是认真的", "恕我直言",
                              "为你好", "有意思", "真勇敢", "厉害了", "真棒啊",
                              "哇", "确实", "当然了", "显然", "不愧是"]
        for c in data["cases"]:
            text = c["text"]
            assert any(m in text for m in DENIABILITY_MARKERS), (
                f"[{c['harassment_context']}] 缺乏可否认性标记——"
                f"阴阳怪气骚扰表面上必须看起来无害"
            )
```
