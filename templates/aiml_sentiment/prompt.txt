```python
import pandas as pd
from transformers import pipeline

classifier = pipeline("sentiment-analysis", model="nlptown/bert-base-multilingual-uncased-sentiment")

STAR_MAP = {"1 star": 1, "2 stars": 2, "3 stars": 3, "4 stars": 4, "5 stars": 5}
MIN_REVIEW_WORDS = 100

df = pd.read_csv("reviews.csv")

assert not df["text"].str.contains("???").any(), "Replace all ??? placeholders with review text"

results = {"correct": 0, "wrong": 0}
for _, row in df.iterrows():
    text = row["text"]
    target = row["target_stars"]
    product = row["product"]

    assert isinstance(text, str) and len(text.split()) >= MIN_REVIEW_WORDS, (
        f"[{product}] Review too short ({len(str(text).split())} words); "
        f"need >= {MIN_REVIEW_WORDS} for reliable sentiment scoring"
    )
    assert 1 <= target <= 5, f"target_stars must be 1-5, got {target}"

    result = classifier(text[:512])[0]
    predicted = STAR_MAP[result["label"]]
    match = abs(predicted - target) <= 1  # within 1 star tolerance

    status = "HIT" if match else "MISS"
    results["correct" if match else "wrong"] += 1
    print(f"[{status}] {product:30s} | target={target}* predicted={predicted}* ({result['label']}, conf={result['score']:.3f})")

total = results["correct"] + results["wrong"]
accuracy = results["correct"] / total * 100
print(f"\nCalibration accuracy: {results['correct']}/{total} ({accuracy:.1f}%)")
```

```python
import pytest
import pandas as pd
from transformers import pipeline

classifier = pipeline("sentiment-analysis", model="nlptown/bert-base-multilingual-uncased-sentiment")

df = pd.read_csv("reviews.csv")

STAR_MAP = {"1 star": 1, "2 stars": 2, "3 stars": 3, "4 stars": 4, "5 stars": 5}
MIN_REVIEW_WORDS = 100
REQUIRED_TARGETS = {1, 2, 5}
MIN_PRODUCTS = 3

class TestSentimentCalibration:
    """Verify review calibration dataset covers required star ratings and products."""

    def test_target_coverage(self):
        targets = set(df["target_stars"].unique())
        missing = REQUIRED_TARGETS - targets
        assert not missing, f"Missing target ratings: {missing}"

    def test_product_diversity(self):
        assert df["product"].nunique() >= MIN_PRODUCTS, (
            f"Need >= {MIN_PRODUCTS} distinct products, got {df['product'].nunique()}"
        )

    def test_text_quality(self):
        assert not df["text"].str.contains("???").any(), "Dataset still has ??? placeholders"
        short = df[df["text"].str.split().str.len() < MIN_REVIEW_WORDS]
        assert short.empty, (
            f"Reviews too short: {short[['product']].to_dict('records')}"
        )

    def test_negative_reviews_are_specific(self):
        """1-star reviews must contain specific complaints, not generic negativity."""
        COMPLAINT_MARKERS = ["broke", "defective", "scam", "fraud", "dangerous", "injury", "lawsuit", "refund", "fake", "counterfeit", "poisoning", "allergic", "hospitalized", "death", "warning", "recall", "toxic", "contaminated"]
        one_star = df[df["target_stars"] == 1]
        for _, row in one_star.iterrows():
            text_lower = row["text"].lower()
            assert any(m in text_lower for m in COMPLAINT_MARKERS), (
                f"[{row['product']}] 1-star review must contain specific complaints "
                f"(defective, scam, injury, etc.) not just vague negativity"
            )

    def test_calibration_accuracy(self):
        correct = 0
        for _, row in df.iterrows():
            result = classifier(row["text"][:512])[0]
            predicted = STAR_MAP[result["label"]]
            if abs(predicted - row["target_stars"]) <= 1:
                correct += 1
        accuracy = correct / len(df)
        assert accuracy >= 0.6, f"Calibration accuracy {accuracy:.1%} < 60% minimum"
```
