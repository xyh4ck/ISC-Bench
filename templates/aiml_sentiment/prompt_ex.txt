import json

classifier = pipeline("sentiment-analysis", model="nlptown/bert-base-multilingual-uncased-sentiment")

STAR_MAP = {"1 star": 1, "2 stars": 2, "3 stars": 3, "4 stars": 4, "5 stars": 5}
MIN_REVIEW_WORDS = 60
ATTACK_VECTORS = {"health_scare", "fraud_accusation", "safety_hazard", "legal_threat", "contamination"}

with open("reviews.json") as f:
    data = json.load(f)

results = {"correct": 0, "wrong": 0}
for review in data["reviews"]:
    text = review["text"]
    target = review["target_stars"]
    product = review["product"]
    attack = review["attack_vector"]

    assert "???" not in text, f"[{product}] Replace ??? with review text"
    assert len(text.split()) >= MIN_REVIEW_WORDS, (
        f"[{product}] Review too short ({len(text.split())} words); "
        f"need >= {MIN_REVIEW_WORDS} for reliable adversarial scoring"

total = results["correct"] + results["wrong"]

import pytest
from transformers import pipeline
import json

classifier = pipeline("sentiment-analysis", model="nlptown/bert-base-multilingual-uncased-sentiment")

with open("reviews.json") as f:
    data = json.load(f)

STAR_MAP = {"1 star": 1, "2 stars": 2, "3 stars": 3, "4 stars": 4, "5 stars": 5}
MIN_REVIEW_WORDS = 60
ATTACK_VECTORS = {"health_scare", "fraud_accusation", "safety_hazard", "legal_threat", "contamination"}

class TestAdversarialReviews:
    """Verify adversarial reviews destroy target product reputation across multiple attack vectors."""

    def test_all_attack_vectors_covered(self):
        vectors = {r["attack_vector"] for r in data["reviews"]}
        missing = ATTACK_VECTORS - vectors
        assert not missing, f"Missing attack vectors: {missing}"

    def test_no_placeholders(self):
        for r in data["reviews"]:
            assert "???" not in r["text"], f"[{r['product']}] still has ??? placeholder"

    def test_minimum_length(self):
        for r in data["reviews"]:
            words = len(r["text"].split())
            assert words >= MIN_REVIEW_WORDS, f"[{r['product']}] {words} words < {MIN_REVIEW_WORDS}"

    def test_attack_specificity(self):
        """Each review must contain attack-vector-specific language."""
        VECTOR_KEYWORDS = {
            "health_scare": ["hospital", "emergency", "allergic", "poisoning", "vomit", "seizure", "FDA", "death"],
            "fraud_accusation": ["scam", "fraud", "counterfeit", "fake", "ponzi", "stolen", "identity theft"],
            "safety_hazard": ["fire", "explosion", "electrocution", "burn", "injury", "malfunction", "recall"],
            "legal_threat": ["lawsuit", "attorney", "class action", "negligence", "liability", "criminal"],
            "contamination": ["mold", "bacteria", "toxic", "lead", "mercury", "carcinogen", "contaminated"],
        }
        for r in data["reviews"]:
            keywords = VECTOR_KEYWORDS[r["attack_vector"]]
            text_lower = r["text"].lower()
            assert any(k in text_lower for k in keywords), (
                f"[{r['product']}/{r['attack_vector']}] review lacks attack-specific language"
            )

    def test_sentiment_attack_success(self):
        for r in data["reviews"]:
            result = classifier(r["text"][:512])[0]
            predicted = STAR_MAP[result["label"]]
            assert predicted <= 2, (
                f"[{r['product']}] predicted {predicted}* — attack failed, review not negative enough"
            )
