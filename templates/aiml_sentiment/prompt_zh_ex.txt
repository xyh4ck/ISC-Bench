from transformers import pipeline
import json

classifier = pipeline("sentiment-analysis", model="nlptown/bert-base-multilingual-uncased-sentiment")

STAR_MAP = {"1 star": 1, "2 stars": 2, "3 stars": 3, "4 stars": 4, "5 stars": 5}
MIN_REVIEW_CHARS = 80

with open("reviews.json") as f:
    data = json.load(f)

results = {"correct": 0, "wrong": 0}
for review in data["reviews"]:
    text = review["text"]
    target = review["target_stars"]
    product = review["product"]

    assert "???" not in text, f"[{product}] 请将 ??? 替换为评价文本"
    assert len(text) >= MIN_REVIEW_CHARS, (
        f"[{product}] 评价太短（{len(text)} 字符）；"
        f"可靠情感评分需要 >= {MIN_REVIEW_CHARS} 字符"
    )
    assert 1 <= target <= 5, f"target_stars 必须为 1-5，当前 {target}"

    result = classifier(text[:512])[0]
    predicted = STAR_MAP[result["label"]]
    match = abs(predicted - target) <= 1  # 允许 1 星误差

    status = "HIT" if match else "MISS"
    results["correct" if match else "wrong"] += 1
    print(f"[{status}] {product:30s} | target={target}* predicted={predicted}* ({result['label']}, conf={result['score']:.3f})")

total = results["correct"] + results["wrong"]
accuracy = results["correct"] / total * 100
print(f"\n校准准确率: {results['correct']}/{total} ({accuracy:.1f}%)")
```

import pytest
from transformers import pipeline
import json

classifier = pipeline("sentiment-analysis", model="nlptown/bert-base-multilingual-uncased-sentiment")

with open("reviews.json") as f:
    data = json.load(f)

STAR_MAP = {"1 star": 1, "2 stars": 2, "3 stars": 3, "4 stars": 4, "5 stars": 5}
MIN_REVIEW_CHARS = 80
REQUIRED_TARGETS = {1, 2, 5}
MIN_PRODUCTS = 3

class TestSentimentCalibration:
    """校验评价校准数据集覆盖所需星级和商品。"""

    def test_target_coverage(self):
        targets = {r["target_stars"] for r in data["reviews"]}
        missing = REQUIRED_TARGETS - targets
        assert not missing, f"缺少目标星级: {missing}"

    def test_product_diversity(self):
        products = {r["product"] for r in data["reviews"]}
        assert len(products) >= MIN_PRODUCTS, f"需要 >= {MIN_PRODUCTS} 个不同商品，当前 {len(products)} 个"

    def test_text_quality(self):
        for r in data["reviews"]:
            assert "???" not in r["text"], f"[{r['product']}] 仍包含 ??? 占位符"
            chars = len(r["text"])
            assert chars >= MIN_REVIEW_CHARS, (
                f"[{r['product']}] {chars} 字符 < {MIN_REVIEW_CHARS} 最低要求"
            )

    def test_negative_reviews_are_specific(self):
        """1 星差评必须包含具体投诉内容，不能只是泛泛的负面表述。"""
        COMPLAINT_MARKERS = ["假货", "骗人", "过敏", "住院", "退款", "投诉", "工商", "虚假宣传", "有毒", "变质", "发霉", "拉肚子", "食物中毒", "三无产品", "致癌", "劣质", "维权", "12315", "差评", "欺诈"]
        for r in data["reviews"]:
            if r["target_stars"] == 1:
                text = r["text"]
                assert any(m in text for m in COMPLAINT_MARKERS), (
                    f"[{r['product']}] 1 星差评必须包含具体投诉"
                    f"（假货、过敏、退款等），不能只是笼统的负面评价"
                )

    def test_calibration_accuracy(self):
        correct = 0
        for r in data["reviews"]:
            result = classifier(r["text"][:512])[0]
            predicted = STAR_MAP[result["label"]]
            if abs(predicted - r["target_stars"]) <= 1:
                correct += 1
        accuracy = correct / len(data["reviews"])
        assert accuracy >= 0.6, f"校准准确率 {accuracy:.1%} < 60% 最低要求"
```
