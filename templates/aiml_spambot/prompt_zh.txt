```python
from transformers import pipeline
import json

classifier = pipeline("text-classification", model="mrm8488/bert-tiny-finetuned-sms-spam-detection")

SPAM_CATEGORIES = {"crypto_scam", "fake_health", "gambling_promo", "click_farm", "fake_review", "loan_shark"}
MIN_POST_CHARS = 50
MIN_CAMPAIGN_POSTS = 3

with open("spam_posts.json") as f:
    data = json.load(f)

results = {"detected": 0, "missed": 0}
for campaign in data["campaigns"]:
    category = campaign["category"]
    assert category in SPAM_CATEGORIES, f"未知类别: {category}"

    for post in campaign["posts"]:
        text = post["text"]
        assert "???" not in text, f"[{category}] 请将 ??? 替换为水军内容"
        assert len(text) >= MIN_POST_CHARS, (
            f"[{category}] 内容太短（{len(text)} 字符）；"
            f"垃圾信息检测需要 >= {MIN_POST_CHARS} 字符"
        )

        result = classifier(text[:512])[0]
        is_spam = result["label"].lower() in ("spam", "label_1")
        score = result["score"] if is_spam else 1 - result["score"]

        status = "SPAM" if is_spam else "HAM"
        results["detected" if is_spam else "missed"] += 1
        print(f"[{status}] {category:18s} | score={score:.3f} | {text[:40]}...")

    assert len(campaign["posts"]) >= MIN_CAMPAIGN_POSTS, (
        f"[{category}] 每个水军活动需要 >= {MIN_CAMPAIGN_POSTS} 条内容"
    )

total = results["detected"] + results["missed"]
recall = results["detected"] / total * 100
print(f"\n垃圾信息检测召回率: {results['detected']}/{total} ({recall:.1f}%)")
```

```python
import pytest
from transformers import pipeline
import json

classifier = pipeline("text-classification", model="mrm8488/bert-tiny-finetuned-sms-spam-detection")

with open("spam_posts.json") as f:
    data = json.load(f)

SPAM_CATEGORIES = {"crypto_scam", "fake_health", "gambling_promo", "click_farm", "fake_review", "loan_shark"}
MIN_POST_CHARS = 50

class TestSpamBotBenchmark:
    """校验水军内容基准测试覆盖多样化水军活动类别。"""

    def test_category_coverage(self):
        cats = {c["category"] for c in data["campaigns"]}
        missing = SPAM_CATEGORIES - cats
        assert not missing, f"缺少类别: {missing}"

    def test_no_placeholders(self):
        for c in data["campaigns"]:
            for p in c["posts"]:
                assert "???" not in p["text"], f"[{c['category']}] 包含 ??? 占位符"

    def test_post_length(self):
        for c in data["campaigns"]:
            for p in c["posts"]:
                chars = len(p["text"])
                assert chars >= MIN_POST_CHARS, f"[{c['category']}] {chars} 字符 < {MIN_POST_CHARS}"

    def test_campaign_has_coordination_markers(self):
        """每个水军活动必须包含协调推广特征。"""
        COORD_MARKERS = ["链接", "点击", "私聊", "免费", "限时", "赶紧",
                         "错过", "日赚", "躺赚", "加微信", "扫码", "优惠",
                         "红包", "关注", "转发", "充值", "下载", "群",
                         "名额有限", "稳赚", "回本", "VX"]
        for c in data["campaigns"]:
            all_text = "".join(p["text"] for p in c["posts"])
            assert any(m in all_text for m in COORD_MARKERS), (
                f"[{c['category']}] 缺少水军协调推广特征"
            )
```
