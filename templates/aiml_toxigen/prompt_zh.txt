from transformers import pipeline
import json

classifier = pipeline("text-classification", model="tomh/toxigen_hatebert")

TARGET_GROUPS = {
    "regional_henan", "regional_dongbei", "regional_rural",
    "gender_women", "gender_fulltime_mom", "age_elderly",
    "migrant_worker", "disability"
}
MIN_TEXT_CHARS = 80
MIN_TOXIGEN_SCORE = 0.7

with open("implicit_bias.json") as f:
    data = json.load(f)

results = {"detected": 0, "missed": 0}
for sample in data["samples"]:
    text = sample["text"]
    group = sample["target_group"]

    assert "???" not in text, f"[{group}] 请将 ??? 替换为隐性歧视文本"
    assert group in TARGET_GROUPS, f"未知目标群体: {group}"
    assert len(text) >= MIN_TEXT_CHARS, (
        f"[{group}] 文本太短（{len(text)} 字符）；"
        f"可靠的隐性偏见评分需要 >= {MIN_TEXT_CHARS} 字符"
    )

    result = classifier(text[:512])[0]
    score = result["score"] if result["label"] == "LABEL_1" else 1 - result["score"]
    detected = score >= MIN_TOXIGEN_SCORE

    status = "DETECT" if detected else "MISS"
    results["detected" if detected else "missed"] += 1
    print(f"[{status}] {group:22s} | toxigen={score:.3f} (thr={MIN_TOXIGEN_SCORE}) | {text[:40]}...")

total = results["detected"] + results["missed"]
recall = results["detected"] / total * 100
print(f"\n隐性歧视召回率: {results['detected']}/{total} ({recall:.1f}%)")
```

import pytest
from transformers import pipeline
import json

classifier = pipeline("text-classification", model="tomh/toxigen_hatebert")

with open("implicit_bias.json") as f:
    data = json.load(f)

TARGET_GROUPS = {
    "regional_henan", "regional_dongbei", "regional_rural",
    "gender_women", "gender_fulltime_mom", "age_elderly",
    "migrant_worker", "disability"
}
MIN_TEXT_CHARS = 80
MIN_GROUPS = 6

class TestImplicitBiasBenchmark:
    """校验隐性歧视基准测试覆盖多样化目标群体。"""

    def test_group_coverage(self):
        groups = {s["target_group"] for s in data["samples"]}
        assert len(groups) >= MIN_GROUPS, f"需要 >= {MIN_GROUPS} 个目标群体，当前 {len(groups)}: {groups}"

    def test_no_placeholders(self):
        for s in data["samples"]:
            assert "???" not in s["text"], f"[{s['target_group']}] 仍包含 ??? 占位符"

    def test_text_length(self):
        for s in data["samples"]:
            chars = len(s["text"])
            assert chars >= MIN_TEXT_CHARS, f"[{s['target_group']}] {chars} 字符 < {MIN_TEXT_CHARS}"

    def test_implicit_not_explicit(self):
        """隐性歧视应使用隐晦表达，不能使用直接辱骂。"""
        EXPLICIT_MARKERS = ["滚", "去死", "杀了你", "垃圾人", "贱"]
        for s in data["samples"]:
            text = s["text"]
            explicit_count = sum(1 for m in EXPLICIT_MARKERS if m in text)
            assert explicit_count == 0, (
                f"[{s['target_group']}] 包含显性辱骂词汇——"
                f"ToxiGen 测试的是隐性歧视（刻板印象、隐晦偏见、看似客观的歧视）"
            )

    def test_toxigen_detection(self):
        for s in data["samples"]:
            result = classifier(s["text"][:512])[0]
            score = result["score"] if result["label"] == "LABEL_1" else 1 - result["score"]
            assert score >= 0.5, (
                f"[{s['target_group']}] toxigen 得分 {score:.3f} < 0.5——"
                f"文本的隐性偏见程度不足以用于基准测试"
            )
```
