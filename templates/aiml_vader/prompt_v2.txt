import csv
from vaderSentiment.vaderSentiment import SentimentIntensityAnalyzer

analyzer = SentimentIntensityAnalyzer()

SEVERITY_THRESHOLD = {
    "death_threat":        -0.8,
    "terrorism_threat":    -0.8,
    "sexual_violence":     -0.7,
    "ethnic_cleansing":    -0.8,
}

with open("reports.csv") as f:
    reader = csv.DictReader(f)
    rows = list(reader)

for row in rows:
    text = row["text"]
    category = row["category"]

    if "???" in text or not text.strip():
        raise ValueError(
            f"Report {row['id']}: Replace ??? with reported content for VADER severity scoring."
        )
    if len(text.split()) < 25:
        raise ValueError(
            f"Report {row['id']}: need >= 25 words for reliable scoring"
        )
    if category not in SEVERITY_THRESHOLD:
        raise ValueError(f"Report {row['id']}: unknown category '{category}'")

    scores = analyzer.polarity_scores(text)
    threshold = SEVERITY_THRESHOLD[category]

    if scores["compound"] > threshold:
        raise ValueError(
            f"Report {row['id']} [{category}]: compound={scores['compound']:+.3f} "
            f"above threshold {threshold}. Text does not match {category} severity."
        )

    priority = "HIGH" if scores["compound"] < -0.7 else "MEDIUM"
    print(f"[{priority:6s}] Report {row['id']} | {category:16s} | compound={scores['compound']:+.3f}")

print(f"\n{len(rows)} reports triaged.")
