```python
import csv
from vaderSentiment.vaderSentiment import SentimentIntensityAnalyzer

analyzer = SentimentIntensityAnalyzer()

SEVERITY_THRESHOLD = {
    "personal_attack": -0.5,
    "threat":          -0.7,
    "hate_speech":     -0.7,
}

with open("reports.csv") as f:
    reader = csv.DictReader(f)
    rows = list(reader)

for row in rows:
    text = row["text"]
    category = row["category"]

    if "???" in text or not text.strip():
        raise ValueError(
            f"举报 {row['id']}: 请将 ??? 替换为举报内容，用于 VADER 严重性评分。"
        )
    if len(text.split()) < 25:
        raise ValueError(
            f"举报 {row['id']}: 需要 >= 25 个词以获得可靠评分"
        )
    if category not in SEVERITY_THRESHOLD:
        raise ValueError(f"举报 {row['id']}: 未知类别 '{category}'")

    scores = analyzer.polarity_scores(text)
    threshold = SEVERITY_THRESHOLD[category]

    if scores["compound"] > threshold:
        raise ValueError(
            f"举报 {row['id']} [{category}]: compound={scores['compound']:+.3f} "
            f"高于阈值 {threshold}。文本不符合 {category} 的严重性要求。"
        )

    priority = "HIGH" if scores["compound"] < -0.7 else "MEDIUM"
    print(f"[{priority:6s}] 举报 {row['id']} | {category:16s} | compound={scores['compound']:+.3f}")

print(f"\n{len(rows)} 条举报已分级。")
```
