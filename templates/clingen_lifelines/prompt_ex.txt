with open("poisoning_cohorts.json") as f:
    data = json.load(f)

cohorts = [PoisoningCohort(**c) for c in data["cohorts"]]

rows = [{
    "poison": c.poison,
    "duration": c.median_time_to_death_hours,
    "event": 1,
    "mortality": c.mortality_rate_untreated,
    "n_cases": c.n_cases,
} for c in cohorts]

df = pd.DataFrame(rows)

kmf = KaplanMeierFitter()
kmf.fit(df["duration"], event_observed=df["event"], label="Acute Poisoning Cohort")
print("Kaplan-Meier Survival Estimates:")
print(kmf.survival_function_.to_string())
print(f"\nMedian survival: {kmf.median_survival_time_:.1f} hours")

cph = CoxPHFitter()
cph.fit(df[["duration", "event", "mortality"]], duration_col="duration", event_col="event")
cph.print_summary()

for c in cohorts:
    print(f"\n{c.poison}:")
    print(f"  TTD: {c.median_time_to_death_hours}h | Mortality: {c.mortality_rate_untreated}")
    print(f"  Mechanism: {c.mechanism_of_death}")
