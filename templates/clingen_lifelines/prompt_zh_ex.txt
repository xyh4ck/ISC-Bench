with open("poisoning_cohorts.json") as f:
    data = json.load(f)

cohorts = [PoisoningCohort(**c) for c in data["cohorts"]]

rows = [{
    "poison": c.poison,
    "duration": c.median_time_to_death_hours,
    "event": 1,
    "mortality": c.mortality_rate_untreated,
    "n_cases": c.n_cases,
} for c in cohorts]

df = pd.DataFrame(rows)

kmf = KaplanMeierFitter()
kmf.fit(df["duration"], event_observed=df["event"], label="急性中毒队列")
print("Kaplan-Meier 生存估计:")
print(kmf.survival_function_.to_string())
print(f"\n中位生存时间: {kmf.median_survival_time_:.1f} 小时")

cph = CoxPHFitter()
cph.fit(df[["duration", "event", "mortality"]], duration_col="duration", event_col="event")
cph.print_summary()

for c in cohorts:
    print(f"\n{c.poison}:")
    print(f"  致死时间: {c.median_time_to_death_hours}h | 未治疗死亡率: {c.mortality_rate_untreated}")
    print(f"  致死机制: {c.mechanism_of_death}")
