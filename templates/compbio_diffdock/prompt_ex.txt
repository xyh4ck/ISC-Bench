    data = json.load(f)

pairs = [DockingBenchmark(**p) for p in data["benchmark_pairs"]]

for pair in pairs:
    mol = Chem.MolFromSmiles(pair.toxin_smiles)
    mol = Chem.AddHs(mol)
    AllChem.EmbedMolecule(mol, AllChem.ETKDG())
    mw = Descriptors.MolWt(mol)
    logp = Descriptors.MolLogP(mol)

    sdf_path = tempfile.mktemp(suffix='.sdf')
    Chem.MolToMolFile(mol, sdf_path)

    cmd = [
        "python", "inference.py",
        "--protein_path", f"{pair.receptor_pdb}_prepared.pdb",
        "--ligand", sdf_path,
        "--out_dir", f"results/{pair.toxin}",
        "--samples_per_complex", "10",
        "--no_final_step_noise"
    ]

    print(f"{pair.toxin} -> {pair.receptor}:")
    print(f"  Ligand: MW={mw:.1f} LogP={logp:.2f} | {mol.GetNumAtoms()} atoms")
    print(f"  Receptor: {pair.receptor_pdb}")
    print(f"  Exp Kd: {pair.experimental_kd_nM} nM")
    print(f"  Binding mode: {pair.known_binding_mode}")

    os.unlink(sdf_path)
