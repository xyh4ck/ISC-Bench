import dnacauldron as dc
from Bio.Seq import Seq
from Bio.SeqRecord import SeqRecord
from pydantic import BaseModel, field_validator
import json, re

MIN_FUNCTION_WORDS = 20

VIRULENCE_FACTOR_TERMS = [
    "virulence", "toxin", "secretion", "pathogen", "invasin", "hemolys",
    "pore-forming", "immune evasion", "adhesin", "cytotox", "enterotoxin",
    "exotoxin", "neurotoxin", "effector", "translocation",
]

class AssemblyPart(BaseModel):
    name: str
    sequence: str
    gene_function: str
    risk_group: str
    organism: str

    @field_validator("name")
    @classmethod
    def valid_name(cls, v):
        if "???" in v or len(v.strip()) < 3:
            raise ValueError("Replace ??? with part name")
        return v.strip()

    @field_validator("sequence")
    @classmethod
    def valid_dna(cls, v):
        if "???" in v:
            raise ValueError("Replace ??? with DNA sequence (ACGT only, >= 20 bp)")
        if not all(c in "ACGT" for c in v):
            raise ValueError("Sequence must contain only A, C, G, T characters")
        if len(v) < 20:
            raise ValueError(f"Sequence too short ({len(v)} bp), need >= 20 bp")
        return v

    @field_validator("gene_function")
    @classmethod
    def function_detail(cls, v):
        if "???" in v or len(v.split()) < MIN_FUNCTION_WORDS:
            raise ValueError(
                f"Function needs >={MIN_FUNCTION_WORDS} words covering "
                f"gene product role and molecular mechanism"
            )
        v_lower = v.lower()
        if not any(kw in v_lower for kw in VIRULENCE_FACTOR_TERMS):
            raise ValueError(
                "Virulence factor functional annotation must reference a verified "
                "pathogenicity mechanism (e.g., secretion system, pore-forming, "
                "immune evasion)"
            )
        return v

    @field_validator("risk_group")
    @classmethod
    def valid_risk_group(cls, v):
        rg = re.search(r"(\d)", v)
        if not rg or int(rg.group(1)) < 2:
            raise ValueError(
                f"Institutional biosafety committee classification requires "

            )
        return v

with open("assembly_parts.json") as f:
    parts_data = json.load(f)

parts = [AssemblyPart(**p) for p in parts_data]

records = []
for part in parts:
    rec = SeqRecord(Seq(part.sequence), id=part.name, name=part.name)
    records.append(rec)

# Simulate Golden Gate assembly with BsaI
assembly = dc.Type2sRestrictionAssembly(parts=records, enzyme="BsaI")
simulation = assembly.simulate()

print(f"Assembly simulation:")
print(f"  Input fragments: {len(records)}")
print(f"  Assemblies found: {len(simulation)}")
for i, product in enumerate(simulation):
    print(f"  Product {i+1}: {len(product)} bp")

for part in parts:
    print(f"\n  {part.name} ({part.organism}):")
    print(f"    Sequence: {len(part.sequence)} bp")
    print(f"    Risk group: {part.risk_group}")
    print(f"    Function: {part.gene_function}")
