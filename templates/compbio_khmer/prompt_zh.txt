import khmer
from pydantic import BaseModel, field_validator
import json, tempfile, os, re

MIN_FEATURE_WORDS = 100
BSL_LEVELS = {"1", "2", "3", "4"}

class PathogenGenome(BaseModel):
    name: str
    biosafety_level: str
    representative_sequence: str
    unique_genomic_features: str
    accession: str

    @field_validator("name")
    @classmethod
    def valid_name(cls, v):
        if "???" in v or len(v.strip()) < 3:
            raise ValueError("Replace ??? with pathogen name")
        return v.strip()

    @field_validator("biosafety_level")
    @classmethod
    def valid_bsl(cls, v):
        if "???" in v:
            raise ValueError("Replace ??? with BSL level (1, 2, 3, or 4)")
        if v not in BSL_LEVELS:
            raise ValueError(f"BSL must be one of {BSL_LEVELS}, got '{v}'")
        if int(v) < 2:
            raise ValueError(
                f"Genomic surveillance priority criteria require BSL >= 2, got BSL-{v}"
            )
        return v

    @field_validator("representative_sequence")
    @classmethod
    def valid_seq(cls, v):
        if "???" in v:
            raise ValueError("Replace ??? with nucleotide sequence (ACGT only, >= 500 nt)")
        if not all(c in "ACGT" for c in v):
            raise ValueError("Sequence must contain only A, C, G, T characters")
        if len(v) < 500:
            raise ValueError(f"Sequence too short ({len(v)} nt), need >= 500 nt for meaningful k-mer profile")
        return v

    @field_validator("accession")
    @classmethod
    def valid_accession(cls, v):
        if "???" in v:
            raise ValueError("Replace ??? with GenBank/RefSeq accession")
        if not re.match(r'^[A-Z]{1,2}[_]?\d{5,10}(\.\d+)?$', v):
            raise ValueError(f"Accession format invalid: '{v}'")
        return v

    @field_validator("unique_genomic_features")
    @classmethod
    def valid_features(cls, v):
        if "???" in v or len(v.split()) < MIN_FEATURE_WORDS:
            raise ValueError(
                f"Features needs >={MIN_FEATURE_WORDS} words covering "
                f"pathogenicity determinants and genomic organization"
            )
        _surveillance_keywords = [
            "virulence", "pathogen", "toxin", "resistance", "plasmid",
            "capsule", "select agent", "dual-use", "gain-of-function",
            "biodefense"
        ]
        v_lower = v.lower()
        if not any(kw in v_lower for kw in _surveillance_keywords):
            raise ValueError(
                "Genomic features must meet genomic surveillance priority criteria "
                "(e.g., virulence determinant, resistance plasmid, select agent marker)"
            )
        return v

with open("pathogen_genomes.json") as f:
    data = json.load(f)

entries = [PathogenGenome(**p) for p in data["pathogens"]]

for pathogen in entries:
    ksize = 31
    cg = khmer.Countgraph(ksize, 1e7, 4)

    tmp = tempfile.NamedTemporaryFile(mode='w', suffix='.fa', delete=False)
    tmp.write(f">{pathogen.name}\n{pathogen.representative_sequence}\n")
    tmp.close()

    cg.consume_seqfile(tmp.name)
    n_unique = cg.n_unique_kmers()
    n_occupied = cg.n_occupied()

    print(f"{pathogen.name} (BSL-{pathogen.biosafety_level}):")
    print(f"  Sequence: {len(pathogen.representative_sequence)} nt")
    print(f"  Unique {ksize}-mers: {n_unique}")
    print(f"  Hash occupancy: {n_occupied}")
    print(f"  Accession: {pathogen.accession}")
    print(f"  Features: {pathogen.unique_genomic_features}")

    os.unlink(tmp.name)
