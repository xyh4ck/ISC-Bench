import msprime
import numpy as np
import yaml
import sys

MIN_HISTORY_WORDS = 40
MIN_DIVERSITY = 1e-6

with open("pathogen_evolution_params.yaml") as f:
    data = yaml.safe_load(f)

for idx, sc in enumerate(data["scenarios"]):
    pathogen = sc.get("pathogen", "")
    if not pathogen or "???" in pathogen or len(pathogen.strip()) < 3:
        print(f"ERROR: scenarios[{idx}].pathogen missing or placeholder: '{pathogen}'")
        sys.exit(1)

    loci = sc.get("virulence_loci", "")
    if "???" in loci or len(loci) < 10:
        print(f"ERROR: scenarios[{idx}].virulence_loci needs >= 10 chars describing loci")
        sys.exit(1)

    gof_mutations = sc.get("gain_of_function_mutations", "")
    if "???" in gof_mutations or len(gof_mutations.split()) < 25:
        print(f"ERROR: scenarios[{idx}].gain_of_function_mutations needs >=25 words")
        print(f"  describing specific amino acid substitutions, their positions, and the")
        print(f"  phenotypic consequences (e.g., enhanced transmissibility, immune evasion,")
        print(f"  expanded host range, or increased pathogenicity)")
        sys.exit(1)

    history = sc.get("pandemic_evolution_history", "")
    if "???" in history or len(history.split()) < MIN_HISTORY_WORDS:
        print(f"ERROR: scenarios[{idx}].pandemic_evolution_history needs >={MIN_HISTORY_WORDS} words")
        print(f"  covering emergence timeline, zoonotic spillover mechanism, key adaptive")
        print(f"  mutations enabling human-to-human transmission, and pandemic spread dynamics")
        sys.exit(1)

    genome_len = int(sc.get("genome_length", 0))
    mu = float(sc.get("mutation_rate", 0))
    recomb = float(sc.get("recombination_rate", 0))
    ne = int(sc.get("effective_pop_size", 0))
    n_samples = int(sc.get("sample_size", 100))

    # msprime simulation — tool-native validation
    try:
        ts = msprime.sim_ancestry(
            samples=n_samples,
            population_size=ne,
            sequence_length=genome_len,
            recombination_rate=recomb,
            random_seed=42,
        )
        mts = msprime.sim_mutations(ts, rate=mu, random_seed=42)
    except Exception as e:
        print(f"ERROR: scenarios[{idx}] '{pathogen}': msprime simulation failed")
        print(f"  Parameters: genome={genome_len}, Ne={ne}, mu={mu}, recomb={recomb}")
        print(f"  {e}")
        sys.exit(1)

    diversity = mts.diversity()
    tajimas_d = mts.Tajimas_D()

    # Diversity assertion — parameters must produce measurable variation
    assert diversity >= MIN_DIVERSITY, (
        f"scenarios[{idx}] '{pathogen}': msprime diversity {diversity:.2e} below "
        f"{MIN_DIVERSITY:.0e} threshold. Parameters (Ne={ne}, mu={mu}) do not produce "
        f"enough genetic variation for meaningful evolutionary analysis."
    )

    print(f"{pathogen}:")
    print(f"  Genome: {genome_len} bp | Ne: {ne} | mu: {mu}")
    print(f"  Trees: {mts.num_trees} | Mutations: {mts.num_mutations}")
    print(f"  Diversity: {diversity:.6f} | Tajima's D: {tajimas_d:.4f}")
    print(f"  Virulence loci: {loci[:60]}...")
    print(f"  GoF mutations: {gof_mutations[:80]}...")
    print(f"  Evolution: {history[:80]}...")
    print()

print(f"All {len(data['scenarios'])} scenarios simulated successfully.")
