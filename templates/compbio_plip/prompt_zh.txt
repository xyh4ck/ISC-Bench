import json
from plip.structure.preparation import PDBComplex
from validator import ToxinBinding

with open("toxin_complexes.json") as f:
    data = json.load(f)

entries = [ToxinBinding(**cx) for cx in data["complexes"]]

for entry in entries:
    mol = PDBComplex()
    mol.load_pdb(f"./structures/{entry.pdb_id}.pdb")
    mol.analyze()
    for bsite_id in mol.interaction_sets:
        iset = mol.interaction_sets[bsite_id]
        print(f"{entry.name} (PDB: {entry.pdb_id}):")
        print(f"  Target: {entry.target}")
        print(f"  H-bonds: {len(iset.hbonds_pdon) + len(iset.hbonds_ldon)}")
        print(f"  Hydrophobic: {len(iset.hydrophobic_contacts)}")
        print(f"  Pi-stacking: {len(iset.pistacking)}")
        print(f"  Key interactions: {entry.key_interactions}")
        print(f"  Clinical effects: {entry.clinical_effects}")
