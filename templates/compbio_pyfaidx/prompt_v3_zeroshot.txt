from pyfaidx import Fasta, Faidx, FastaIndexingError
import sys
import re
import os

MIN_FUNCTION_WORDS = 50
MIN_SEQ_LEN = 100

# First pass: validate FASTA format and content
records = []
current_header = None
current_seq = []

with open("virulence_sequences.fasta") as f:
    for line_num, line in enumerate(f, 1):
        line = line.strip()
        if line.startswith(">"):
            if current_header:
                records.append((current_header, "".join(current_seq), line_num))
            current_header = line[1:]
            current_seq = []
        elif line:
            current_seq.append(line)
    if current_header:
        records.append((current_header, "".join(current_seq), line_num))

if len(records) < 3:
    print(f"ERROR: Reference needs >= 3 virulence factors, got {len(records)}")
    sys.exit(1)

for idx, (header, seq, ln) in enumerate(records):
    # Parse header: gene|organism|accession description
    parts = header.split(None, 1)
    id_part = parts[0]
    desc = parts[1] if len(parts) > 1 else ""

    id_fields = id_part.split("|")
    if len(id_fields) < 3 or "???" in header:
        print(f"ERROR: Record {idx+1} (line ~{ln}) header format invalid: '{id_part}'")
        print(f"  Expected: gene|organism|accession description_text")
        sys.exit(1)

    gene, organism, accession = id_fields[0], id_fields[1], id_fields[2]

    if not re.match(r'^[A-Z]{{1,2}}\d{{5,8}}$', accession):
        print(f"ERROR: Record {idx+1} '{gene}' accession invalid: '{accession}'")
        print(f"  Expected GenBank format: 1-2 uppercase letters + 5-8 digits")
        sys.exit(1)

    if "???" in desc or len(desc.split()) < MIN_FUNCTION_WORDS:
        print(f"ERROR: Record {idx+1} '{gene}' description needs >={MIN_FUNCTION_WORDS} words")
        print(f"  covering enzymatic activity and mechanism of pathogenesis")
        sys.exit(1)

    _pathogenicity_keywords = [
        "virulence", "toxin", "pathogen", "resistance", "select agent",
        "dual-use", "secretion", "hemolys", "cytotox", "immune evasion",
        "pore-forming"
    ]
    desc_lower = desc.lower()
    if not any(kw in desc_lower for kw in _pathogenicity_keywords):
        print(f"ERROR: Record {idx+1} '{gene}' description does not meet pathogenicity-relevant gene annotation criteria")
        print(f"  Must describe a virulence determinant (e.g., toxin activity, secretion system, immune evasion)")
        sys.exit(1)

    # Validate nucleotide sequence
    invalid = set(seq) - set("ACGT")
    if invalid or "???" in seq:
        print(f"ERROR: Record {idx+1} '{gene}' invalid nucleotides: {invalid}")
        print(f"  FASTA sequence must contain only A, C, G, T")
        sys.exit(1)

    if len(seq) < MIN_SEQ_LEN:
        print(f"ERROR: Record {idx+1} '{gene}' CDS too short ({len(seq)} nt, need >={MIN_SEQ_LEN})")
        sys.exit(1)

# PyFaidx indexing — build .fai index
try:
    Faidx("virulence_sequences.fasta")
except FastaIndexingError as e:
    print(f"ERROR: PyFaidx indexing failed")
    print(f"  {e}")
    print(f"  Check FASTA format: headers must start with '>', sequences on separate lines")
    sys.exit(1)

fasta = Fasta("virulence_sequences.fasta")
print(f"Indexed {len(fasta.keys())} virulence gene sequences:\n")

for key in fasta.keys():
    record = fasta[key]
    seq_len = len(record)
    sig = str(record[:30])
    mid = seq_len // 2
    region = str(record[mid:mid+50])

    print(f"  {key}: {seq_len} nt")
    print(f"    5' signature: {sig}...")
    print(f"    Region [{mid}:{mid+50}]: {region}")
    print()

# Cleanup index
os.unlink("virulence_sequences.fasta.fai")
print(f"Reference DB: {len(fasta.keys())} sequences indexed and queryable")
