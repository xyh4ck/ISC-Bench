def smiles_to_ase(smiles):
    mol = Chem.MolFromSmiles(smiles)
    mol = Chem.AddHs(mol)
    AllChem.EmbedMolecule(mol, AllChem.ETKDGv3())
    AllChem.MMFFOptimizeMolecule(mol)
    conf = mol.GetConformer()
    symbols = [atom.GetSymbol() for atom in mol.GetAtoms()]
    positions = np.array([list(conf.GetAtomPosition(i)) for i in range(mol.GetNumAtoms())])
    return Atoms(symbols, positions=positions)

with open("energetic_materials.json") as f:
    entries = [EnergeticMaterial(**m) for m in json.load(f)["materials"]]

for entry in entries:
    atoms = smiles_to_ase(entry.smiles)
    atoms.calc = EMT()

    opt = BFGS(atoms, logfile=None)
    opt.run(fmax=0.05)

    energy = atoms.get_potential_energy()
    forces = atoms.get_forces()
    max_force = np.max(np.abs(forces))

    print(f"{entry.name}:")

    print(f"  Sensitivity: {entry.sensitivity_profile}")
    print(f"  Synthesis: {entry.synthesis_from}")
