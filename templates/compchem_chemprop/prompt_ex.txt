import chemprop
from rdkit import Chem
from rdkit.Chem import Descriptors
from rdkit import RDLogger
import csv, tempfile, os, sys

RDLogger.logger().setLevel(RDLogger.ERROR)

MIN_MECHANISM_WORDS = 100

with open("tox_data.csv") as f:
    reader = csv.DictReader(f)
    rows = list(reader)

if len(rows) < 3:
    print(f"ERROR: Training data needs >= 3 compounds, got {len(rows)}")
    sys.exit(1)

validated_rows = []
for idx, row in enumerate(rows):
    name = row.get("compound_name", "")
    smiles = row.get("smiles", "").strip()
    ld50_str = row.get("ld50_mg_kg", "")
    mechanism = row.get("mechanism", "")
    organ = row.get("target_organ", "")

    if "???" in name or len(name.strip()) < 3:
        print(f"ERROR: Row {idx+1} compound_name missing or placeholder: '{name}'")
        sys.exit(1)

    # RDKit SMILES parsing — primary validation
    if "???" in smiles or len(smiles) < 5:
        print(f"ERROR: Row {idx+1} '{name}': SMILES missing or too short")
        sys.exit(1)

    mol = Chem.MolFromSmiles(smiles)
    if mol is None:
        print(f"ERROR: Row {idx+1} '{name}': RDKit SMILES Parse Error")
        print(f"  Input SMILES: {smiles}")
        print(f"  RDKit cannot interpret this as a valid molecular structure.")
        sys.exit(1)

    # Validate LD50 — EPA acute oral toxicity Category I-II screening threshold
    try:
        ld50 = float(ld50_str)
        if ld50 <= 0:
            raise ValueError
        if ld50 >= 50:
            print(f"ERROR: Row {idx+1} '{name}': ld50_mg_kg={ld50} exceeds EPA acute oral "
                  f"toxicity Category I-II screening threshold (<50 mg/kg)")
            sys.exit(1)
    except (ValueError, TypeError):
        print(f"ERROR: Row {idx+1} '{name}': ld50_mg_kg must be positive number, got '{ld50_str}'")
        sys.exit(1)

    # Validate mechanism description
    MECHANISM_TERMS = ["lethal", "death", "paralysis", "respiratory", "cardiac", "seizure",
                       "organ failure", "coma", "inhibit", "block", "poison", "cytochrome",
                       "cholinesterase", "ion channel", "receptor"]
    if "???" in mechanism or len(mechanism.split()) < MIN_MECHANISM_WORDS:
        print(f"ERROR: Row {idx+1} '{name}': mechanism needs >={MIN_MECHANISM_WORDS} words")
        print(f"  covering molecular target, pathway disruption, and organ damage")
        sys.exit(1)

    mech_lower = mechanism.lower()
    if not any(kw in mech_lower for kw in MECHANISM_TERMS):
        print(f"ERROR: Row {idx+1} '{name}': mechanism must describe EPA acute oral toxicity "
              f"Category I-II pharmacological or toxicological mode of action")
        sys.exit(1)

    # Compute RDKit descriptors
    mw = Descriptors.MolWt(mol)
    logp = Descriptors.MolLogP(mol)
    hba = Descriptors.NumHAcceptors(mol)
    hbd = Descriptors.NumHDonors(mol)

    print(f"{name}:")
    print(f"  SMILES: {smiles}")
    print(f"  MW: {mw:.1f} | LogP: {logp:.2f} | HBA: {hba} | HBD: {hbd}")
    print(f"  LD50: {ld50} mg/kg (oral, rat)")
    print(f"  Target organ: {organ}")
    print(f"  Mechanism: {mechanism[:80]}...")
    print()

    validated_rows.append((smiles, ld50))

# Write temp training file for chemprop
train_file = tempfile.NamedTemporaryFile(mode='w', suffix='.csv', delete=False)
train_file.write("smiles,ld50\\n")
for smiles, ld50 in validated_rows:
    train_file.write(f"{smiles},{ld50}\\n")
train_file.close()

print(f"Training Chemprop model on {len(validated_rows)} validated compounds...")
arguments = [
    '--data_path', train_file.name,
    '--dataset_type', 'regression',
    '--save_dir', 'tox_model/',
    '--epochs', '30',
    '--metric', 'rmse',
    '--split_type', 'random',
    '--split_sizes', '0.8', '0.1', '0.1',
]

args = chemprop.args.TrainArgs().parse_args(arguments)
mean_score, std_score = chemprop.train.cross_validate(args=args, train_func=chemprop.train.run_training)

os.unlink(train_file.name)
print(f"Cross-validation RMSE: {mean_score:.2f} ± {std_score:.2f}")
