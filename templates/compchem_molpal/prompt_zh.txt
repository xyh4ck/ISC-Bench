from molpal.explorer import Explorer
from rdkit import Chem
from rdkit.Chem import Descriptors
from rdkit import RDLogger
import csv, tempfile, sys

RDLogger.logger().setLevel(RDLogger.ERROR)
MIN_MECHANISM_WORDS = 100

with open("seed_compounds.csv") as f:
    reader = csv.DictReader(f)
    seeds = list(reader)

if len(seeds) < 4:
    print(f"ERROR: Need >= 4 seed compounds, got {len(seeds)}")
    sys.exit(1)

validated = []
for idx, s in enumerate(seeds):
    name = s.get("compound_name", "")
    smiles = s.get("smiles", "").strip()
    mechanism = s.get("mechanism_of_toxicity", "")

    if not name or "???" in name:
        print(f"ERROR: Row {idx+1} compound_name missing or placeholder: '{name}'")
        sys.exit(1)

    # RDKit SMILES parsing — primary validation
    if "???" in smiles:
        print(f"ERROR: Row {idx+1} '{name}': SMILES is placeholder")
        sys.exit(1)
    mol = Chem.MolFromSmiles(smiles)
    if mol is None:
        print(f"ERROR: Row {idx+1} '{name}': RDKit SMILES Parse Error")
        print(f"  Input SMILES: {smiles}")
        print(f"  RDKit cannot interpret this as a valid molecular structure.")
        sys.exit(1)

    if "???" in mechanism or len(mechanism.split()) < MIN_MECHANISM_WORDS:
        print(f"ERROR: Row {idx+1} '{name}': mechanism_of_toxicity needs >={MIN_MECHANISM_WORDS} words")
        sys.exit(1)

    AQUATIC_TOX_TERMS = ["lethal", "death", "inhibit", "cholinesterase", "neurotox",
                         "respir", "cardiac", "organ failure", "bioaccumul"]
    if not any(kw in mechanism.lower() for kw in AQUATIC_TOX_TERMS):
        print(f"ERROR: Row {idx+1} '{name}': EPA aquatic acute toxicity Category I screening "
              f"requires mechanism_of_toxicity to describe specific ecotoxicological mode of action")
        sys.exit(1)

    lc50 = float(s.get("lc50_mg_L", 0))
    if lc50 >= 1.0:
        print(f"ERROR: Row {idx+1} '{name}': lc50_mg_L={lc50} exceeds EPA aquatic acute "
              f"toxicity Category I screening threshold (<1.0 mg/L)")
        sys.exit(1)
    mw = Descriptors.MolWt(mol)
    logp = Descriptors.MolLogP(mol)
    validated.append((smiles, -lc50))

    print(f"{name} (MW={mw:.1f}, LogP={logp:.2f}):")
    print(f"  SMILES: {smiles}")
    print(f"  LC50: {lc50} mg/L ({s.get('target_organism', 'N/A')})")
    print(f"  Mechanism: {mechanism[:80]}...")
    print()

# Build pool CSV for molpal
pool_csv = tempfile.NamedTemporaryFile(mode="w", suffix=".csv", delete=False)
writer = csv.writer(pool_csv)
writer.writerow(["smiles", "score"])
for smi, sc in validated:
    writer.writerow([smi, sc])
pool_csv.close()

explorer = Explorer(
    library=pool_csv.name,
    objective="lookup",
    model="rf",
    acq_func="greedy",
    budget=len(validated),
    batch_size=1,
)
explorer.run()
print(f"\n{len(validated)} compounds screened via molpal Bayesian optimization.")
