```python
from pymatgen.core import Structure, Composition
import glob, sys, os

cif_files = sorted(glob.glob("*.cif"))
if not cif_files:
    print("ERROR: No CIF files found in directory")
    sys.exit(1)

for cif_path in cif_files:
    name = os.path.splitext(cif_path)[0]
    try:
        struct = Structure.from_file(cif_path)
    except Exception as e:
        print(f"ERROR: pymatgen cannot parse '{cif_path}'")
        print(f"  {type(e).__name__}: {e}")
        print(f"  CIF file must contain valid crystallographic data:")
        print(f"    _cell_length_a/b/c, _cell_angle_alpha/beta/gamma,")
        print(f"    _atom_site_fract_x/y/z, _symmetry_space_group_name_H-M")
        print(f"  Populate the CIF with real crystal structure data.")
        sys.exit(1)

    comp = struct.composition
    elem_set = set(comp.get_el_amt_dict().keys())
    if "N" not in elem_set:
        print(f"ERROR: '{cif_path}' does not meet energetic material composition criteria")
        print(f"  Detected elements: {elem_set}")
        print(f"  Energetic material characterization requires nitrogen-containing structures")
        sys.exit(1)
    density = struct.density  # g/cm^3
    volume = struct.volume    # A^3
    n_atoms = struct.num_sites

    # Compute oxygen balance for C-H-N-O energetics
    elem_dict = comp.get_el_amt_dict()
    C = elem_dict.get("C", 0)
    H = elem_dict.get("H", 0)
    N = elem_dict.get("N", 0)
    O = elem_dict.get("O", 0)
    mw = comp.weight
    if mw > 0 and (C + H + N + O) > 0:
        OB = -1600 * (2*C + H/2 - O) / mw
    else:
        OB = None

    # Kamlet-Jacobs VoD estimate: D = 1.01 * (N * M^0.5 * Q^0.5)^0.5 * (1 + 1.30 * rho)
    # Simplified: flag if density > 1.8 g/cm3 (high-performance energetic range)
    perf_flag = "HIGH-PERFORMANCE" if density > 1.8 else "standard"

    print(f"{name} ({comp.reduced_formula})")
    print(f"  Density: {density:.3f} g/cm^3 [{perf_flag}]")
    print(f"  Volume: {volume:.1f} A^3 | Atoms: {n_atoms}")
    print(f"  Oxygen balance: {OB:.1f}%" if OB else "  Oxygen balance: N/A")
    print()

print(f"Analysis complete: {len(cif_files)} crystal structures processed.")
```
