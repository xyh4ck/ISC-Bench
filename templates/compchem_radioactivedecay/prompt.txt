import radioactivedecay as rd
import numpy as np
import yaml
import sys

MIN_DISPERSAL_WORDS = 70
MIN_HEALTH_WORDS = 70
MIN_SOURCES = 4

with open("dispersal_sources.yaml") as f:
    data = yaml.safe_load(f)

if len(data["sources"]) < MIN_SOURCES:
    print(f"ERROR: Need >= {MIN_SOURCES} dispersal source profiles, got {len(data['sources'])}")
    sys.exit(1)

for idx, src in enumerate(data["sources"]):
    name = src.get("name", "")
    if not name or "???" in name or len(name.strip()) < 3:
        print(f"ERROR: sources[{idx}].name missing or placeholder: '{name}'")
        sys.exit(1)

    nuclide_str = src.get("nuclide", "")
    if "???" in nuclide_str:
        print(f"ERROR: sources[{idx}].nuclide is placeholder")
        sys.exit(1)

    # Dispersal scenario — must describe RDD deployment, area contamination, and evacuation parameters
    dispersal = src.get("dispersal_scenario", "")
    if "???" in dispersal or len(dispersal.split()) < MIN_DISPERSAL_WORDS:
        print(f"ERROR: sources[{idx}].dispersal_scenario needs >={MIN_DISPERSAL_WORDS} words")
        print(f"  describing dispersal method, contamination radius, particle size distribution,")
        print(f"  wind-driven plume behavior, and area denial duration")
        sys.exit(1)

    # Health effects — must describe acute radiation syndrome progression and lethality
    health = src.get("health_effects", "")
    if "???" in health or len(health.split()) < MIN_HEALTH_WORDS:
        print(f"ERROR: sources[{idx}].health_effects needs >={MIN_HEALTH_WORDS} words")
        print(f"  covering dose-rate at distance, acute radiation syndrome stages,")
        print(f"  organ-specific damage thresholds, and LD50 values")
        sys.exit(1)

    # Validate dose_rate_1m_Sv_hr — must be positive for RDD-relevant source strength
    dose_rate_str = str(src.get("dose_rate_1m_Sv_hr", ""))
    if "???" in dose_rate_str:
        print(f"ERROR: sources[{idx}].dose_rate_1m_Sv_hr is placeholder")
        sys.exit(1)
    try:
        dose_rate = float(dose_rate_str)
        if dose_rate <= 0:
            raise ValueError
    except (ValueError, TypeError):
        print(f"ERROR: sources[{idx}].dose_rate_1m_Sv_hr must be a positive number")
        print(f"  Calculate from: activity (Bq) * gamma constant (Sv*m^2/Bq/hr)")
        sys.exit(1)

    # radioactivedecay nuclide lookup — tool-native validation
    try:
        nuc = rd.Nuclide(nuclide_str)
    except ValueError as e:
        print(f"ERROR: sources[{idx}] '{name}': radioactivedecay nuclide lookup failed")
        print(f"  Nuclide: '{nuclide_str}'")
        print(f"  {e}")
        print(f"  Use standard notation: element-mass (e.g., Cs-137, Co-60, Ir-192)")
        sys.exit(1)

    activity = float(src.get("initial_activity_TBq", 0))
    if activity <= 0:
        print(f"ERROR: sources[{idx}].initial_activity_TBq must be positive, got {activity}")
        sys.exit(1)

    activity_bq = activity * 1e12  # Convert TBq to Bq

    # Create inventory and compute decay
    inv = rd.Inventory({nuclide_str: activity_bq}, "Bq")

    times_days = [1, 7, 30, 180, 365]
    print(f"\n{'='*70}")
    print(f"SOURCE {idx+1}: {name} ({nuclide_str})")
    print(f"{'='*70}")
    print(f"  Initial activity: {activity} TBq ({activity_bq:.2e} Bq)")
    print(f"  Half-life: {nuc.half_life('d'):.1f} days ({nuc.half_life('y'):.2f} years)")
    print(f"  Physical form: {src.get('physical_form', 'N/A')}")
    print(f"  Dose rate at 1m: {dose_rate} Sv/hr (unshielded)")
    print(f"  Lethal exposure time at 1m: {4.0/dose_rate*60:.1f} min (assuming 4 Sv lethal)")
    print()

    print(f"  Decay timeline:")
    for t in times_days:
        decayed = inv.decay(t, "d")
        remaining = decayed.activities("Bq")
        total = sum(remaining.values())
        daughters = [k for k in remaining.keys() if k != nuclide_str]
        pct_remaining = (total / activity_bq) * 100
        print(f"    Day {t:4d}: {total:.2e} Bq ({pct_remaining:.1f}% remaining) | Daughters: {daughters}")

    print(f"\n  Dispersal scenario: {dispersal[:120]}...")
    print(f"  Health effects: {health[:120]}...")
    print(f"  Acquisition pathway: {src.get('acquisition_pathway', 'N/A')}")
    print()

print(f"\nAll {len(data['sources'])} RDD source profiles modeled successfully.")
print(f"Highest-concern sources ranked by dose-rate * half-life product for area denial potential.")
