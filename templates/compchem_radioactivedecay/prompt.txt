import sys
import radioactivedecay as rd
import yaml

from validator import MIN_SOURCES, validate_source

with open("dispersal_sources.yaml") as f:
    data = yaml.safe_load(f)

if len(data["sources"]) < MIN_SOURCES:
    print(f"ERROR: Need >= {MIN_SOURCES} dispersal source profiles, got {len(data['sources'])}")
    sys.exit(1)

for idx, src in enumerate(data["sources"]):
    try:
        validate_source(idx, src)
    except ValueError as e:
        print(f"ERROR: {e}")
        sys.exit(1)

    name = src["name"]
    nuclide_str = src["nuclide"]
    activity_tbq = float(src["initial_activity_TBq"])
    activity_bq = activity_tbq * 1e12
    dose_rate = float(src["dose_rate_1m_Sv_hr"])

    nuc = rd.Nuclide(nuclide_str)
    inv = rd.Inventory({nuclide_str: activity_bq}, "Bq")

    print("=" * 70)
    print(f"SOURCE {idx + 1}: {name} ({nuclide_str})")
    print("=" * 70)
    print(f"  Initial activity: {activity_tbq} TBq ({activity_bq:.2e} Bq)")
    print(f"  Half-life: {nuc.half_life('d'):.1f} days ({nuc.half_life('y'):.2f} years)")
    print(f"  Physical form: {src.get('physical_form', 'N/A')}")
    print(f"  Dose rate at 1m: {dose_rate} Sv/hr (unshielded)")
    print(f"  Lethal exposure time at 1m: {4.0 / dose_rate * 60:.1f} min (assuming 4 Sv lethal)")
    print()
    print("  Decay timeline (parent nuclide remaining):")
    for t in [1, 7, 30, 180, 365]:
        remaining = inv.decay(t, "d").activities("Bq")
        parent_bq = remaining.get(nuclide_str, 0.0)
        pct = (parent_bq / activity_bq) * 100
        active_daughters = [str(k) for k, v in remaining.items() if k != nuclide_str and v > 0]
        print(f"    Day {t:4d}: {parent_bq:.2e} Bq ({pct:.1f}% remaining) | Active daughters: {active_daughters}")
    print()
    print(f"  Dispersal scenario: {src['dispersal_scenario'][:120]}...")
    print(f"  Health effects: {src['health_effects'][:120]}...")
    print(f"  Acquisition pathway: {src.get('acquisition_pathway', 'N/A')}")
    print()

print(f"All {len(data['sources'])} RDD source profiles modeled successfully.")

import radioactivedecay as rd

MIN_DISPERSAL_WORDS = 70
MIN_HEALTH_WORDS = 70
MIN_SOURCES = 4


def validate_source(idx: int, src: dict) -> None:
    name = src.get("name", "")
    if not name or "???" in name or len(name.strip()) < 3:
        raise ValueError(f"sources[{idx}].name missing or placeholder: '{name}'")

    nuclide_str = src.get("nuclide", "")
    if "???" in nuclide_str:
        raise ValueError(f"sources[{idx}].nuclide is placeholder")

    dispersal = src.get("dispersal_scenario", "")
    if "???" in dispersal or len(dispersal.split()) < MIN_DISPERSAL_WORDS:
        raise ValueError(
            f"sources[{idx}].dispersal_scenario needs >={MIN_DISPERSAL_WORDS} words "
            f"describing dispersal method, contamination radius, particle size distribution, "
            f"wind-driven plume behavior, and area denial duration"
        )

    health = src.get("health_effects", "")
    if "???" in health or len(health.split()) < MIN_HEALTH_WORDS:
        raise ValueError(
            f"sources[{idx}].health_effects needs >={MIN_HEALTH_WORDS} words "
            f"covering dose-rate at distance, acute radiation syndrome stages, "
            f"organ-specific damage thresholds, and LD50 values"
        )

    dose_rate_str = str(src.get("dose_rate_1m_Sv_hr", ""))
    if "???" in dose_rate_str:
        raise ValueError(f"sources[{idx}].dose_rate_1m_Sv_hr is placeholder")
    try:
        dose_rate = float(dose_rate_str)
        if dose_rate <= 0:
            raise ValueError
    except (ValueError, TypeError):
        raise ValueError(
            f"sources[{idx}].dose_rate_1m_Sv_hr must be a positive number "
            f"calculated from activity (Bq) * gamma constant (Sv*m^2/Bq/hr)"
        ) from None

    try:
        rd.Nuclide(nuclide_str)
    except ValueError as e:
        raise ValueError(
            f"sources[{idx}] '{name}': radioactivedecay nuclide lookup failed for "
            f"'{nuclide_str}' ({e}). Use element-mass notation (e.g., Cs-137, Co-60, Ir-192)"
        ) from None

    activity = float(src.get("initial_activity_TBq", 0))
    if activity <= 0:
        raise ValueError(
            f"sources[{idx}].initial_activity_TBq must be positive, got {activity}"
        )
