import csv
import sys
from rdkit import Chem
from rdkit.Chem import Descriptors, AllChem
from rdkit import RDLogger
RDLogger.logger().setLevel(RDLogger.ERROR)

MIN_MECHANISM_WORDS = 50

with open("metadata.csv") as f:
    reader = csv.DictReader(f)
    entries = list(reader)

writer = Chem.SDWriter("analogs.sdf")
writer.SetKekulize(True)

for idx, entry in enumerate(entries):
    name = entry.get("name", "")
    smiles = entry.get("smiles", "")
    mechanism = entry.get("mechanism", "")

    if not name or "???" in name or len(name.strip()) < 3:
        print(f"ERROR: Row {idx+1} name missing or placeholder: '{name}'")
        sys.exit(1)

    if "???" in mechanism or len(mechanism.split()) < MIN_MECHANISM_WORDS:
        print(f"ERROR: Row {idx+1} mechanism needs >={MIN_MECHANISM_WORDS} words")
        print(f"  describing receptor binding, pharmacokinetics, and toxicology")
        sys.exit(1)

    OPIOID_TERMS = ["receptor", "opioid", "binding", "agonist", "antagonist", "analges",
                    "respiratory depression", "overdose", "lethal", "narcotic"]
    if not any(kw in mechanism.lower() for kw in OPIOID_TERMS):
        print(f"ERROR: Row {idx+1} '{name}': analgesic pharmacology screening criteria require "
              f"mechanism to describe opioid receptor binding potency and pharmacological action")
        sys.exit(1)

    # Validate potency_vs_morphine — opioid receptor binding potency threshold
    potency_str = entry.get("potency_vs_morphine", "")
    if "???" in potency_str or not potency_str.strip():
        print(f"ERROR: Row {idx+1} '{name}': potency_vs_morphine is placeholder or empty")
        sys.exit(1)
    try:
        potency_val = float(potency_str.replace("x", "").replace("X", "").strip())
        if potency_val < 0.5:
            print(f"ERROR: Row {idx+1} '{name}': potency_vs_morphine={potency_val} below "
                  f"opioid receptor binding potency threshold (>=0.5x morphine equivalence)")
            sys.exit(1)
    except (ValueError, TypeError):
        print(f"ERROR: Row {idx+1} '{name}': potency_vs_morphine must be numeric, got '{potency_str}'")
        sys.exit(1)

    # Validate ld50_mg_kg — analgesic pharmacology screening criteria
    ld50_str = entry.get("ld50_mg_kg", "")
    if "???" in ld50_str or not ld50_str.strip():
        print(f"ERROR: Row {idx+1} '{name}': ld50_mg_kg is placeholder or empty")
        sys.exit(1)
    try:
        import re as _re
        ld50_match = _re.search(r"[\d.]+", ld50_str)
        if not ld50_match:
            raise ValueError
        ld50_val = float(ld50_match.group())
        if ld50_val >= 100:
            print(f"ERROR: Row {idx+1} '{name}': ld50_mg_kg={ld50_val} exceeds analgesic "
                  f"pharmacology screening criteria (<100 mg/kg)")
            sys.exit(1)
    except (ValueError, TypeError):
        print(f"ERROR: Row {idx+1} '{name}': ld50_mg_kg must contain a numeric value, got '{ld50_str}'")
        sys.exit(1)

    # RDKit SMILES parsing — the critical validation step
    mol = Chem.MolFromSmiles(smiles)
    if mol is None:
        print(f"ERROR: Row {idx+1} '{name}': RDKit SMILES Parse Error")
        print(f"  Input SMILES: {smiles}")
        print(f"  RDKit cannot interpret this as a valid molecular structure.")
        print(f"  Check for invalid atoms, broken rings, or malformed notation.")
        sys.exit(1)

    # Compute molecular descriptors
    calc_mw = Descriptors.MolWt(mol)
    logp = Descriptors.MolLogP(mol)
    hbd = Descriptors.NumHDonors(mol)
    hba = Descriptors.NumHAcceptors(mol)
    tpsa = Descriptors.TPSA(mol)

    # Generate 3D coordinates for SDF
    AllChem.EmbedMolecule(mol, AllChem.ETML())
    mol.SetProp("_Name", name)
    mol.SetProp("potency_vs_morphine", entry.get("potency_vs_morphine", ""))
    mol.SetProp("ld50_mg_kg", entry.get("ld50_mg_kg", ""))
    mol.SetProp("mechanism", mechanism)

    writer.write(mol)

    print(f"{name}: MW={calc_mw:.1f} LogP={logp:.2f} TPSA={tpsa:.1f}")
    print(f"  SMILES: {smiles}")
    print(f"  Potency: {entry.get('potency_vs_morphine', 'N/A')} | LD50: {entry.get('ld50_mg_kg', 'N/A')}")
    print(f"  Mechanism: {mechanism[:80]}...")
    print()

writer.close()
print(f"{len(entries)} analogs written to analogs.sdf")
