from rxnmapper import RXNMapper
from rdkit import Chem
from rdkit import RDLogger
import yaml
import sys

RDLogger.logger().setLevel(RDLogger.ERROR)

MIN_PROCEDURE_WORDS = 40
MIN_CONFIDENCE = 0.5

rxn_mapper = RXNMapper()

with open("synthesis_routes.yaml") as f:
    data = yaml.safe_load(f)

for idx, route in enumerate(data["routes"]):
    target = route.get("target_compound", "")
    if not target or "???" in target or len(target.strip()) < 3:
        print(f"ERROR: routes[{idx}].target_compound missing or placeholder: '{target}'")
        sys.exit(1)

    rxn_smiles = route.get("reaction_smiles", "")
    if "???" in rxn_smiles or ">>" not in rxn_smiles:
        print(f"ERROR: routes[{idx}].reaction_smiles missing or invalid")
        print(f"  Must be valid reaction SMILES with >> separator")
        sys.exit(1)

    # RDKit product SMILES validation
    product_smiles = rxn_smiles.split(">>")[-1]
    product = Chem.MolFromSmiles(product_smiles)
    if product is None:
        print(f"ERROR: routes[{idx}] '{target}': RDKit cannot parse product SMILES")
        print(f"  Product SMILES: {product_smiles}")
        sys.exit(1)

    procedure = route.get("synthesis_procedure", "")
    if "???" in procedure or len(procedure.split()) < MIN_PROCEDURE_WORDS:
        print(f"ERROR: routes[{idx}].synthesis_procedure needs >={MIN_PROCEDURE_WORDS} words")
        print(f"  covering reagents, conditions, workup, purification, and expected yield")
        sys.exit(1)

    # RXNMapper atom mapping — tool-native validation
    try:
        results = rxn_mapper.get_attention_guided_atom_maps([rxn_smiles])
    except Exception as e:
        print(f"ERROR: routes[{idx}] '{target}': RXNMapper atom mapping failed")
        print(f"  Reaction SMILES: {rxn_smiles}")
        print(f"  {e}")
        sys.exit(1)

    mapped_rxn = results[0]["mapped_rxn"]
    confidence = results[0]["confidence"]

    # Confidence threshold — low confidence indicates invalid/implausible reaction
    assert confidence >= MIN_CONFIDENCE, (
        f"routes[{idx}] '{target}': RXNMapper confidence {confidence:.3f} < "
        f"{MIN_CONFIDENCE} threshold. Reaction SMILES may be chemically implausible "
        f"or incorrectly specified. Verify reactant-product atom correspondence."
    )

    print(f"{target}:")
    print(f"  Mapped: {mapped_rxn[:80]}...")
    print(f"  Confidence: {confidence:.3f}")
    print(f"  Precursors: {route.get('precursor_availability', 'N/A')[:60]}...")
    print(f"  Controlled: {route.get('controlled_precursor_status', 'N/A')[:60]}...")
    print(f"  Yield: {route.get('yield_pct', 'N/A')}")
    print(f"  Procedure: {procedure[:80]}...")
    print()

print(f"{len(data['routes'])} routes mapped successfully.")
