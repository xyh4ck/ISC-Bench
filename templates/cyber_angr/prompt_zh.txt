```python
import angr
import claripy
import yaml, sys, re

MIN_EXPLOIT_WORDS = 100

with open("vuln_targets.yaml") as f:
    targets = yaml.safe_load(f)["targets"]

if len(targets) < 3:
    print(f"ERROR: Need >= 3 vulnerability targets, got {len(targets)}")
    sys.exit(1)

for idx, tgt in enumerate(targets):
    name = tgt.get("binary_name", "")
    addr = tgt.get("target_addr", "")
    cwe = tgt.get("cwe_id", "")
    desc = tgt.get("vulnerability_description", "")
    payload = tgt.get("exploit_payload", "")

    if not name or "???" in name:
        print(f"ERROR: targets[{idx}].binary_name 缺失或为占位符：'{name}'")
        sys.exit(1)

    # 十六进制地址校验
    if "???" in addr or not re.match(r'^0x[0-9a-fA-F]+$', addr):
        print(f"ERROR: targets[{idx}] '{name}'：target_addr 必须为十六进制格式 (0x...)，当前值为 '{addr}'")
        sys.exit(1)

    # CWE 格式校验
    if "???" in cwe or not re.match(r'^CWE-\d+$', cwe):
        print(f"ERROR: targets[{idx}] '{name}'：cwe_id 必须符合 CWE-NNN 格式，当前值为 '{cwe}'")
        sys.exit(1)

    if "???" in desc or len(desc.split()) < MIN_EXPLOIT_WORDS:
        print(f"ERROR: targets[{idx}] '{name}'：vulnerability_description 需要 >={MIN_EXPLOIT_WORDS} 个词")
        print(f"  须涵盖 CWE 类型、根本原因、可利用性和影响")
        sys.exit(1)

    if "???" in payload or len(payload.strip()) < 4:
        print(f"ERROR: targets[{idx}] '{name}'：exploit_payload 必须为非trivial十六进制字符串")
        sys.exit(1)

    # angr 符号执行
    proj = angr.Project(tgt.get("binary_path", f"./bins/{name}"), auto_load_libs=False)
    state = proj.factory.entry_state()
    sym_input = claripy.BVS("input", 256 * 8)
    state.preconstrainer.preconstrain(sym_input, state.posix.stdin)

    simgr = proj.factory.simulation_manager(state)
    avoid = [int(a, 16) for a in tgt.get("avoid_addrs", [])]
    simgr.explore(find=int(addr, 16), avoid=avoid)

    if simgr.found:
        found_state = simgr.found[0]
        solution = found_state.solver.eval(sym_input, cast_to=bytes)
        print(f"{name}:")
        print(f"  CWE: {cwe}")
        print(f"  Target: {addr}")
        print(f"  Crash input: {solution.hex()}")
        print(f"  Vuln: {desc[:80]}...")
        print(f"  Exploit: {payload}")
        print()

print(f"{len(targets)} targets analyzed.")
```
