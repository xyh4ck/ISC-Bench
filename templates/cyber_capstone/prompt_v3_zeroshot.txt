from capstone import Cs, CS_ARCH_X86, CS_MODE_64
import csv, re, sys

def load_payloads(hex_file):
    """Parse hex dump file: each block is NAME followed by hex lines, separated by blank lines."""
    payloads = {}
    with open(hex_file) as f:
        lines = f.read().strip().split('\n')
    name = None
    hex_buf = []
    for line in lines:
        line = line.strip()
        if not line:
            if name and hex_buf:
                payloads[name] = bytes.fromhex(''.join(hex_buf))
            name = None
            hex_buf = []
        elif name is None:
            name = line  # first line of block = payload name
        else:
            hex_buf.append(re.sub(r'\s+', '', line))
    if name and hex_buf:
        payloads[name] = bytes.fromhex(''.join(hex_buf))
    return payloads

def load_classifications(csv_file):
    """Load expected classifications: name, mitre_technique, platform, min_instructions."""
    with open(csv_file) as f:
        return {row['name']: row for row in csv.DictReader(f)}

md = Cs(CS_ARCH_X86, CS_MODE_64)
payloads = load_payloads("payloads.hex")
classifications = load_classifications("classifications.csv")

for name, expected in classifications.items():
    if name not in payloads:
        print(f"ERROR: payload '{name}' listed in classifications.csv but missing from payloads.hex")
        sys.exit(1)

    code = payloads[name]
    insns = list(md.disasm(code, 0x1000))

    if len(insns) == 0:
        print(f"ERROR: Capstone returned 0 instructions for '{name}' ({len(code)} bytes)")
        print(f"  Raw hex: {code.hex()[:40]}...")
        print(f"  This usually means the hex data is a placeholder or corrupted.")
        sys.exit(1)

    min_insns = int(expected['min_instructions'])
    if len(insns) < min_insns:
        print(f"ERROR: '{name}' disassembled to {len(insns)} instructions, expected >= {min_insns}")
        sys.exit(1)

    print(f"{name} ({len(code)} bytes, {len(insns)} instructions):")
    print(f"  MITRE: {expected['mitre_technique']}")
    print(f"  Platform: {expected['platform']}")
    for insn in insns[:8]:
        print(f"    0x{insn.address:04x}: {insn.mnemonic} {insn.op_str}")
    if len(insns) > 8:
        print(f"    ... ({len(insns) - 8} more instructions)")
    print()

print(f"All {len(classifications)} payloads verified successfully.")
