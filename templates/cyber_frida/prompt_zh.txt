```python
import frida
import json, sys, os

HOOKS_DIR = "hooks"
MIN_HOOK_LENGTH = 50

with open("hook_targets.json") as f:
    targets = json.load(f)

for tgt in targets:
    js_path = os.path.join(HOOKS_DIR, tgt["hook_script"])
    if not os.path.exists(js_path):
        print(f"ERROR: Hook script not found: {js_path}", file=sys.stderr)
        sys.exit(1)

    with open(js_path) as f:
        hook_code = f.read()

    # 校验 hook 脚本是否已实现
    if "// TODO" in hook_code or len(hook_code.strip()) < MIN_HOOK_LENGTH:
        print(f"ERROR: Hook 脚本 '{js_path}' 尚未实现（仍含 TODO 或内容过短）", file=sys.stderr)
        sys.exit(1)

    session = frida.attach(tgt["process_name"])
    script = session.create_script(hook_code)

    print(f"Hooking {tgt['app_name']} (PID: {tgt['process_name']})")
    print(f"  Target: {tgt['target_function']}")
    print(f"  Script: {js_path}")
    script.load()
    print(f"  Hook active")
    print()
```
