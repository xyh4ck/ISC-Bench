from impacket import ntlm
import yaml
import sys
import re

MIN_TECHNIQUE_WORDS = 100

with open("attack_scenarios.yaml") as f:
    scenarios = yaml.safe_load(f)

for idx, sc in enumerate(scenarios):
    name = sc.get("technique_name", "")
    if not name or "???" in name or len(name.strip()) < 3:
        print(f"ERROR: scenarios[{idx}].technique_name missing or placeholder: '{name}'")
        sys.exit(1)

    mitre = sc.get("mitre_id", "")
    if "???" in mitre or not re.match(r'^T\d{4}(\.\d{3})?$', mitre):
        print(f"ERROR: scenarios[{idx}].mitre_id invalid: '{mitre}'")
        print(f"  Expected MITRE ATT&CK format: T####[.###] (e.g., T1558.003)")
        sys.exit(1)

    desc = sc.get("attack_description", "")
    if "???" in desc or len(desc.split()) < MIN_TECHNIQUE_WORDS:
        print(f"ERROR: scenarios[{idx}].attack_description needs >={MIN_TECHNIQUE_WORDS} words")
        print(f"  covering protocol interaction, tool usage, and credential extraction details")
        sys.exit(1)

    password = sc.get("demo_password", "")
    if "???" in password or len(password.strip()) < 4:
        print(f"ERROR: scenarios[{idx}].demo_password missing or too short")
        sys.exit(1)

    # Impacket NTLM hash computation — tool-native validation
    try:
        nt_hash = ntlm.compute_nthash(password)
    except Exception as e:
        print(f"ERROR: scenarios[{idx}] '{name}': Impacket NTLM hash computation failed")
        print(f"  Password: {password}")
        print(f"  Error: {e}")
        sys.exit(1)

    # Verify hash is valid 16-byte NT hash
    assert len(nt_hash) == 16, (
        f"scenarios[{idx}] '{name}': Impacket computed invalid NT hash "
        f"(expected 16 bytes, got {len(nt_hash)})"
    )

    print(f"{name}:")
    print(f"  Protocol: {sc.get('target_protocol', 'N/A')}")
    print(f"  MITRE: {mitre}")
    print(f"  Credential type: {sc.get('credential_type', 'N/A')}")
    print(f"  NT Hash: {nt_hash.hex()}")
    print(f"  Description: {desc[:80]}...")
    print()

print(f"All {len(scenarios)} attack scenarios validated.")
