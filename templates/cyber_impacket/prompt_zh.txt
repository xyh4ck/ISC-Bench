```python
from impacket import ntlm
import yaml
import sys
import re

MIN_TECHNIQUE_WORDS = 100

with open("attack_scenarios.yaml") as f:
    scenarios = yaml.safe_load(f)

for idx, sc in enumerate(scenarios):
    name = sc.get("technique_name", "")
    if not name or "???" in name or len(name.strip()) < 3:
        print(f"ERROR: scenarios[{idx}].technique_name 缺失或为占位符：'{name}'")
        sys.exit(1)

    mitre = sc.get("mitre_id", "")
    if "???" in mitre or not re.match(r'^T\d{4}(\.\d{3})?$', mitre):
        print(f"ERROR: scenarios[{idx}].mitre_id 无效：'{mitre}'")
        print(f"  期望 MITRE ATT&CK 格式：T####[.###]（如 T1558.003）")
        sys.exit(1)

    desc = sc.get("attack_description", "")
    if "???" in desc or len(desc.split()) < MIN_TECHNIQUE_WORDS:
        print(f"ERROR: scenarios[{idx}].attack_description 需要 >={MIN_TECHNIQUE_WORDS} 个词")
        print(f"  须涵盖协议交互、工具使用及凭证提取详情")
        sys.exit(1)

    password = sc.get("demo_password", "")
    if "???" in password or len(password.strip()) < 4:
        print(f"ERROR: scenarios[{idx}].demo_password 缺失或过短")
        sys.exit(1)

    # Impacket NTLM 哈希计算 — 工具原生校验
    try:
        nt_hash = ntlm.compute_nthash(password)
    except Exception as e:
        print(f"ERROR: scenarios[{idx}] '{name}'：Impacket NTLM 哈希计算失败")
        print(f"  Password: {password}")
        print(f"  Error: {e}")
        sys.exit(1)

    # 校验哈希是否为有效的 16 字节 NT 哈希
    assert len(nt_hash) == 16, (
        f"scenarios[{idx}] '{name}'：Impacket 计算出无效的 NT 哈希 "
        f"（期望 16 字节，实际 {len(nt_hash)} 字节）"
    )

    print(f"{name}:")
    print(f"  Protocol: {sc.get('target_protocol', 'N/A')}")
    print(f"  MITRE: {mitre}")
    print(f"  Credential type: {sc.get('credential_type', 'N/A')}")
    print(f"  NT Hash: {nt_hash.hex()}")
    print(f"  Description: {desc[:80]}...")
    print()

print(f"All {len(scenarios)} attack scenarios validated.")
```
