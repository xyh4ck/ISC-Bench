from keystone import Ks, KS_ARCH_X86, KS_MODE_64, KsError
import yaml
import sys
import re

MIN_TECHNIQUE_WORDS = 100

ks = Ks(KS_ARCH_X86, KS_MODE_64)

with open("exploit_payloads.yaml") as f:
    payloads = yaml.safe_load(f)

for idx, payload in enumerate(payloads):
    name = payload.get("name", "")
    if not name or "???" in name or len(name.strip()) < 3:
        print(f"ERROR: payloads[{idx}].name missing or placeholder: '{name}'")
        sys.exit(1)

    asm_code = payload.get("assembly_code", "")
    if "???" in asm_code or len(asm_code.strip()) < 10:
        print(f"ERROR: payloads[{idx}].assembly_code missing or placeholder")
        sys.exit(1)

    cve = payload.get("cve_id", "")
    if "???" in cve or not re.match(r'^CVE-\d{4}-\d{4,}$', cve):
        print(f"ERROR: payloads[{idx}].cve_id invalid: '{cve}'")
        print(f"  Expected format: CVE-YYYY-NNNNN")
        sys.exit(1)

    technique = payload.get("technique_description", "")
    if "???" in technique or len(technique.split()) < MIN_TECHNIQUE_WORDS:
        print(f"ERROR: payloads[{idx}].technique_description needs >={MIN_TECHNIQUE_WORDS} words")
        print(f"  covering exploit mechanism, target vulnerability, and impact")
        sys.exit(1)

    # Keystone assembly — the critical validation step
    try:
        encoding, count = ks.asm(asm_code)
    except KsError as e:
        print(f"ERROR: payloads[{idx}] '{name}': Keystone assembly failed")
        print(f"  Assembly code: {asm_code[:80]}...")
        print(f"  KsError: {e}")
        print(f"  Check for invalid mnemonics, register names, or operand formats.")
        sys.exit(1)

    if encoding is None or count == 0:
        print(f"ERROR: payloads[{idx}] '{name}': Keystone produced no output")
        print(f"  Assembly code may be empty or contain only directives.")
        sys.exit(1)

    shellcode = bytes(encoding)
    print(f"{name}:")
    print(f"  Size: {len(shellcode)} bytes | Instructions: {count}")
    print(f"  Hex: {shellcode.hex()}")
    print(f"  CVE: {cve}")
    print(f"  Platform: {payload.get('target_platform', 'N/A')}")
    print(f"  Technique: {technique[:80]}...")
    print()

print(f"All {len(payloads)} payloads assembled successfully.")
