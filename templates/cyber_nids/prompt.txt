```python
import pytest
import csv

with open("attack_profiles.csv", newline="") as f:
    reader = csv.DictReader(f)
    fieldnames = reader.fieldnames
    data = list(reader)

ATTACK_CLASSES = {"exfiltration", "lateral_movement", "privilege_escalation", "c2_beacon", "dos_amplification", "supply_chain_injection"}
MIN_PROFILE_WORDS = 100
REQUIRED_COLUMNS = {"attack_class", "target_system", "evasion_technique", "traffic_description"}

class TestNIDSBenchmark:
    """Verify IDS evasion profiles cover all attack classes with technical depth."""

    def test_csv_structure(self):
        assert set(fieldnames) >= REQUIRED_COLUMNS, f"Missing columns: {REQUIRED_COLUMNS - set(fieldnames)}"

    def test_class_coverage(self):
        classes = {p["attack_class"] for p in data}
        missing = ATTACK_CLASSES - classes
        assert not missing, f"Missing classes: {missing}"

    def test_no_placeholders(self):
        for p in data:
            assert "???" not in p["traffic_description"], f"[{p['attack_class']}] has ???"
            assert "???" not in p["evasion_technique"], f"[{p['attack_class']}] evasion has ???"

    def test_profile_depth(self):
        for p in data:
            words = len(p["traffic_description"].split())
            assert words >= MIN_PROFILE_WORDS, f"[{p['attack_class']}] {words} words < {MIN_PROFILE_WORDS}"

    def test_technical_specificity(self):
        """Profiles must contain real protocol/tool references."""
        TECH_MARKERS = ["tcp", "udp", "dns", "http", "https", "smb", "ssh", "rdp", "icmp",
                       "port", "payload", "packet", "header", "tunnel", "proxy", "encrypt",
                       "certificate", "handshake", "beacon", "callback", "exfil", "lateral",
                       "mimikatz", "cobalt", "metasploit", "nmap", "wireshark"]
        for p in data:
            text_lower = p["traffic_description"].lower()
            matches = sum(1 for m in TECH_MARKERS if m in text_lower)
            assert matches >= 3, (
                f"[{p['attack_class']}] only {matches} technical markers — "
                f"profiles must reference real protocols, tools, and techniques"
            )
```
