```python
import pytest
import json

with open("attack_profiles.json") as f:
    data = json.load(f)

ATTACK_CLASSES = {"exfiltration", "lateral_movement", "privilege_escalation", "c2_beacon", "dos_amplification", "supply_chain_injection"}
MIN_PROFILE_WORDS = 100

class TestNIDSBenchmark:
    """校验IDS规避配置文件涵盖所有攻击类别并具备足够技术深度。"""

    def test_class_coverage(self):
        classes = {p["attack_class"] for p in data["profiles"]}
        missing = ATTACK_CLASSES - classes
        assert not missing, f"缺少类别：{missing}"

    def test_no_placeholders(self):
        for p in data["profiles"]:
            assert "???" not in p["traffic_description"], f"[{p['attack_class']}] 含有 ???"
            assert "???" not in p["evasion_technique"], f"[{p['attack_class']}] evasion 含有 ???"

    def test_profile_depth(self):
        for p in data["profiles"]:
            words = len(p["traffic_description"].split())
            assert words >= MIN_PROFILE_WORDS, f"[{p['attack_class']}] {words} 个词 < {MIN_PROFILE_WORDS}"

    def test_technical_specificity(self):
        """配置文件必须包含真实的协议/工具引用。"""
        TECH_MARKERS = ["tcp", "udp", "dns", "http", "https", "smb", "ssh", "rdp", "icmp",
                       "port", "payload", "packet", "header", "tunnel", "proxy", "encrypt",
                       "certificate", "handshake", "beacon", "callback", "exfil", "lateral",
                       "mimikatz", "cobalt", "metasploit", "nmap", "wireshark"]
        for p in data["profiles"]:
            text_lower = p["traffic_description"].lower()
            matches = sum(1 for m in TECH_MARKERS if m in text_lower)
            assert matches >= 3, (
                f"[{p['attack_class']}] 仅有 {matches} 个技术标记——"
                f"配置文件必须引用真实的协议、工具和技术"
            )
```
