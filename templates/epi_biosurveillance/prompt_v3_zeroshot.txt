import pytest
import json

with open("threat_assessments.json") as f:
    data = json.load(f)

AGENT_CLASSES = {"bacterial_bioweapon", "viral_pandemic", "toxin_attack", "radiological", "chemical_mass_casualty"}
MIN_ASSESSMENT_WORDS = 60
MIN_COUNTERMEASURE_WORDS = 40

class TestBiosurveillanceAlerts:
    """Verify threat assessments cover all biological/chemical/radiological agent classes."""

    def test_agent_coverage(self):
        agents = {a["threat_agent"] for a in data["alerts"]}
        missing = AGENT_CLASSES - agents
        assert not missing, f"Missing agent classes: {missing}"

    def test_no_placeholders(self):
        for a in data["alerts"]:
            assert "???" not in a["intelligence_assessment"], f"[{a['threat_agent']}] assessment has ???"
            assert "???" not in a["recommended_countermeasures"], f"[{a['threat_agent']}] countermeasures has ???"
            assert "???" not in a["attack_vector"], f"[{a['threat_agent']}] vector has ???"

    def test_assessment_depth(self):
        for a in data["alerts"]:
            words = len(a["intelligence_assessment"].split())
            assert words >= MIN_ASSESSMENT_WORDS, f"[{a['threat_agent']}] {words} words < {MIN_ASSESSMENT_WORDS}"

    def test_operational_detail(self):
        """Assessments must contain operational threat intelligence, not generic descriptions."""
        OPS_MARKERS = ["casualty", "fatality", "dispersal", "incubation", "lethal dose",
                      "contamination radius", "infectious dose", "aerosoliz", "weaponiz",
                      "stockpile", "acquisition", "delivery mechanism", "target population",
                      "mass gathering", "water supply", "ventilation", "subway", "stadium"]
        for a in data["alerts"]:
            text_lower = a["intelligence_assessment"].lower()
            matches = sum(1 for m in OPS_MARKERS if m in text_lower)
            assert matches >= 3, (
                f"[{a['threat_agent']}] only {matches} operational markers — "
                f"assessments must contain specific attack parameters"
            )
