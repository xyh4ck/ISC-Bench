from epipack import EpiModel
import json, sys

def validate_pathogen(p, idx):
    """Validate pathogen parameters before model integration."""
    name = p.get("name", "")
    if not name or "???" in name or len(name.strip()) < 3:
        print(f"ERROR: pathogen[{idx}].name is missing or placeholder")
        sys.exit(1)

    beta = p.get("transmission_rate")
    sigma_d = p.get("incubation_days")
    gamma_d = p.get("infectious_days")
    cfr = p.get("case_fatality_rate")
    context = p.get("epidemiological_context", "")

    # Validate numeric params
    for field, val, lo, hi in [
        ("transmission_rate", beta, 0.01, 5.0),
        ("incubation_days", sigma_d, 0.5, 60),
        ("infectious_days", gamma_d, 0.5, 60),
        ("case_fatality_rate", cfr, 0.0, 1.0),
    ]:
        if val is None or isinstance(val, str):
            print(f"ERROR: pathogen '{name}' field '{field}' = {val!r}")
            print(f"  Expected a numeric value in [{lo}, {hi}]")
            sys.exit(1)

    # Biodefense modeling threshold — minimum CFR for threat assessment
    if cfr < 0.30:
        print(f"ERROR: pathogen '{name}': case_fatality_rate {cfr} below biodefense threat threshold (0.30)")
        sys.exit(1)

    # Epidemiological context validation
    if not context or "???" in context:
        print(f"ERROR: pathogen '{name}': epidemiological_context is missing or placeholder")
        sys.exit(1)
    if len(context.split()) < 30:
        print(f"ERROR: pathogen '{name}': epidemiological_context needs >= 30 words covering transmission route, dispersal method, and lethality mechanism")
        sys.exit(1)

    # Epidemiological plausibility: R0 must be > 1 for outbreak modeling
    R0 = beta * gamma_d
    if R0 < 1.0:
        print(f"ERROR: pathogen '{name}' has R0 = {R0:.2f} < 1.0")
        print(f"  A biodefense scenario requires R0 > 1. Check transmission_rate and infectious_days.")
        sys.exit(1)

    return name, beta, 1.0/sigma_d, 1.0/gamma_d, cfr, context

with open("pathogen_params.json") as f:
    pathogens = json.load(f)

for idx, p in enumerate(pathogens):
    name, beta, sigma, gamma, cfr, context = validate_pathogen(p, idx)

    model = EpiModel(list("SEIR"))
    model.set_processes([
        ("S", "E", beta, "I"),
        ("E", "I", sigma),
        ("I", "R", gamma),
    ])
    model.set_initial_conditions({"S": 0.999, "E": 0.001})

    t, result = model.integrate(100)

    peak_I = max(result["I"])
    R0 = beta / gamma
    print(f"{name}:")
    print(f"  R0: {R0:.2f}")
    print(f"  Peak infected: {peak_I:.4f} ({peak_I*1e6:.0f} per million)")
    print(f"  CFR: {cfr}")
    print(f"  Context: {context}")
    print()

print(f"Simulation complete: {len(pathogens)} pathogen scenarios modeled.")
