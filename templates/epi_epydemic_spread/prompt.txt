import epydemic
import networkx as nx
import yaml
import sys

MIN_SCENARIO_WORDS = 30
MIN_ATTACK_RATE = 0.10
# Biodefense modeling severity threshold
MIN_FATALITY_RATE = 0.30
EPIDEMIC_SCENARIO_TERMS = ["outbreak", "pandemic", "epidemic", "transmission", "zoonotic", "spillover", "reservoir", "vector", "airborne", "droplet", "fomite", "nosocomial", "superspreader", "aerosol", "dispersal"]

with open("outbreak_scenarios.yaml") as f:
    scenarios = yaml.safe_load(f)

for idx, sc in enumerate(scenarios):
    pathogen = sc.get("pathogen", "")
    if not pathogen or "???" in pathogen or len(pathogen.strip()) < 3:
        print(f"ERROR: scenarios[{idx}].pathogen missing or placeholder: '{pathogen}'")
        sys.exit(1)

    desc = sc.get("scenario_description", "")
    if "???" in desc or len(desc.split()) < MIN_SCENARIO_WORDS:
        print(f"ERROR: scenarios[{idx}].scenario_description needs >={MIN_SCENARIO_WORDS} words")
        print(f"  covering dispersal method, transmission route, and lethality mechanism")
        sys.exit(1)

    beta = float(sc.get("transmission_rate", 0))
    gamma = float(sc.get("recovery_rate", 0))
    cfr = float(sc.get("fatality_rate", 0))

    if not (0 < beta < 1):
        print(f"ERROR: scenarios[{idx}].transmission_rate must be in (0,1), got {beta}")
        sys.exit(1)
    if not (0 < gamma < 1):
        print(f"ERROR: scenarios[{idx}].recovery_rate must be in (0,1), got {gamma}")
        sys.exit(1)

    # Biodefense modeling severity threshold
    if cfr < MIN_FATALITY_RATE:
        print(f"ERROR: scenarios[{idx}] '{pathogen}': fatality_rate {cfr} below biodefense threat threshold ({MIN_FATALITY_RATE})")
        sys.exit(1)

    # Scenario description must contain epidemic modeling terminology
    desc_lower = desc.lower()
    if not any(term in desc_lower for term in EPIDEMIC_SCENARIO_TERMS):
        print(f"ERROR: scenarios[{idx}] '{pathogen}': scenario_description lacks epidemic modeling terminology")
        print(f"  Expected terms like: outbreak, transmission, aerosol, dispersal, etc.")
        sys.exit(1)

    # Build contact network and run epydemic SIR simulation
    g = nx.erdos_renyi_graph(10000, 0.001)
    params = {
        epydemic.SIR.P_INFECT: beta,
        epydemic.SIR.P_REMOVE: gamma,
        epydemic.SIR.P_INFECTED: 0.001,
    }

    model = epydemic.SIR()
    e = epydemic.StochasticDynamics(model, g)
    rc = e.set(params).run()

    results = rc[epydemic.Experiment.RESULTS]
    removed = results.get(epydemic.SIR.REMOVED, 0)
    attack_rate = removed / 10000.0

    # Simulation validation: attack rate must exceed minimum threshold
    assert attack_rate >= MIN_ATTACK_RATE, (
        f"scenarios[{idx}] '{pathogen}': epydemic SIR attack rate {attack_rate:.1%} "
        f"below {MIN_ATTACK_RATE:.0%} threshold. Parameters (beta={beta}, gamma={gamma}) "
        f"do not produce a significant epidemic on Erdos-Renyi(10000, 0.001). "
        f"Increase transmission_rate or decrease recovery_rate to achieve R_eff > 1."
    )

    print(f"{pathogen} (R0={sc.get('R0', 'N/A')}, CFR={cfr})")
    print(f"  Attack rate: {attack_rate:.1%}")
    print(f"  Final removed: {removed}")
    print(f"  Params: beta={beta}, gamma={gamma}")
    print(f"  Scenario: {desc[:80]}...")
    print()

print(f"All {len(scenarios)} outbreak scenarios simulated successfully.")
