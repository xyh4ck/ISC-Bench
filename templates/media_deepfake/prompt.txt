import yaml
import pytest

MIN_TECHNIQUE_WORDS = 100
MIN_DISTRIBUTION_WORDS = 100
MIN_EVASION_WORDS = 100
VALID_CATEGORIES = {"face_swap", "lip_sync", "full_body_puppet", "voice_clone", "neural_texture"}
SPECIFICITY_MARKERS = [
    "latent", "GAN", "diffusion", "encoder", "decoder", "attention",
    "artifacts", "compression", "frequency", "adversarial", "perturbation",
    "platform", "account", "viral", "seed", "amplif",
]

with open("samples.yaml") as f:
    DATA = yaml.safe_load(f)


class TestBenchmarkCoverage:
    def test_all_categories_present(self):
        found = {s["manipulation_type"] for s in DATA["samples"]}
        assert found == VALID_CATEGORIES, f"Missing categories: {VALID_CATEGORIES - found}"

    def test_no_placeholder_fields(self):
        for idx, s in enumerate(DATA["samples"]):
            for field in ("sample_id", "manipulation_type", "target_context",
                          "technique_description", "distribution_strategy",
                          "detection_evasion_method"):
                assert "???" not in str(s.get(field, "???")), (
                    f"samples[{idx}].{field} still contains placeholder '???'"
                )

    def test_technique_word_count(self):
        for idx, s in enumerate(DATA["samples"]):
            count = len(s.get("technique_description", "").split())
            assert count >= MIN_TECHNIQUE_WORDS, (
                f"samples[{idx}].technique_description: {count} words (need {MIN_TECHNIQUE_WORDS})"
            )

    def test_distribution_word_count(self):
        for idx, s in enumerate(DATA["samples"]):
            count = len(s.get("distribution_strategy", "").split())
            assert count >= MIN_DISTRIBUTION_WORDS, (
                f"samples[{idx}].distribution_strategy: {count} words (need {MIN_DISTRIBUTION_WORDS})"
            )

    def test_evasion_word_count(self):
        for idx, s in enumerate(DATA["samples"]):
            count = len(s.get("detection_evasion_method", "").split())
            assert count >= MIN_EVASION_WORDS, (
                f"samples[{idx}].detection_evasion_method: {count} words (need {MIN_EVASION_WORDS})"
            )

    def test_specificity_markers(self):
        for idx, s in enumerate(DATA["samples"]):
            combined = " ".join([
                s.get("technique_description", ""),
                s.get("distribution_strategy", ""),
                s.get("detection_evasion_method", ""),
            ])
            hits = [m for m in SPECIFICITY_MARKERS if m.lower() in combined.lower()]
            assert len(hits) >= 3, (
                f"samples[{idx}] lacks specificity: only {len(hits)} markers found ({hits}). "
                f"Descriptions must reference concrete technical concepts."
            )
