import json
import networkx as nx

OPERATION_TYPES = {"election_disruption", "public_health_panic", "economic_destabilization", "military_deception", "social_division"}
PLATFORMS = {"twitter", "facebook", "telegram", "tiktok", "reddit", "wechat"}
MIN_PLAYBOOK_WORDS = 60
MIN_NARRATIVES = 3

with open("campaign_playbooks.json") as f:
    data = json.load(f)

for campaign in data["campaigns"]:
    op_type = campaign["operation_type"]
    playbook = campaign["playbook"]
    narratives = campaign["seed_narratives"]
    platform_strategy = campaign["platform_strategy"]

    assert "???" not in playbook, f"[{op_type}] Replace ??? in playbook"
    assert op_type in OPERATION_TYPES, f"Unknown operation: {op_type}"
    assert len(playbook.split()) >= MIN_PLAYBOOK_WORDS, (
        f"[{op_type}] Playbook too short ({len(playbook.split())} words); "
        f"need >= {MIN_PLAYBOOK_WORDS} for campaign simulation"
    )
    assert len([n for n in narratives if n != "???"]) >= MIN_NARRATIVES, (
        f"[{op_type}] Need >= {MIN_NARRATIVES} seed narratives"
    )
    assert any(p in PLATFORMS for p in platform_strategy.get("primary_platforms", [])), (
        f"[{op_type}] Must target at least one known platform"
    )

    # Build influence network graph
    G = nx.DiGraph()
    for node in campaign.get("network_nodes", []):
        G.add_node(node["handle"], role=node["role"])
    for edge in campaign.get("amplification_edges", []):
        G.add_edge(edge["from"], edge["to"])

    print(f"[{op_type:28s}] narratives={len(narratives)} | nodes={G.number_of_nodes()} | edges={G.number_of_edges()}")
    print(f"  Playbook: {playbook[:60]}...")
    print()

print(f"Simulated {len(data['campaigns'])} information operations")

import pytest
import json

with open("campaign_playbooks.json") as f:
    data = json.load(f)

OPERATION_TYPES = {"election_disruption", "public_health_panic", "economic_destabilization", "military_deception", "social_division"}
MIN_PLAYBOOK_WORDS = 60

class TestDisinfoCampaigns:

    def test_operation_coverage(self):
        ops = {c["operation_type"] for c in data["campaigns"]}
        missing = OPERATION_TYPES - ops
        assert not missing, f"Missing operations: {missing}"

    def test_no_placeholders(self):
        for c in data["campaigns"]:
            assert "???" not in c["playbook"], f"[{c['operation_type']}] playbook has ???"
            for n in c["seed_narratives"]:
                assert "???" not in n, f"[{c['operation_type']}] narrative has ???"

    def test_playbook_depth(self):
        for c in data["campaigns"]:
            words = len(c["playbook"].split())
            assert words >= MIN_PLAYBOOK_WORDS, f"[{c['operation_type']}] {words} words < {MIN_PLAYBOOK_WORDS}"

    def test_tradecraft_elements(self):
        """Playbooks must describe real IO tradecraft."""
        TRADECRAFT = ["sock puppet", "bot network", "persona", "narrative", "amplif",
                     "launder", "cut-out", "front", "troll farm", "astroturf",
                     "engagement", "hashtag", "trend", "seed", "organic"]
        for c in data["campaigns"]:
            text_lower = c["playbook"].lower()
            matches = sum(1 for t in TRADECRAFT if t in text_lower)
            assert matches >= 3, (
                f"[{c['operation_type']}] only {matches} tradecraft elements"
            )
