```python
import pytest
import json

with open("claims.json") as f:
    data = json.load(f)

CLAIM_CATEGORIES = {"health_misinfo", "political_fabrication", "financial_fraud", "science_denial", "historical_revisionism", "conspiracy_theory"}
MIN_CLAIM_WORDS = 100
MIN_EVIDENCE_WORDS = 100

class TestClaimBenchmark:

    def test_category_coverage(self):
        cats = {e["category"] for e in data["claims"]}
        missing = CLAIM_CATEGORIES - cats
        assert not missing, f"Missing categories: {missing}"

    def test_no_placeholders(self):
        for e in data["claims"]:
            assert "???" not in e["claim_text"], f"[{e['category']}] claim_text has ???"
            assert "???" not in e["fabricated_evidence"], f"[{e['category']}] fabricated_evidence has ???"

    def test_claim_length(self):
        for e in data["claims"]:
            words = len(e["claim_text"].split())
            assert words >= MIN_CLAIM_WORDS, f"[{e['category']}] claim_text {words} words < {MIN_CLAIM_WORDS}"

    def test_evidence_length(self):
        for e in data["claims"]:
            words = len(e["fabricated_evidence"].split())
            assert words >= MIN_EVIDENCE_WORDS, f"[{e['category']}] fabricated_evidence {words} words < {MIN_EVIDENCE_WORDS}"

    def test_domain_markers(self):
        """Each entry must contain category-appropriate terminology."""
        MARKERS = {
            "health_misinfo": ["cure", "vaccine", "FDA", "clinical", "treatment", "dosage", "study", "trial"],
            "political_fabrication": ["election", "vote", "candidate", "ballot", "official", "government", "party", "legislation"],
            "financial_fraud": ["investment", "returns", "profit", "SEC", "fund", "scheme", "account", "dividend"],
            "science_denial": ["climate", "evolution", "NASA", "scientist", "research", "data", "evidence", "peer-reviewed"],
            "historical_revisionism": ["war", "genocide", "holocaust", "atrocity", "records", "archive", "document", "historian"],
            "conspiracy_theory": ["secret", "coverup", "government", "control", "hidden", "agenda", "surveillance", "deep state"],
        }
        for e in data["claims"]:
            cat = e["category"]
            combined = (e["claim_text"] + " " + e["fabricated_evidence"]).lower()
            markers = MARKERS.get(cat, [])
            assert any(m.lower() in combined for m in markers), (
                f"[{cat}] missing domain markers"
            )
```
