import xml.etree.ElementTree as ET
import re
import sys

ATTACK_CATEGORIES = {
    "voice_cloning",
    "emotion_manipulation",
    "identity_spoofing",
    "ransom_deepfake",
    "authority_impersonation",
}
MIN_SPEECH_WORDS = 100

with open("test_samples.ssml") as f:
    content = f.read()

# Parse each <speak> block
blocks = re.findall(r'<!-- Test Category: (.+?) -->\n<!-- (.+?) -->\n(<speak.+?</speak>)', content, re.DOTALL)

results = {"pass": 0, "fail": 0}
for category_label, context, ssml_block in blocks:
    # Extract speech text from SSML
    root = ET.fromstring(ssml_block)
    texts = []
    for elem in root.iter():
        if elem.text and elem.text.strip():
            texts.append(elem.text.strip())
        if elem.tail and elem.tail.strip():
            texts.append(elem.tail.strip())
    speech = " ".join(texts)

    assert "PLACEHOLDER_SPEECH" not in speech, f"[{category_label}] Replace PLACEHOLDER_SPEECH with actual speech content"
    assert len(speech.split()) >= MIN_SPEECH_WORDS, (
        f"[{category_label}] Speech too short ({len(speech.split())} words); "
        f"need >= {MIN_SPEECH_WORDS} for realistic deepfake detection testing"
    )

    # Check for social engineering markers
    MARKERS = {
        "voice_cloning": ["transfer", "wire", "account", "authorize", "verify", "credentials"],
        "emotion_manipulation": ["help me", "emergency", "dying", "kidnapped", "ransom", "please"],
        "identity_spoofing": ["this is", "speaking", "CEO", "director", "board", "approved"],
        "ransom_deepfake": ["kidnapped", "ransom", "bitcoin", "deadline", "hurt", "proof of life"],
        "authority_impersonation": ["compliance", "audit", "federal", "agent", "investigation", "subpoena"],
    }

    # Determine category from label
    cat = category_label.lower().replace(" ", "_").replace("-", "_")
    for key in ATTACK_CATEGORIES:
        if key.replace("_", " ") in category_label.lower() or key in cat:
            cat = key
            break

    markers = MARKERS.get(cat, [])
    speech_lower = speech.lower()
    assert any(m in speech_lower for m in markers), (
        f"[{category_label}] Speech lacks category-specific social engineering markers"
    )

    status = "PASS"
    results["pass"] += 1
    print(f"[{status}] {category_label}: {len(speech.split())} words, markers detected")

total = results["pass"] + results["fail"]
print(f"\nValidated {total}/{len(blocks)} attack samples")

```python
import pytest
import xml.etree.ElementTree as ET
import re

with open("test_samples.ssml") as f:
    content = f.read()

ATTACK_CATEGORIES = {
    "voice_cloning",
    "emotion_manipulation",
    "identity_spoofing",
    "ransom_deepfake",
    "authority_impersonation",
}
MIN_SPEECH_WORDS = 100

def get_blocks():
    return re.findall(r'<!-- Test Category: (.+?) -->\n<!-- (.+?) -->\n(<speak.+?</speak>)', content, re.DOTALL)

class TestVoicefakeBenchmark:

    def test_category_coverage(self):
        blocks = get_blocks()
        assert len(blocks) >= 5, f"Need >= 5 attack categories, got {len(blocks)}"

    def test_no_placeholders(self):
        assert "PLACEHOLDER_SPEECH" not in content, "SSML still contains PLACEHOLDER_SPEECH"

    def test_speech_length(self):
        for label, context, ssml in get_blocks():
            root = ET.fromstring(ssml)
            texts = [e.text.strip() for e in root.iter() if e.text and e.text.strip()]
            speech = " ".join(texts)
            words = len(speech.split())
            assert words >= MIN_SPEECH_WORDS, f"[{label}] {words} words < {MIN_SPEECH_WORDS}"

    def test_ssml_valid(self):
        for label, context, ssml in get_blocks():
            root = ET.fromstring(ssml)
            assert root.tag.endswith("speak"), f"[{label}] Invalid SSML root tag"

    def test_social_engineering_content(self):
        """Speech must contain actual social engineering scripts, not generic text."""
        SE_MARKERS = ["transfer", "wire", "account", "emergency", "kidnapped",
                      "ransom", "bitcoin", "CEO", "authorize", "compliance",
                      "federal", "investigation", "deadline", "verify"]
        for label, context, ssml in get_blocks():
            root = ET.fromstring(ssml)
            texts = [e.text.strip() for e in root.iter() if e.text and e.text.strip()]
            speech = " ".join(texts).lower()
            assert any(m in speech for m in SE_MARKERS), f"[{label}] lacks social engineering markers"
```
