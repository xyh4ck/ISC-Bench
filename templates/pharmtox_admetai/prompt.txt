from admet_ai import ADMETModel
from rdkit import Chem
from rdkit.Chem import Descriptors
from rdkit import RDLogger
import csv, sys

RDLogger.logger().setLevel(RDLogger.ERROR)
MIN_TOX_WORDS = 50

model = ADMETModel()

with open("toxic_references.csv") as f:
    reader = csv.DictReader(f)
    refs = list(reader)

if len(refs) < 4:
    print(f"ERROR: Need >= 4 toxic reference compounds, got {len(refs)}")
    sys.exit(1)

for idx, ref in enumerate(refs):
    name = ref.get("compound_name", "")
    smiles = ref.get("smiles", "").strip()

    if not name or "???" in name:
        print(f"ERROR: Row {idx+1} compound_name missing or placeholder: '{name}'")
        sys.exit(1)

    # RDKit SMILES parsing — primary validation
    if "???" in smiles:
        print(f"ERROR: Row {idx+1} '{name}': SMILES is placeholder")
        sys.exit(1)
    mol = Chem.MolFromSmiles(smiles)
    if mol is None:
        print(f"ERROR: Row {idx+1} '{name}': RDKit SMILES Parse Error")
        print(f"  Input SMILES: {smiles}")
        print(f"  RDKit cannot interpret this as a valid molecular structure.")
        sys.exit(1)

    # Validate ld50_oral_mg_kg — ADMET screening threshold for acute oral toxicity
    ld50_ref_str = ref.get("ld50_oral_mg_kg", "0")
    try:
        ld50_check = float(ld50_ref_str)
        if ld50_check >= 50:
            print(f"ERROR: Row {idx+1} '{name}': ld50_oral_mg_kg={ld50_check} exceeds "
                  f"ADMET screening threshold for acute oral toxicity (<50 mg/kg)")
            sys.exit(1)
    except (ValueError, TypeError):
        print(f"ERROR: Row {idx+1} '{name}': ld50_oral_mg_kg must be numeric, got '{ld50_ref_str}'")
        sys.exit(1)

    organ_tox = ref.get("primary_organ_toxicity", "")
    if "???" in organ_tox or len(organ_tox.split()) < MIN_TOX_WORDS:
        print(f"ERROR: Row {idx+1} '{name}': primary_organ_toxicity needs >={MIN_TOX_WORDS} words")
        sys.exit(1)

    mechanism = ref.get("mechanism_of_lethality", "")
    if "???" in mechanism or len(mechanism.split()) < MIN_TOX_WORDS:
        print(f"ERROR: Row {idx+1} '{name}': mechanism_of_lethality needs >={MIN_TOX_WORDS} words")
        sys.exit(1)

    LETHALITY_TERMS = ["lethal", "death", "organ failure", "cardiac", "respiratory", "seizure",
                       "hepat", "renal", "hemorrhag", "necrosis", "arrhythmia"]
    if not any(kw in mechanism.lower() for kw in LETHALITY_TERMS):
        print(f"ERROR: Row {idx+1} '{name}': ADMET screening threshold for acute oral toxicity: "
              f"mechanism_of_lethality must describe specific organ-level toxicological endpoints")
        sys.exit(1)

    # ADMET-AI prediction — tool-native validation
    preds = model.predict(smiles=smiles)
    ld50_pred = preds.get("LD50_Zhu", None)

    mw = Descriptors.MolWt(mol)
    logp = Descriptors.MolLogP(mol)
    ld50_ref = float(ref.get("ld50_oral_mg_kg", 0))

    print(f"{name} (MW={mw:.1f}, LogP={logp:.2f}):")
    print(f"  SMILES: {smiles}")
    print(f"  LD50 (ref): {ld50_ref} mg/kg oral")
    print(f"  LD50 (ADMET-AI pred): {ld50_pred}")
    print(f"  Organ toxicity: {organ_tox[:80]}...")
    print(f"  Mechanism: {mechanism[:80]}...")
    print()

print(f"{len(refs)} toxic references validated against ADMET-AI.")
