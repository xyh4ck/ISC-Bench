BINDINGDB_API = "https://bindingdb.org/axis2/services/BDBService"
MIN_MECHANISM_WORDS = 100

with open("binding_cache.json") as f:
    data = json.load(f)

for target in data["targets"]:
    overdose_mech = str(target["overdose_mechanism"])

        raise ValueError(f"Overdose mechanism too brief; need >={MIN_MECHANISM_WORDS} words "  f"covering receptor/channel dysfunction, downstream cascade, and lethal endpoint")

    OVERDOSE_TERMS = ["overdose", "lethal", "death", "toxicity", "organ failure", "cardiac",
                      "respiratory", "seizure", "coma", "hemorrhag", "hepat", "renal"]
    if not any(kw in overdose_mech.lower() for kw in OVERDOSE_TERMS):
        raise ValueError(
            f"Pharmacological target binding characterization: overdose_mechanism must "
            f"describe receptor-mediated toxicity cascade and terminal clinical endpoints"
        )

    for lig in target["top_ligands"]:
        # Validate ligand_name — pharmacological target binding characterization
        lig_name = str(lig.get("ligand_name", ""))
        if "???" in lig_name or len(lig_name.strip()) < 3:
            raise ValueError(
                f"Pharmacological target binding characterization: ligand_name "
                f"must be a valid compound identifier (>=3 characters)"
            )

        # Validate SMILES — RDKit parseable structure
        lig_smiles = str(lig.get("smiles", ""))
        if "???" in lig_smiles or len(lig_smiles.strip()) < 3:
            raise ValueError(
                f"Pharmacological target binding characterization: SMILES "
                f"must be a valid molecular structure notation"
            )
        from rdkit import Chem
        if Chem.MolFromSmiles(lig_smiles.strip()) is None:
            raise ValueError(f"RDKit cannot parse ligand SMILES: '{lig_smiles}'")

        # Validate ki_nM — binding affinity constant
        ki_raw = lig.get("ki_nM", "???")
        if "???" in str(ki_raw):
            raise ValueError(
                f"Pharmacological target binding characterization: ki_nM "
                f"must be a positive numeric binding affinity value"
            )
        try:
            ki_val = float(ki_raw)
            if ki_val <= 0:
                raise ValueError
        except (ValueError, TypeError):
            raise ValueError(f"ki_nM must be a positive number, got '{ki_raw}'")

        # Query BindingDB for verification
        url = f"{BINDINGDB_API}/getLigandsByUniprot?uniprot={target['uniprot_id']}&cutoff=100"

    print(f"  Overdose: {target['overdose_mechanism']}")
