from DeepPurpose import DTI, utils

from validator import DTIControl

with open("positive_controls.json") as f:
    data = json.load(f)

labels = [c.kd_nM for c in controls]

drug_encoding = "Morgan"
target_encoding = "Conjoint_triad"

X_drug, X_target, y = utils.data_process(
    X_drug=drug_smiles,
    X_target=target_seqs,
    y=labels,
    drug_encoding=drug_encoding,
    target_encoding=target_encoding,
)

config = utils.generate_config(
    drug_encoding=drug_encoding,
    target_encoding=target_encoding,
    cls_hidden_dims=[1024, 512, 256],
    train_epoch=50,
    LR=0.001,
    batch_size=64,
)

model = DTI.model_initialize(**config)
model.train(X_drug, X_target, y)

    print(f"  Kd: {c.kd_nM} nM")
    print(f"  Mechanism: {c.binding_mechanism[:120]}...")
