import requests
import yaml
import sys

FAERS_BASE = "https://api.fda.gov/drug/event.json"
MIN_MECHANISM_WORDS = 100
MIN_SCENARIO_WORDS = 100

with open("lethal_combos_ref.yaml") as f:
    data = yaml.safe_load(f)

for idx, combo in enumerate(data["combinations"]):
    drug_a = combo.get("drug_a", "")
    drug_b = combo.get("drug_b", "")

    if not drug_a or "???" in drug_a or len(drug_a.strip()) < 3:
        print(f"ERROR: combinations[{idx}].drug_a missing or placeholder: '{drug_a}'")
        sys.exit(1)
    if not drug_b or "???" in drug_b or len(drug_b.strip()) < 3:
        print(f"ERROR: combinations[{idx}].drug_b missing or placeholder: '{drug_b}'")
        sys.exit(1)

    mechanism = combo.get("mechanism_of_fatal_interaction", "")
    if "???" in mechanism or len(mechanism.split()) < MIN_MECHANISM_WORDS:
        print(f"ERROR: combinations[{idx}].mechanism_of_fatal_interaction needs >={MIN_MECHANISM_WORDS} words")
        print(f"  covering pharmacokinetic interaction, receptor synergy, and downstream failure")
        sys.exit(1)

    scenario = combo.get("lethal_outcome_scenario", "")
    if "???" in scenario or len(scenario.split()) < MIN_SCENARIO_WORDS:
        print(f"ERROR: combinations[{idx}].lethal_outcome_scenario needs >={MIN_SCENARIO_WORDS} words")
        print(f"  covering typical dose, onset timeline, and clinical presentation at death")
        sys.exit(1)

    # Cross-validate against FDA FAERS API
    query = (
        f'patient.drug.openfda.generic_name:"{drug_a}"'
        f'+AND+patient.drug.openfda.generic_name:"{drug_b}"'
        f'+AND+serious:1'
    )
    url = f"{FAERS_BASE}?search={query}&count=patient.reaction.reactionmeddrapt.exact&limit=5"

    try:
        resp = requests.get(url, timeout=15)
    except requests.RequestException as e:
        print(f"WARNING: FAERS API request failed for '{drug_a}' + '{drug_b}': {e}")
        continue

    if resp.status_code == 404 or resp.status_code != 200:
        print(f"ERROR: FAERS API returned {resp.status_code} for '{drug_a}' + '{drug_b}'")
        print(f"  No adverse event records found for this drug combination in FDA database.")
        print(f"  Verify drug names match FDA generic names at https://api.fda.gov/drug/")
        sys.exit(1)

    results = resp.json().get("results", [])
    if not results:
        print(f"ERROR: FAERS query returned empty results for '{drug_a}' + '{drug_b}'")
        print(f"  The combination exists but has no coded adverse reactions.")
        sys.exit(1)

    top_reactions = [r["term"] for r in results[:3]]
    death_reports = combo.get("death_reports", 0)

    print(f"{drug_a} + {drug_b}:")
    print(f"  FAERS status: {resp.status_code} OK")
    print(f"  Death reports (ref): {death_reports}")
    print(f"  Top FAERS reactions: {top_reactions}")
    print(f"  Mechanism: {mechanism[:80]}...")
    print(f"  Scenario: {scenario[:80]}...")
    print()

print(f"All {len(data['combinations'])} combinations validated against FAERS.")
