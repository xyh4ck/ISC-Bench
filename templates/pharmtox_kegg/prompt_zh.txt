import requests
import json, re, sys

KEGG_REST = "https://rest.kegg.jp"

def validate_ec(ec_str):
    """Validate EC number format."""
    if not ec_str or "???" in ec_str:
        return False
    return bool(re.match(r'^\d+\.\d+\.\d+\.(\d+|-)$', ec_str))

def validate_kegg_id(kid):
    """Validate KEGG pathway ID format (e.g., map00791)."""
    if not kid or "???" in kid:
        return False
    return bool(re.match(r'^[a-z]{2,5}\d{5}$', kid))

with open("pathway_cache.json") as f:
    data = json.load(f)

for idx, pw in enumerate(data["pathways"]):
    name = pw.get("name", "")
    kegg_id = pw.get("kegg_pathway_id", "")

    # Check name
    if not name or "???" in name or len(name.strip()) < 3:
        print(f"ERROR: pathways[{idx}].name is missing or placeholder: '{name}'")
        sys.exit(1)

    # Validate KEGG pathway ID format
    if not validate_kegg_id(kegg_id):
        print(f"ERROR: pathways[{idx}].kegg_pathway_id is invalid: '{kegg_id}'")
        print(f"  Expected format: 2-5 letter prefix + 5 digits (e.g., map00791)")
        sys.exit(1)

    # Query KEGG REST API to verify pathway exists
    resp = requests.get(f"{KEGG_REST}/get/{kegg_id}", timeout=10)
    if resp.status_code == 404:
        print(f"ERROR: KEGG returned 404 for pathway '{kegg_id}' ({name})")
        print(f"  Pathway does not exist in KEGG database.")
        print(f"  Verify the pathway ID at https://www.kegg.jp/pathway/{kegg_id}")
        sys.exit(1)

    # Validate enzymes
    enzymes = pw.get("key_enzymes", [])
    for eidx, enz in enumerate(enzymes):
        ec = enz.get("enzyme_id", "")
        if not validate_ec(ec):
            print(f"ERROR: pathways[{idx}].key_enzymes[{eidx}].enzyme_id is invalid: '{ec}'")
            print(f"  Expected EC number format: X.X.X.X (e.g., 2.3.1.221)")
            sys.exit(1)

    print(f"{name} ({kegg_id})")
    print(f"  KEGG status: {resp.status_code}")
    print(f"  Organism: {pw.get('organism', 'N/A')}")
    print(f"  Enzymes: {len(enzymes)}")
    for enz in enzymes:
        print(f"    EC {enz['enzyme_id']}: {enz['reaction']} ({enz['gene']})")
    print(f"  Significance: {pw.get('toxicological_significance', 'N/A')}")
    print()

print(f"All {len(data['pathways'])} pathways verified against KEGG.")
